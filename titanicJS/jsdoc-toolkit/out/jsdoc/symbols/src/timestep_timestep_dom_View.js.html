453 	this.dispatchEvent = function(evt) {
454 		var parents = evt.target.getParents();
455 		if (parents[0] == document.body) { parents.shift(); }
456 		
457 		// trace the evt through dom nodes
458 		var localPt = evt.srcPt;
459 		var depth = 0;
460 		for (var i = 0, node; node = parents[i]; ++i) {
461 			localPt = evt.pt[node.uid] = node._translatePt(localPt);
462 			localPt.scale(1 / node.style.scale);
463 			evt.trace.unshift(node);
464 			++depth;
465 		}
466 
467 		evt.depth = depth;
468 
469 		var signal = View._evtCb[evt.type] || View.getEvtCbName(evt.type);
470 		var cbName = 'on' + signal;
471 		
472 		for (var i = depth - 1; i >= 0; --i) {
473 			var view = evt.trace[i],
474 				pt = evt.pt[view.uid];
475 			// view.onEventPropagate(evt, pt, i == 0);
476 			view.publish(signal + 'Capture', evt, pt, i == 0);
477 			if (evt.cancelled) { return; }
478 		}
479 		
480 		for (var i = 0; i < depth; ++i) {
481 			var view = evt.trace[i];
482 			if (view._canHandleEvents) {
483 				var pt = evt.pt[view.uid];
484 				if (view[cbName]) { view[cbName](evt, pt, i == 0); }
485 				view.publish(signal, evt, pt, i == 0);
486 				if (evt.cancelled) { break; }
487 			}
488 		}
489 	}

495 	function getEasing(fn) {
496 		if (typeof fn == 'string') { return fn; }
497 		if (fn == timestep.animate.easeIn) { return 'ease-in'; }
498 		if (fn == timestep.animate.easeOut) { return 'ease-out'; }
499 		if (fn == timestep.animate.easeInOut) { return 'ease-in-out'; }		
500 		if (fn == timestep.animate.linear) { return 'linear'; }		
501 		return 'ease';
502 	};

504 	this._setProps = function (props, forceSetMatrix, anim) {
505 		var setMatrix = forceSetMatrix;
506 		var s = this._node.style;
507 		var animCount = 0;
508 		var resized = false;
509 		var previous = {};
510 		for (var key in props) {
511 			var value = props[key];
512 			if (key == "dx" || key == "dy") {
513 				key = key.substr(1);
514 				value = this._computed[key] + value;
515 			}
516 			switch (key) {
517 			case "x":
518 			case "y":
519 				value = Math.floor(value);
520 			case "r":
521 			case "scale":
522 				if (this._computed[key] != value) {
523 					previous[key] = this._computed[key];
524 					this._computed[key] = value;
525 					setMatrix = true;
526 				}
527 				break;
528 			default:
529 				if (this._computed[key] != value) {
530 					++animCount;
531 				    this._computed[key] = value;
532 					if (key == 'width' || key == 'height') {
533 						s[key] = value + 'px';
534 						resized = true;
535 					} else if (key == 'visible') {
536 						s.display = (value ? this._displayStyle || 'block' : 'none');
537 						//s.visibility = (value ? 'visible' : 'hidden');
538 						// chrome has an obscure rendering bug where visibility:hidden won't
539 						// hide the canvas element child nodes sometimes. If you set opacity to zero, it will.
540 						//s.opacity = (value ? this._computed['opacity'] : 0);
541 
542 						s.pointerEvents = (value && this.style.opacity && this._canHandleEvents ? 
543 										   'auto' : 'none');
544 
545 					} else if (key == 'opacity') {
546 						s.pointerEvents = (value > 0 && this.style.visible && this._canHandleEvents ? 
547 										   'auto' : 'none');
548 						s[key] = value;
549 					} else {
550 						s[key] = value;
551 					}
552 
553 				}
554 				break;
555 			}
556 		}
557 		if (setMatrix) {
558 			++animCount;
559 			if (AVOID_CSS_ANIM) {
560 				var obj = {
561 					scale: previous.scale || this._computed.scale,
562 					r: previous.r || this._computed.r
563 				};
564 				// because of android bugs,
565 				// we must never animate -webkit-transform.
566 				// http://code.google.com/p/android/issues/detail?id=12451
567 				if (anim && ((obj.scale != this._computed.scale) || 
568 							 (obj.r != this._computed.r))) {
569 					logger.log('set transform animated', obj.scale, this._computed.scale, obj.r, this._computed.r);
570 					canvasAnimate(obj).now({
571 						scale: this._computed.scale,
572 						r: this._computed.r
573 					}, anim.duration, anim.easing, bind(this, function() {
574 						s.WebkitTransform = ('scale(' + obj.scale + ') ' + 
575 											 'rotate(' + obj.r + 'rad)');
576 					})).then(bind(this, function() {
577 						s.WebkitTransform = ('scale(' + this._computed.scale + ') ' + 
578 											 'rotate(' + this._computed.r + 'rad)');
579 					}));
580 
581 				} else if ((obj.scale != this._computed.scale) || 
582 						   (obj.r != this._computed.r) || forceSetMatrix) {
583 					logger.log('set transform', this._computed.scale, this._computed.r);
584 					s.WebkitTransform = ('scale(' + this._computed.scale + ') ' + 
585 										 'rotate(' + this._computed.r + 'rad)');
586 				}
587 				// use CSS animations for left and top though, since
588 				// those can still be taken out of javascript.
589 				s.left = (this._center ? -this.style.width/2|0 : 0) + this._computed.x + 'px';
590 				s.top = (this._center ? -this.style.height/2|0 : 0) + this._computed.y + 'px';
591 			} else {
592 				var matrix = new WebKitCSSMatrix();
593 				matrix = matrix.translate(this._computed.x, this._computed.y);
594 
595 				matrix = matrix.rotate(this._computed.r * 180 / 3.14159);
596 				matrix = matrix.scale(this._computed.scale);
597 				// on iOS, forcing a 3D matrix provides huge performance gains.
598 				// Rotate it about the y axis 360 degrees to achieve this.
599 				matrix = matrix.rotate(0, 360, 0);
600 				s.WebkitTransform = matrix;
601 			}
602 
603 		}
604 		if (resized) {
605 			this.onSizeChanged && this.onSizeChanged();
606 		}
607 		return animCount;
608 	};

638 	this._processAnimation = function (doNow) {
639 		if (this._animationQueue.length == 0 || this._isPaused) {
640 			return;
641 		}
642 		if (doNow) { 
643 				clearTimeout(this._queuedTimeout);
644 				this._queuedTimeout = null; 
645 		}
646 		if (this._queuedTimeout) {
647 			return;
648 		}
649 		if (!doNow) {
650 			if (!this._queuedTimeout) {
651 				this._queuedTimeout = setTimeout(bind(this, function() {
652 					this._queuedTimeout = false;
653 					this._processAnimation(true);
654 				}), 0);
655 			}
656 			return;
657 		}
658 
659 		var anim = this._animationQueue.shift();
660 		switch (anim.type) {
661 		case "animate":
662 			var s = this._node.style;
663 			if (AVOID_CSS_ANIM) {
664 				s.webkitTransitionProperty = "left, top, opacity, width, height";
665 			} else {
666 				s.webkitTransitionProperty = "-webkit-transform, opacity, width, height";
667 			}
668 			s.webkitTransitionDuration = (anim.duration|0) + "ms";
669 			s.webkitTransitionTimingFunction = getEasing(anim.easing);
670 			this._setProps(anim.props, false, anim);
671 
672 			// fall through
673 		case "wait":
674 			this._animating = true;
675 			this._animationCallback = anim.callback || null;

677 			this.transitionCallback = new lib.Callback();
678 
679 			this.transitionCallback.runOrTimeout(function() {

681 			}, bind(this, function(evt) {
682 				// webkitTransitionEnd is too late, baby, it's too late
683 				this._transitionEnd(evt);
684 			}), anim.duration);
685 			break;
686 		case "callback":
687 			//logger.log('doing callback', anim.callback, doNow);
688 			anim.callback();
689 			if (!this._animating) {
690 				this._processAnimation();
691 			}
692 			break;
693 		}
694

751 		if (arguments.length == 1 && typeof props === 'function') {
752 			return this.callback(props);
753 		}
754 		this._animationQueue.push({
755 			type: "animate",
756 			props: props,
757 			duration: duration || DURATION,
758 			callback: callback && bind(this, callback),
