1 "use import";
  2 
  3 import .device;
  4 import lib.Enum;
  5 
  6 var mobile = new lib.Enum('ipad', 'iphone', 'ios', 'nexus');
  7 var mobileBrowser = new lib.Enum('iphone-browser', 'ipad-browser', 'nexus-s-browser');
  8 
  9 exports.simulate = function(params) {
 10 	var deviceName = params.name.toLowerCase();
 11 	
 12 	// can only simulate in the browser
 13 	if (device.name != 'browser' || deviceName == 'browser') { return; }
 14 	device.simulating = params;
 15 	device.simulatingMobileNative = deviceName in mobile;
 16 	device.simulatingMobileBrowser = deviceName in mobileBrowser;
 17 	if (device.simulatingMobileNative) {
 18 		device.useDOM = false;
 19 	}
 20 }
