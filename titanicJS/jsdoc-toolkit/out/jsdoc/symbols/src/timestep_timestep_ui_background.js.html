1 "use import";
  2 
  3 import lib.Callback;
  4 import timestep.Image;
  5 
  6 exports.BackgroundImage = Class(function() {
  7 	this.init = function(opts) {
  8 		this._cb = new lib.Callback();
  9 		
 10 		this._offsetX = opts.offsetX || 0;
 11 		this._offsetY = opts.offsetY || 0;
 12 		this._states = {};
 13 		
 14 		this.image = new timestep.Image({
 15 			url: opts.url
 16 		}).doOnLoad(this, function(url, width, height) {
 17 			this.leftWidth = opts.leftWidth || 0;
 18 			this.rightWidth = opts.rightWidth || 0;
 19 			this.centerWidth = opts.centerWidth || ((opts.width || width) - this.leftWidth - this.rightWidth);
 20 
 21 			this.topHeight = opts.topHeight || 0;
 22 			this.bottomHeight = opts.bottomHeight || 0;
 23 			this.centerHeight = opts.centerHeight || ((opts.height || height) - this.topHeight - this.bottomHeight);
 24 			
 25 			this._cb.fire();
 26 		});
 27 	}
 28 	
 29 	this.doOnLoad = function() { this._cb.forward(arguments); }
 30 
 31 	this.getWidth = function() {
 32 		return this.topWidth + this.bottomWidth + this.centerWidth;
 33 	}
 34 	
 35 	this.getHeight = function() {
 36 		return this.topHeight + this.bottomHeight + this.centerHeight;
 37 	}
 38 	
 39 	this.addState = function(key, x, y) {
 40 		this._states[key] = {x: x, y: y};
 41 	}
 42 	
 43 	this.setState = function(key) {
 44 		var state = this._states[key];
 45 		if (state) {
 46 			this._offsetX = this._states[key].x;
 47 			this._offsetY = this._states[key].y;
 48 		}
 49 	}
 50 	
 51 	this.setOffset = function(x, y) {
 52 		this._offsetX = x;
 53 		this._offsetY = y;
 54 	}
 55 	
 56 	this.render = function(ctx, width, height) {
 57 		var x0 = this._offsetX,
 58 			x1 = this.leftWidth,
 59 			x2 = this.centerWidth,
 60 			x3 = this.rightWidth,
 61 			y0 = this._offsetY,
 62 			y1 = this.topHeight,
 63 			y2 = this.centerHeight,
 64 			y3 = this.bottomHeight,
 65 			img = this.image.getSource();
 66 
 67 		if (width == undefined) { width = img.width; }
 68 		if (height == undefined) { height = img.height; }
 69 
 70 		if (y1) {
 71 			if (x1) {
 72 				ctx.drawImage(img,
 73 					x0, y0, x1, y1,
 74 					0, 0, x1, y1);
 75 			}
 76 			
 77 			ctx.drawImage(img,
 78 					x0 + x1, y0, x2, y1,
 79 					x1, 0, width - x1 - x3, y1);
 80 			
 81 			if (x3) {
 82 				ctx.drawImage(img,
 83 					x0 + x1 + x2, y0, x3, y1,
 84 					width - x3, 0, x3, y1);
 85 			}
 86 		}
 87 		
 88 		var h = height - y1 - y3;
 89 		
 90 		if (x1) {
 91 			ctx.drawImage(img,
 92 				x0, y0 + y1, x1, y2,
 93 				0, y1, x1, h);
 94 		}
 95 		
 96 		ctx.drawImage(img, 
 97 				x0 + x1, y0 + y1, x2, y2,
 98 				x1, y1, width - x1 - x3, h);
 99 		
100 		if (x3) {
101 			ctx.drawImage(img,
102 				x0 + x1 + x2, y0 + y1, x3, y2,
103 				width - x3, y1, x3, h);
104 		}
105 		
106 		if (y3) {
107 			var y = height - y3;
108 			if (x1) {
109 				ctx.drawImage(img,
110 					x0, y0 + y1 + y2, x1, y3,
111 					0, y, x1, y3);
112 			}
113 			
114 			ctx.drawImage(img, x0 + x1, y0 + y1 + y2, x2, y3, x1, y, width - x1 - x3, y3);
115 			
116 			if (x3) {
117 				ctx.drawImage(img, x0 + x1 + x2, y0 + y1 + y2, x3, y3, width - x3, y, x3, y3);
118 			}
119 		}
120 	}
121 });
