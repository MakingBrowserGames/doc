  2 	this.init = function(rgba) {

 37 	var poundParser = /#([0-9,a-f,A-F]{1,2})([0-9,a-f,A-F]{1,2})([0-9,a-f,A-F]{1,2})/;

 39 	function hexToRGB(a) {
 40 		a = '0x' + a.slice(1).replace(a.length < 5 && /./g, '$&$&');
 41 		return [a >> 16, a >> 8 & 255, a & 255];
 42 	}
 43 
 44 	this.parse = function(str) {
 45 		var match = str.match(rgbParser);
 46 		if (match) {
 47 			this.r = parseInt(match[1]) || 0;
 48 			this.g = parseInt(match[2]) || 0;
 49 			this.b = parseInt(match[3]) || 0;
 50 			if (4 in match) {
 51 				var a = parseFloat(match[4]);
 52 				this.a = isNaN(a) ? 1 : a;
 53 			} else {
 54 				this.a = 1;
 55 			}
 56 		} else {
 57 			
 58 			this.a = 1;
 59 
 60 			if (str.length == 9) {
 61 				str = str.substring(0, 7);
 62 				this.a = '0x' + str.substring(7, 9) | 0;
 63 			}
 64 
 65 			var match = hexToRGB(str);
 66 			if (match) {
 67 				this.r = match[0];
 68 				this.g = match[1];
 69 				this.b = match[2];
 70 			}
 71 		}
 72 	};
 73 	
 74 	this.toString = function() {
 75 		return 'rgba(' + this.r + ',' + this.g + ',' + this.b + ',' + this.a + ')';
 76 	}
 77 });
