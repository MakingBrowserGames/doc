1 "use import";
  2 
  3 import math2D.Rect as Rect;
  4 import math2D.intersect as intersect;
  5 
  6 import ..View;
  7 import ..device;
  8 
  9 var Canvas = device.get('Canvas');
 10 
 11 /**
 12  * @extends timestep.View
 13  */
 14 var ScrollBuffer = exports = Class(View, function(supr) {
 15 
 16 	// the top-left tile in the buffer
 17 	this._tileX = 0;
 18 	this._tileY = 0;
 19 
 20 	// the offset within the top-left tile where the screen rendering starts
 21 	// (the location of the screen origin within the buffer)
 22 	this._offsetX = 0;
 23 	this._offsetY = 0;
 24 
 25 	// the tile index where the origin is
 26 	this._bufferOriginX = 0;
 27 	this._bufferOriginY = 0;
 28 
 29 	this.init = function(opts) {
 30 		supr(this, 'init', arguments);
 31 
 32 		this.setSource(opts.source);
 33 		this.setTileSize(opts.tileSize);
 34 		this.resize(opts.width, opts.height);
 35 		this.setSmoothScrolling('smoothScrolling' in opts ? opts.smoothScrolling : true);
 36 	}
 37 
 38 	this.setSmoothScrolling = function(smoothScrollingEnabled) {
 39 		this._smoothScrolling = smoothScrollingEnabled;
 40 		if (smoothScrollingEnabled) {
 41 			this._scrollBuffer = new Canvas({width: this._width + 1, height: this._height + 1});
 42 		}
 43 	}
 44 
 45 	this.setSource = function(source) {
 46 		this._source = source; // TODO update buffer if changed?
 47 	}
 48 
 49 	this.setTileSize = function(tileWidth, tileHeight) {
 50 		this._tileWidth = tileWidth;
 51 		this._tileHeight = tileHeight || tileWidth;
 52 	}
 53 
 54 	// warning -- currently will reset the scroll offset to (0, 0)
 55 	// TODO: fix this to preserve scroll offsets?
 56 	this.resize = function(width, height) {
 57 		this._width = width;
 58 		this._height = height;
 59 
 60 		// the buffer size is at minimum 1 more tile size than the width/height of the viewable area (1 <= size < 2)
 61 		this._numCols = Math.ceil(width / this._tileWidth) + 1;
 62 		this._numRows = Math.ceil(height / this._tileHeight) + 1;
 63 		this._bufferWidth = this._numCols * this._tileWidth;
 64 		this._bufferHeight = this._numRows * this._tileHeight;
 65 		
 66 		// initialize the offscreen buffer
 67 		this._buffer = new Canvas({width: this._bufferWidth, height: this._bufferHeight});
 68 		this._bufferCtx = this._buffer.getContext('2d');
 69 
 70 		// initialize the native buffer
 71 		if (device.isMobile || device.simulatingMobileNative) {
 72 			this._backBuffer = new Canvas({width: this._bufferWidth, height: this._bufferHeight});
 73 			this._backBufferCtx = this._backBuffer.getContext('2d');
 74 		}
 75 
 76 		if (this._smoothScrolling) {
 77 			// resize the smooth scrolling buffer
 78 			this.setSmoothScrolling(this._smoothScrolling);
 79 		}
 80 
 81 		// update buffer
 82 		this.fillAtZero();
 83 	}
 84 
 85 	this.fillAtZero = function() {
 86 		this._source(this._bufferCtx, 0, 0, this._numCols, this._numRows, 0, 0);
 87 		this._save();
 88 	}
 89 
 90 	this._save = function() {
 91 		if (this._backBufferCtx) {
 92 			this._backBufferCtx.clear();
 93 			this._backBufferCtx.drawImage(this._buffer, 0, 0);
 94 		}
 95 	}
 96 
 97 	this._restore = function() {
 98 		if (this._backBufferCtx) {
 99 			this._bufferCtx.clear();
100 			this._bufferCtx.drawImage(this._backBuffer, 0, 0);
101 		}
102 	}
103 
104 	this.scrollBy = function(dx, dy) {
105 		this._restore();
106 
107 		this._scrollHorizontal(dx);
108 		this._scrollVertical(dy);
109 
110 		this._save();
111 	}
112 
113 	this.getOffsetX = function() { return this._tileX * this._tileWidth + this._offsetX; }
114 	this.getOffsetY = function() { return this._tileY * this._tileHeight + this._offsetY; }
115 
116 	this._copyBuffer = function(ctx, sx, sy, sw, sh, dx, dy) {
117 		if (sw <= 0 || sh <= 0) { return; }
118 
119 		ctx.drawImage(this._buffer, sx, sy, sw, sh, dx, dy, sw, sh);
120 	}
121 
122 	this._scrollVertical = function(value) {
123 		this._offsetY += value;
124 
125 		// cycle the buffer until the _offsetX value is within [0, this._tileWidth)
126 		var ctx = this._bufferCtx;
127 		while (this._offsetY < 0) {
128 			// disable negative offset?
129 			// if (this._tileY > 0) {
130 				this._offsetY += this._tileHeight;
131 				this._bufferOriginY--;
132 				this._tileY--;
133 				if (this._bufferOriginY < 0) { this._bufferOriginY += this._numRows; }
134 			// } else {
135 			// 	this._offsetY = 0;
136 			// 	break;
137 			// }
138 			
139 			this._source(ctx,
140 				0, this._bufferOriginY, this._bufferOriginX, this._bufferOriginY,
141 				this._tileX + this._numCols - this._bufferOriginX, this._tileY);
142 			
143 			this._source(ctx,
144 				this._bufferOriginX, this._bufferOriginY, this._numCols, this._bufferOriginY + 1,
145 				this._tileX, this._tileY);
146 		}
147 
148 		while (this._offsetY >= this._tileHeight) {
149 			this._source(ctx,
150 				0, this._bufferOriginY, this._bufferOriginX, this._bufferOriginY,
151 				this._tileX + this._numCols - this._bufferOriginX, this._numRows + this._tileY);
152 			
153 			this._source(ctx,
154 				this._bufferOriginX, this._bufferOriginY, this._numCols, this._bufferOriginY + 1,
155 				this._tileX, this._numRows + this._tileY);
156 			
157 			this._offsetY -= this._tileHeight;
158 			this._bufferOriginY++;
159 			this._tileY++;
160 			if (this._bufferOriginY >= this._numRows) { this._bufferOriginY -= this._numRows; }
161 		}
162 	}
163 
164 	this._scrollHorizontal = function(value) {
165 		this._offsetX += value;
166 
167 		// cycle the buffer until the _offsetX value is within [0, this._tileWidth)
168 		var ctx = this._bufferCtx;
169 		while (this._offsetX < 0) {
170 			// if (this._tileX > 0) {
171 				this._offsetX += this._tileWidth;
172 				this._bufferOriginX--;
173 				this._tileX--;
174 				if (this._bufferOriginX < 0) { this._bufferOriginX += this._numCols; }
175 			// } else {
176 			// 	this._offsetX = 0;
177 			// 	break;
178 			// }
179 
180 			this._source(ctx,
181 				this._bufferOriginX, 0, this._bufferOriginX + 1, this._numRows,
182 				this._tileX, this._tileY + this._numRows - this._bufferOriginY);
183 			
184 			this._source(ctx,
185 				this._bufferOriginX, this._bufferOriginY, this._bufferOriginX + 1, this._numRows,
186 				this._tileX, this._tileY);
187 		}
188 
189 		while (this._offsetX >= this._tileWidth) {
190 			this._source(ctx,
191 				this._bufferOriginX, 0, this._bufferOriginX + 1, this._bufferOriginY + 1,
192 				this._numCols + this._tileX, this._tileY + this._numRows - this._bufferOriginY);
193 			
194 			this._source(ctx,
195 				this._bufferOriginX, this._bufferOriginY, this._bufferOriginX + 1, this._numRows,
196 				this._numCols + this._tileX, this._tileY);
197 			
198 			this._offsetX -= this._tileWidth;
199 			this._bufferOriginX++;
200 			this._tileX++;
201 			
202 			if (this._bufferOriginX >= this._numCols) { this._bufferOriginX -= this._numCols; }
203 		}
204 	}
205 
206 	this.refreshRegion = function (x, y, w, h) {
207 
208 		this._restore();
209 
210 		// make sure to clear the smooth scrolling buffer offset
211 		// since we want to recreate the scrolling buffer even if it
212 		// hasn't moved
213 		this._sbOffsetX = null;
214 
215 		// compute the range of affected tiles
216 		var tileX = Math.floor(x / this._tileWidth);
217 		var numCols = Math.ceil((w + x % this._tileWidth) / this._tileWidth);
218 		var tileY = Math.floor(y / this._tileHeight);
219 		var numRows = Math.ceil((h + y % this._tileHeight) / this._tileHeight);
220 
221 		// compute the overlap of buffer tiles with affected tiles
222 		// and return if there is no overlap
223 		var rx = Math.max(tileX, this._tileX);
224 		var ry = Math.max(tileY, this._tileY);
225 
226 		var redrawRect = new Rect({
227 			x: rx,
228 			y: ry,
229 			width: Math.min((tileX + numCols || Infinity), this._tileX + this._numCols) - rx,
230 			height: Math.min((tileY + numRows || Infinity), this._tileY + this._numRows) - ry
231 		});
232 
233 		if (redrawRect.width <= 0 || redrawRect.height <= 0) { return; }
234 
235 		var ctx = this._bufferCtx;
236 		this.getQuadrants().forEach(function(quadrant) {
237 			var rect = intersect.util.rectAndRect(quadrant, redrawRect);
238 			if (rect) {
239 				// translate intersection rectangle to the quadrant's origin (bx, by)
240 				// to figure out where to draw to in the buffer
241 				var startX = quadrant.bx + (rect.x - quadrant.x);
242 				var startY = quadrant.by + (rect.y - quadrant.y);
243 
244 				// the actual scroll offset to sample from
245 				var offsetX = rect.x;
246 				var offsetY = rect.y;
247 
248 				// draw
249 				this._source(ctx,
250 					startX, startY, startX + rect.width, startY + rect.height,
251 					offsetX, offsetY);
252 			}
253 		}, this);
254 
255 		this._save();
256 	}
257 
258 	// returns the quadrants
259 	//  - x, y, width, height are in screen coordinates
260 	//  - bx, by are the absolute coordinates of the quadrant inside the buffer
261 	this.getQuadrants = function() {
262 		var width2 = this._numCols - this._bufferOriginX;
263 		var height2 = this._numRows - this._bufferOriginY;
264 
265 		if ((this._bufferOriginX === 0) && (this._bufferOriginY === 0)) {
266 			return [{
267 				bx: 0, by: 0,
268 				x: this._tileX, y: this._tileY, width: this._numCols, height: this._numRows
269 			}];
270 		} else if (this._bufferOriginY === 0) {
271 			return [
272 				// left-side of the screen (right-half of buffer)
273 				{bx: 0, by: 0,
274 					x: this._tileX + width2, y: this._tileY, width: this._bufferOriginX, height: this._numRows},
275 
276 				// right-side of the screen (left-half of buffer)
277 				{bx: this._bufferOriginX, by: 0,
278 					x: this._tileX, y: this._tileY, width: width2, height: this._numRows}
279 			];
280 		} else if (this._bufferOriginX == 0) {
281 			return [
282 				// top-half of screen (bottom-half of buffer)
283 				{bx: 0, by: this._bufferOriginY,
284 					x: this._tileX, y: this._tileY, width: this._numCols, height: height2},
285 
286 				// bottom-half of screen (top-half of buffer)
287 				{bx: 0, by: 0,
288 					x: this._tileX, y: this._tileY + height2, width: this._numCols, height: this._bufferOriginY}
289 			];
290 		} else {
291 			// TODO: untested
292 			return [
293 				{bx: this._bufferOriginX, by: this._bufferOriginY,
294 					x: this._tileX, y: this._tileY, width: width2, height: height2},
295 				{bx: this._bufferOriginX, by: 0,
296 					x: this._tileX, y: this._tileY + height2, width: width2, height: this._bufferOriginY},
297 				{bx: 0, by: this._bufferOriginY,
298 					x: this._tileX + width2, y: this._tileY, width: this._numCols, height: height2},
299 				{bx: 0, by: 0,
300 					x: this._tileX + width2, y: this._tileY + height2, width: this._bufferOriginX, height: this._bufferOriginY}
301 			];
302 		}
303 	}
304 
305 	this.repaintAll = function() {
306 		this.refreshRegion(-Infinity, -Infinity, Infinity, Infinity);
307 	}
308 
309 	this.render = function(ctx) {
310 		var w = this._width;
311 		var h = this._height;
312 
313 		if (this._smoothScrolling) {
314 			if (this._sbOffsetX != this._offsetX || this._sbOffsetY != this._offsetY) {
315 				this._sbOffsetX = this._offsetX;
316 				this._sbOffsetY = this._offsetY;
317 				var scrollCtx = this._scrollBuffer.getContext('2d');
318 				scrollCtx.clear();
319 				this._drawBuffer(scrollCtx, w + 1, h + 1);
320 			}
321 
322 			ctx.drawImage(this._scrollBuffer, this._offsetX % 1, this._offsetY % 1, w, h, 0, 0, w, h);
323 		} else {
324 			this._drawBuffer(ctx, w, h);
325 		}
326 	}
327 
328 	this._drawBuffer = function(ctx, width, height) {
329 		var sizeX = this._bufferOriginX * this._tileWidth,
330 			sizeY = this._bufferOriginY * this._tileHeight,
331 			sx    =   0,
332 			sy    =   0,
333 			sw    = this._width,
334 			sh    = this._height,
335 			dx    =   0,
336 			dy    =   0,
337 			offsetX = Math.floor(this._offsetX),
338 			offsetY = Math.floor(this._offsetY);
339 		
340 		if ((this._bufferOriginX === 0) && (this._bufferOriginY === 0)) {
341 			// if the origin of the buffer is at the top left, we can just blit the buffer straight to the screen!
342 			this._copyBuffer(ctx, offsetX, offsetY, sw, sh, dx, dy);
343 
344 		} else if (this._bufferOriginY === 0) {
345 			// if the buffer origin is at the top, but we have a vertical split, render the left and the right pieces
346 			sx = offsetX + sizeX;
347 			sy = offsetY;
348 			sw = this._bufferWidth - offsetX - sizeX;
349 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
350 
351 			sx = 0;
352 			sy = offsetY;
353 			sw = sizeX + offsetX - this._tileWidth;
354 			dx = this._bufferWidth - offsetX - sizeX;
355 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
356 		} else if (this._bufferOriginX === 0) {
357 			// if the buffer origin is at the left, but we have a horizontal split, render the top and bottom pieces
358 			// first, the bottom part of the buffer which is the top of the composite image
359 			sx = offsetX;
360 			sy = offsetY + sizeY;
361 			sh = this._bufferHeight - offsetY - sizeY;
362 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
363 
364 			// second, the top part of the buffer, which joins the bottom of the buffer at the bottom to complete the image
365 			sx = offsetX;
366 			sy = 0;
367 			sh = sizeY + offsetY - this._tileHeight;
368 			dy = this._bufferHeight - offsetY - sizeY;
369 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
370 		} else {
371 			// Copy right bottom to left top...
372 			sx = offsetX + sizeX;
373 			sy = offsetY + sizeY;
374 			sw = this._bufferWidth - offsetX - sizeX;
375 			sh = this._bufferHeight - offsetY - sizeY;
376 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
377 
378 			// Copy left top to right bottom...
379 			sx = 0;
380 			sy = 0;
381 			sw = offsetX + sizeX - this._tileWidth;
382 			sh = offsetY + sizeY - this._tileHeight;
383 			dx = this._bufferWidth - offsetX - sizeX;
384 			dy = this._bufferHeight - offsetY - sizeY;
385 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
386 
387 			// Copy right top to left bottom...
388 			sx = offsetX + sizeX;
389 			sy = 0;
390 			sw = this._bufferWidth - offsetX - sizeX;
391 			sh = offsetY + sizeY - this._tileHeight;
392 			dx = 0;
393 			dy = this._bufferHeight - offsetY - sizeY;
394 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
395 
396 			// Copy right top to left bottom...
397 			sx = 0;
398 			sy = offsetY + sizeY;
399 			sw = offsetX + sizeX - this._tileWidth;
400 			sh = this._bufferHeight - offsetY - sizeY;
401 			dx = this._bufferWidth - offsetX - sizeX;
402 			dy = 0;
403 			this._copyBuffer(ctx, sx, sy, sw, sh, dx, dy);
404 		}
405 	}
