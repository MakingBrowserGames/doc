  3 var Node = function (index, x, y, parent) {
  4 	this.index = index;
  5 	this.x = x;
  6 	this.y = y;
  7 	this.parent = parent || null;
  8 	this.d = 0;
  9 	this.f = 0;
 10 	this.g = 0;
 11 };
 12 
 13 Node.prototype.getDistance = function (node) {
 14 	return (
 15 		Math.abs(this.x - node.x)
 16 		+ Math.abs(this.y - node.y)
 17 	);
 18 };
 19 
 20 Node.prototype.getPathList = function () {
 21 	var node = this;
 22 	var path = [];
 23 	do {
 24 		path.push([node.x, node.y]);
 25 	} while (node = node.parent);
 26 
 27 	// ...and flip it so it's in the right order
 28 	path.reverse();
 29 
 30 	return path;
 31 };
 32 
 33 var Map = function (grid) {
 34 	this.grid = grid;
 35 	this.width = grid[0].length;
 36 	this.height = grid.length;
 37 };
 38 
 39 Map.prototype.isWalkable = function (x, y) {
 40 	if (
 41 		x < 0
 42 		|| y < 0
 43 		|| x >= this.width
 44 		|| y >= this.height
 45 	) {
 46 		// Outside the map. Totally NOT walkable!
 47 		return false;
 48 	}
 49 	return (!this.grid[y][x]);
 50 };
 51 
 52 Map.prototype.makeNode = function (x, y, parent) {
 53 	return new Node(
 54 		(x + (y * this.width)),
 55 		x, y, parent
 56 	);
 57 };
 58 
 59 Map.prototype.getAdjacent = function (n) {
 60 	// Threre's probably a more elegant way to write this function...
 61 	var a = [];
 62 	// North
 63 	if (this.isWalkable(n.x, n.y - 1)) {
 64 		a.push(this.makeNode(n.x, n.y - 1, n));
 65 	}
 66 	// South
 67 	if (this.isWalkable(n.x, n.y + 1)) {
 68 		a.push(this.makeNode(n.x, n.y + 1, n));
 69 	}
 70 	// West
 71 	if (this.isWalkable(n.x - 1, n.y)) {
 72 		a.push(this.makeNode(n.x - 1, n.y, n));
 73 	}
 74 	// East
 75 	if (this.isWalkable(n.x + 1, n.y)) {
 76 		a.push(this.makeNode(n.x + 1, n.y, n));
 77 	}
 78 	return a;
 79 };

 81 exports = {};
 82 
 83 exports.findPath = function (grid, startX, startY, goalX, goalY) {
 84 
 85 	// The search area
 86 	var map = new Map(grid);
 87 
 88 	// The start and goal nodes
 89 	var start = map.makeNode(startX, startY);
 90 	var goal = map.makeNode(goalX, goalY);
 91 
 92 	// List to keep track of open and closed nodes
 93 	var openList = [start];
 94 	var closedList = [];
 95 	closedList[start.index] = true;
 96 
 97 	// Track closest to use as best guess if no path is found
 98 	var closestNode = start;
 99 	closestNode.d = closestNode.getDistance(goal);
100 
101 	var len = 0;
102 	while (len = openList.length) {
103 
104 		var minF = {
105 			index: -1,
106 			f: Infinity
107 		};
108 
109 		// Find the node on the open list
110 		// with the lowest F score
111 		for (var i = 0; i < len; ++i) {
112 			if (openList[i].f < minF.f) {
113 				minF.f = openList[i].f;
114 				minF.index = i;
115 			}
116 		}
117 
118 		// Remove this node from the open list
119 		var node = openList.splice(minF.index, 1)[0];
120 
121 		// Check if this node is the closest node so far
122 		if (node.d < closestNode.d) {
123 			closestNode = node;
124 		} else if (node.d == closestNode.d) {
125 			// If distance is the same, use node with
126 			// least path so far
127 			if (node.g < closestNode.g) {
128 				closestNode = node;
129 			}
130 		}
131 
132 		if (node.index === goal.index) {
133 
134 			// Hooray! We found the goal node!
135 			return node.getPathList();
136 
137 		} else {
138 
139 			// Haven't found the goal node yet...
140 
141 			// Get adjacent nodes
142 			var adjacent = map.getAdjacent(node);
143 
144 			// Calculate values for adjacent nodes
145 			// and add them to the open list if
146 			// they aren't already closed
147 			for (var i = 0, j = adjacent.length; i < j; ++i) {
148 				var n = adjacent[i];
149 				if (!closedList[n.index]) {
150 					n.d = n.getDistance(goal);
151 					n.g = (node.g + n.getDistance(node));
152 					n.f = (n.g + n.d);
153 					openList.push(n);
154 					closedList[n.index] = true;
155 				}
156 			}
157 
158 		}
159 
160 	}
161 
162 	// We didn't find a path to the goal,
163 	// so return the closest path
164 	return closestNode.getPathList();
165 
166 };
