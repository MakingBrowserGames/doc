1 "use import";
  2 
  3 // Defines a pure JS implementation for the remainder of 
  4 // the View class.  For web browsers and native if native 
  5 // views are disabled.
  6 
  7 exports.install = function() {
  8 	import ..View;
  9 	var proto = View.prototype;
 10 
 11 	import .ViewStyle;
 12 	proto._createStyle = function() {
 13 		this._subviews = [];
 14 		this.style = new ViewStyle(this);
 15 	}
 16 	
 17 
 18 	proto.getSuperview = function() { return this._superview; }
 19 
 20 	var ADD_COUNTER = 900000;
 21 	
 22 	proto.getSubviews = function() {
 23 		return this._subviews;
 24 	}
 25 
 26 	proto.addSubview = function(view) {
 27 		var superview = view._superview;
 28 		if (superview == this) { return; }
 29 		if (superview) { superview.removeSubview(view); }
 30 
 31 		var n = this._subviews.length;
 32 		this._subviews[n] = view;
 33 
 34 		view._superview = this;
 35 		view.style.setAddedAt(++ADD_COUNTER);
 36 		view.needsReflow();
 37 		if (n && view.__sortKey < this._subviews[n - 1].__sortKey) {
 38 			this.needsSort();
 39 		}
 40 
 41 		return view;
 42 	}
 43 
 44 	proto.removeSubview = function(targetView) {
 45 		for (var i = 0, view; view = this._subviews[i]; ++i) {
 46 			if (view === targetView) {
 47 				this._subviews.splice(i, 1);
 48 				this.needsRepaint();
 49 				targetView._clearSuperview();
 50 				break;
 51 			}
 52 		}
 53 	}
 54 
 55 	proto._localizePt = function(pt) {
 56 			var s = this.style;
 57 			pt.x -= s.x + s.anchorX;
 58 			pt.y -= s.y + s.anchorY;
 59 			if (s.r) { pt.rotate(-s.r); }
 60 			pt.scale(1 / s.scale);
 61 			pt.x += s.anchorX;
 62 			pt.y += s.anchorY;
 63 			return pt;
 64 	}
 65 
 66 	proto.wrapTick = function(dt, app) {
 67 		this.tick && this.tick(dt, app);
 68 
 69 		for (var i = 0, view; view = this._subviews[i]; ++i) {
 70 			view.wrapTick(dt, app);
 71 		}
 72 		
 73 		if (this._needsRepaint) {
 74 			this._needsRepaint = false;
 75 			app.needsRepaint();
 76 		}
 77 	}
 78 
 79 	proto._firstRender = true;
 80 
 81 	proto.wrapRender = function(ctx, opts) {
 82 		var s = this.style;
 83 		if (!s.visible) { return; }
 84 		
 85 		if (this._firstRender) { this._firstRender = false; this.buildView(); }
 86 		if (this._needsSort) { this._needsSort = false; this._subviews.sort(); }
 87 		if (this._needsReflow) { this.wrapReflow(); }
 88 		
 89 		ctx.save();
 90 		ctx.translate(s.x + s.anchorX, s.y + s.anchorY);
 91 		
 92 		if (s.r) { ctx.rotate(s.r); }
 93 		
 94 		// clip this render to be within its view;
 95 		if (s.scale != 1) { ctx.scale(s.scale, s.scale); }
 96 		if (s.opacity != 1) { ctx.globalAlpha *= s.opacity; }
 97 
 98 		ctx.translate(-s.anchorX, -s.anchorY);
 99 
100 		if (this.style.clip) { ctx.clipRect(0, 0, s.width, s.height); }
101 		var filters = this.getFilters();
102 		ctx.setFilters(filters);
103 
104 		try {
105 			if (this.style.backgroundColor) {
106 				ctx.fillStyle = this.style.backgroundColor;
107 				ctx.fillRect(0, 0, s.width, s.height);
108 			}
109 
110 			this.render && this.render(ctx, opts);
111 			this._renderSubviews(ctx, opts);
112 		} catch(e) {
113 		 	logger.error(this, e.message, e.stack);
114 		} finally {
115 			ctx.clearFilters();
116 			ctx.restore();
117 		}
118 	}
119 
120 	proto._renderSubviews = function(ctx, opts) {
121 		var i = 0, view;
122 		while (view = this._subviews[i++]) {
123 			view.wrapRender(ctx, opts);
124 		}
125 	}
126 	
127 
128 	proto.toString = function() { return this.__sortKey; }
129 }
