1 "use import";
  2 
  3 import lib.PubSub;
  4 from util.browser import $;
  5 
  6 /**
  7  * @extends lib.PubSub
  8  */
  9 exports = Class(lib.PubSub, function() {
 10 	this.init = function(opts) {
 11 		this._el = $({
 12 			tag: 'input',
 13 			parent: document.body,
 14 			style: {
 15 				position: 'absolute',
 16 				left: '-100px',
 17 				top: '-100px'
 18 			},
 19 			attrs: {
 20 				type: 'text',
 21 				value: opts && opts.value || ''
 22 			}
 23 		});
 24 		this._value = this._el.value;
 25 		this._selectionStart = 0;
 26 		this._selectionEnd = 0;
 27 		
 28 		$.onEvent(this._el, 'keydown', this, 'checkValue');
 29 		$.onEvent(this._el, 'keyup', this, 'checkValue');
 30 		$.onEvent(this._el, 'keypress', this, 'checkValue');
 31 		$.onEvent(this._el, 'focus', this, 'onFocus');
 32 		$.onEvent(this._el, 'blur', this, 'onBlur');
 33 	}
 34 	
 35 	this.onFocus = function() { this.publish('Focus'); }
 36 	this.onBlur = function() { this.publish('Blur'); }
 37 	
 38 	this.checkValue = function(evt) {
 39 		var target = evt.target,
 40 			start = target.selectionStart,
 41 			end = target.selectionEnd;
 42 		
 43 		var value = this._el.value;
 44 		
 45 		if (value != this._value) {
 46 			this.publish('ChangeValue', value);
 47 			this._value = value;
 48 		}
 49 		
 50 		if (start != this._selectionStart) {
 51 			this._selectionStart = start;
 52 			this.publish('ChangeSelectionStart', start);
 53 		}
 54 		
 55 		if (end != this._selectionEnd) {
 56 			this._selectionEnd = end;
 57 			this.publish('ChangeSelectionEnd', end);
 58 		}
 59 	}
 60 	
 61 	this.focus = function() { logger.log('focus'); this._el.focus(); }
 62 	this.blur = function() { this._el.blur(); }
 63 });

 66 // Set desired tab- defaults to four space softtab
 67 var tab = '    ',
 68 	tabLength = 4;
 69 
 70 Array.prototype.map.call(document.getElementsByTagName('textarea'), function(el) {
 71 	el.addEventListener('keydown', checkTab, false);
 72 });
 73 
 74 function checkTab(evt) {
 75 	var t = evt.target;
 76 	var ss = t.selectionStart;
 77 	var se = t.selectionEnd;
 78  
 79 	// Tab key - insert tab expansion
 80 	if (evt.keyCode == 9) {
 81 		evt.preventDefault();
 82 		
 83 		if (evt.shiftKey) {
 84 			// Special case of multi line selection
 85 			if (ss != se && t.value.slice(ss,se).indexOf('\n') != -1) {
 86 				// In case selection was not of entire lines (e.g. selection begins in the middle of a line)
 87 				// we ought to tab at the beginning as well as at the start of every following line.
 88 				var i = ss;
 89 				while(i && t.value.charAt(i - 1) != '\n') { --i; }
 90 				var pre = t.value.slice(0, i);
 91 				var post = t.value.slice(se, t.value.length);
 92 				var sel = t.value.slice(i, se).replace(
 93 					new RegExp('(^|\n)' + tab, 'g'),
 94 					function(match) {
 95 						se -= tab.length;
 96 						if (match.charAt(0) == '\n') {
 97 							return '\n';
 98 						} else {
 99 							ss -= tab.length;
100 							return '';
101 						}
102 					});
103 				t.value = pre.concat(sel).concat(post);
104 
105 				t.selectionStart = ss;
106 				t.selectionEnd = se;
107 			} else {
108 				// "Normal" case (no selection or selection on one line only)
109 
110 				var i = ss;
111 				while(i && t.value.charAt(i - 1) != '\n') { --i; }
112 				if (t.value.substring(i, i + tab.length) == tab) {
113 					t.value = t.value.slice(0, i).concat(t.value.slice(i + tab.length, t.value.length));
114 					if (ss == se) {
115 						t.selectionStart = t.selectionEnd = ss - (ss == i ? 0 : tab.length);
116 					} else {
117 						t.selectionStart = ss - (ss == i ? 0 : tab.length);
118 						t.selectionEnd = se - tab.length;
119 					}
120 				}
121 			}
122 		} else {
123 			// Special case of multi line selection
124 			if (ss != se && t.value.slice(ss,se).indexOf("\n") != -1) {
125 				// In case selection was not of entire lines (e.g. selection begins in the middle of a line)
126 				// we ought to tab at the beginning as well as at the start of every following line.
127 				var i = ss;
128 				while(i && t.value.charAt(i - 1) != '\n') { --i; }
129 				var pre = t.value.slice(0, i);
130 				var post = t.value.slice(se, t.value.length);
131 				var sel = t.value.slice(i, se).replace(/\n/g, function() {
132 					se += tab.length;
133 					return '\n' + tab;
134 				});
135 				t.value = pre.concat(tab).concat(sel).concat(post);
136 
137 				t.selectionStart = ss + tab.length;
138 				t.selectionEnd = se + tab.length;
139 			} else {
140 				// "Normal" case (no selection or selection on one line only)
141 
142 				var i = ss;
143 				while(i && t.value.charAt(i - 1) != '\n') { --i; }
144 
145 				t.value = t.value.slice(0, i).concat(tab).concat(t.value.slice(i, t.value.length));
146 
147 				if (ss == se) {
148 					t.selectionStart = t.selectionEnd = ss + tab.length;
149 				} else {
150 					t.selectionStart = ss + tab.length;
151 					t.selectionEnd = se + tab.length;
152 				}
153 			}
154 		}
155 	} else if (evt.shiftKey || evt.metaKey || evt.ctrlKey) {
156 		return;
157 	} else if (evt.keyCode == 8 && t.value.slice(ss - tabLength, ss) == tab) {
158 		// Backspace key - delete preceding tab expansion, if exists
159 		evt.preventDefault();
160 		
161 		t.value = t.value.slice(0,ss - tabLength).concat(t.value.slice(ss,t.value.length));
162 		t.selectionStart = t.selectionEnd = ss - tab.length;
163 	} else if (evt.keyCode == 46 && t.value.slice(se, se + tabLength) == tab) {
164 		// Delete key - delete following tab expansion, if exists
165 		evt.preventDefault();
166 		
167 		t.value = t.value.slice(0,ss).concat(t.value.slice(ss + tabLength,t.value.length));
168 		t.selectionStart = t.selectionEnd = ss;
169 	} else if (evt.keyCode == 37 && t.value.slice(ss - tabLength, ss) == tab) {
170 		// Left/right arrow keys - move across the tab in one go
171 		evt.preventDefault();
172 		t.selectionStart = t.selectionEnd = ss - tabLength;
173 	} else if (evt.keyCode == 39 && t.value.slice(ss, ss + tabLength) == tab) {
174 		evt.preventDefault();
175 		t.selectionStart = t.selectionEnd = ss + tabLength;
176 	}
177 }
