1 exports.getDeviceImports = function(device) {
  2 	if (device == 'browser') {
  3 		return [
  4 				'.env.browser.initialize',
  5 				'.env.browser.Input',
  6 				'.env.browser.Canvas',
  7 				'.env.browser.Context2D',
  8 				'.env.browser.KeyListener',
  9 				'.env.browser.Audio',
 10 				'.env.browser.SoundManager',
 11 				'.env.browser.Timer',
 12 				'.env.browser.initialize',
 13 
 14 				// TODO: move this
 15 				// 'sdk._api.client.exportSettings'
 16 			];
 17 	}
 18 
 19 	if (device == 'native') {
 20 		return [
 21 				// runtimeNative/device/__imports__.js in will handle the remaining imports
 22 				'runtimeNative.device.initialize',
 23 			];
 24 	}
 25

 29 exports.getBackendImports = function(backend) {

 45 				'.canvas.ScrollView',
 46 				'.canvas.TextView',
 47 				'.canvas.View',

 58 exports.resolve = function(env, opts) {
 59 	var imports = [];
 60 	var add = function(list) { imports = imports.concat(list); };
 61 
 62 	add(exports.getDeviceImports(env));
 63 
 64 	// backend-method
 65 	var backend = opts.dynamic && opts.dynamic.timestepBackend;
 66 	if (backend) {
 67 		add(exports.getBackendImports(backend));
 68 	} else if (env == 'browser') {
 69 		add(exports.getBackendImports('dom'));
 70 		add(exports.getBackendImports('canvas'));
 71 	} else {
 72 		add(exports.getBackendImports('canvas'));
 73 	}
 74 	
 75 	return imports;
 76 }
