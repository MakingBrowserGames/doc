1 jsio('import .View');
  2 jsio('import .Image');
  3 
  4 /**
  5  * @extends timestep.View
  6  */
  7 exports = Class(View, function(supr) {
  8 	this.init = function(opts) {
  9 		supr(this, 'init', arguments);
 10 		this._dataSource = opts.dataSource;
 11 		this._tileWidth = opts.tileWidth;
 12 		this._tileHeight = opts.tileHeight;
 13 		this._fullWidth = this._dataSource.getNumCols() * this._tileWidth;
 14 		this._fullHeight = this._dataSource.getNumRows() * this._tileHeight;
 15 		
 16 //		this._canvas = document.createElement('canvas')
 17 		this._lastState = { _dirty: null };
 18 		this._dirty = true;
 19 	}
 20 	
 21 	this.render = function(ctx, viewport, scale) {
 22 		var tileWidth = this._tileWidth;
 23 		var tileHeight = this._tileHeight;
 24 		
 25 //		logger.info('viewport.y', viewport.y, 'this._tileHeight', this._tileHeight);
 26 		var iStart = Math.floor(viewport.y / tileHeight);
 27 		var iEnd = Math.min(Math.ceil((viewport.height + viewport.y) / tileHeight), this._dataSource.getNumRows()-1);
 28 		var jStart = Math.floor(viewport.x / tileWidth);
 29 		var jEnd = Math.min(Math.ceil((viewport.width + viewport.x) / tileWidth), this._dataSource.getNumCols()-1);
 30 		
 31 		var xOffset = viewport.x % tileWidth;
 32 		var yOffset = viewport.y % tileHeight;
 33 		
 34 		var numRows = this._dataSource.getNumRows();
 35 		var numCols = this._dataSource.getNumCols();
 36 
 37 		
 38 		var state = {
 39 			xOffset: xOffset,
 40 			yOffset: yOffset,
 41 			iStart: iStart,
 42 			iEnd: iEnd,
 43 			jStart: jStart,
 44 			jEnd: jEnd,
 45 			width: viewport.width,
 46 			height: viewport.height,
 47 			x: viewport.x,
 48 			y: viewport.y
 49 		}
 50 
 51 		for (key in state) {
 52 			if (state[key] != this._lastState[key]) {
 53 				this._dirty = true;
 54 				break;
 55 			}
 56 		}
 57 		
 58 
 59 		if (this._dirty) {
 60 			this._canvas.height = viewport.height;
 61 			this._canvas.width = viewport.width;
 62 			var buffer = this._canvas.getContext('2d');
 63 			this._lastState = state;
 64 		
 65 			for (var i = iStart; i <= iEnd; ++i) {
 66 				for (var j = jStart; j <= jEnd; ++j) {
 67 					var img = this._dataSource.getTileAt(i,j);
 68 					if (!img) { continue; }
 69 					var x = tileWidth * (j-jStart) - xOffset;
 70 					var y = tileHeight * (i-iStart) - yOffset;
 71 					img.render(buffer, x, y, tileWidth, tileHeight);
 72 					this._dirty = false;
 73 				}
 74 			}
 75 			logger.info('dirty', xOffset, yOffset);
 76 
 77 		}
 78 		
 79 		ctx.drawImage(this._canvas, 0, 0, viewport.width, viewport.height, 0, 0, viewport.width*scale, viewport.height*scale);
 80 	}
 81 })
