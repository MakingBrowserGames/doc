1 "use import";
  2 
  3 import timestep.View;
  4 
  5 /**
  6  * @extends timestep.View
  7  */
  8 exports = Class(timestep.View, function(supr) {
  9 	this.init = function(opts) {
 10 		supr(this, 'init', [opts]);
 11 		if (opts) {
 12 			var s = this.style;
 13 			if (opts.insetX) {
 14 				s.insetX = opts.insetX;
 15 				s.left = opts.insetX[0];
 16 				s.right = opts.insetX[1];
 17 			}
 18 			
 19 			if (opts.insetY) {
 20 				s.insetY = opts.insetY;
 21 				s.top = opts.insetY[0];
 22 				s.bottom = opts.insetY[1];
 23 			}
 24 
 25 			s.centerX = opts.centerX;
 26 			s.centerY = opts.centerY;
 27 			s.widthPercent = opts.widthPercent;
 28 			s.heightPercent = opts.heightPercent;
 29 		}
 30 	}
 31 	
 32 	this.wrapReflow = function() {
 33 		var s = this.style;
 34 		var sv = this._superview;
 35 		var svWidth = sv && sv.style.width / s.scale;
 36 		var svHeight = sv && sv.style.height / s.scale;
 37 		
 38 		this.reflow();
 39 		this._needsReflow = false;
 40 
 41 		// resize this view
 42 		var w = sv && this.computeWidth(svWidth) || s.width;
 43 		var h = sv && this.computeHeight(svHeight) || s.height;
 44 		
 45 		// did the size change? if so, we'll need to set it directly so we 
 46 		// don't force another reflow (infinite loop)
 47 		if (!this._reflowProp || this._reflowProp.width != w || this._reflowProp.height != h) {
 48 			this._reflowProp = {width: w, height: h};
 49 			//this.style._directSetSize(w, h);
 50 			s.width = w;
 51 			s.height = h;
 52 			var reflow = true;
 53 		}
 54 		
 55 		if (sv) {
 56 			if (s.centerX) { s.x = (svWidth - s.width) / 2; }
 57 			if (s.centerY) { s.y = (svHeight - s.height) / 2; }
 58 			if (!('left' in s) && ('right' in s)) { s.x = svWidth - s.right - s.width; }
 59 			if (!('top' in s) && ('bottom' in s)) { s.y = svHeight - s.bottom - s.height; }
 60 		}
 61 		
 62 		supr(this, 'wrapReflow', arguments);
 63 		
 64 		if (reflow) {
 65 			for (var i = 0, view; view = this._subviews[i]; ++i) {
 66 				view.wrapReflow();
 67 			}
 68 		}
 69 	}
 70 	
 71 	this.computeWidth = function(parentWidth) {
 72 		var s = this.style;
 73 		if (s.widthPercent) { return parentWidth * s.widthPercent / 100; }
 74 		if (s.insetX) { return parentWidth - s.left - s.right; }
 75 		
 76 	}
 77 	
 78 	this.computeHeight = function(parentHeight) {
 79 		var s = this.style;
 80 		if (s.heightPercent) { return parentHeight * s.heightPercent / 100; }
 81 		if (s.insetY) { return parentHeight - s.top - s.bottom; }
 82 	}
 83 });
