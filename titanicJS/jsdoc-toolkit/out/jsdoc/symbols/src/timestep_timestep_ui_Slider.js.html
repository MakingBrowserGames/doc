1 "use import";
  2 
  3 import ..View;
  4 import std.js as JS;
  5 import .roundedRect;
  6 
  7 /**
  8  * @extends timestep.View
  9  */
 10 exports = Class(View, function(supr) {
 11 	this.init = function(opts) {
 12 		opts = JS.merge(opts, {
 13 			min: 0,
 14 			max: 1,
 15 			initial: 0.5,
 16 			discrete: false,
 17 			isHorizontal: true
 18 		});
 19 		
 20 		this._discrete = opts.discrete;
 21 		this._value = opts.initial;
 22 		this._min = opts.min;
 23 		this._max = opts.max;
 24 		this._isHorizontal = opts.isHorizontal;
 25 		this._key = this._isHorizontal ? 'x' : 'y';
 26 		
 27 		supr(this, 'init', arguments);
 28 	}
 29 	
 30 	this.inputStart = function() {
 31 		this.startDrag();
 32 	}
 33 	
 34 	this.onDragStart = function() {
 35 		this._prevValue = this._value;
 36 	}
 37 	
 38 	this._setValFromPt = function(pt) {
 39 		this._value = pt[this._key] / this.style.width * (this._max - this._min) + this._min;
 40 		if (this._value < this._min) { this._value = this._min; }
 41 		if (this._value > this._max) { this._value = this._max; }
 42 	}
 43 	
 44 	this.onDrag = function(startPt, currentPt, delta) {
 45 		var oldValue = this._value;
 46 		this._setValFromPt(currentPt);
 47 		this.publish('Changing', this._value, oldValue);
 48 	}
 49 	
 50 	this.onDragStop = function(startPt, currentPt, delta) {
 51 		var oldValue = this._value;
 52 		this._setValFromPt(currentPt);
 53 		this.publish('Changing', this._value, oldValue);
 54 		this.publish('Change', this._value, this._prevValue);
 55 	}
 56 	
 57 	var style = {
 58 		thickness: 2,
 59 		borderRadius: 5,
 60 		thickness: 20,
 61 		barRadius: 2
 62 	}
 63 	
 64 	this.render = function(ctx) {
 65 		var s = this.style,
 66 			c = s.height / 2,
 67 			x = (this._value - this._min) / (this._max - this._min) * s.width + s.x,
 68 			t = style.thickness / 2;
 69 
 70 		ctx.setLineWidth(1);
 71 		ctx.setStrokeStyle('#888');
 72 		ctx.setFillStyle('#AAA');
 73 		
 74 		ctx.beginPath();
 75 		roundedRect(ctx, 0, c - style.barRadius, s.width, style.barRadius * 2, style.barRadius);
 76 		roundedRect(ctx, x - t, 0, style.thickness, s.height, style.borderRadius);
 77 		ctx.stroke();
 78 		ctx.fill();
 79 	}
 80 });
