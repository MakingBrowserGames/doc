1 "use import";
  2 
  3 import std.js as JS;
  4 import ..Sprite;
  5 import timestep.ui.CanvasUtils as CanvasUtils;
  6 import timestep.device;
  7 
  8 /**
  9  * @extends timestep.Sprite
 10  */
 11 exports = Class(Sprite, function(supr) {
 12 
 13 	Class.ctor(this, supr, {
 14 		text: 'Click Here',
 15 		width: 100,
 16 		background: '#56d',
 17 		color: '#FFF',
 18 		height: 50,
 19 		isEnabled: true
 20 	}, function(opts) {
 21 		this._state = '';
 22 		this._isEnabled = opts.isEnabled;
 23 		this._isPressed = false;
 24 		this._isToggled = false;
 25 		this._state = 'normal';
 26 		
 27 		this.label = opts.label || opts.text;
 28 		this.style.color = opts.color;
 29 		this.style.font = opts.font || 'bold 16px ' + timestep.device.defaultFontFamily;
 30 		this.style.backgroundColor = opts.background;
 31 	});
 32 	
 33 	this.setEnabled = function(isEnabled) {
 34 		this._isEnabled = isEnabled;
 35 		this.updateState();
 36 	}
 37 	
 38 	this.setToggled = function(isToggled) {
 39 		this._isToggled = isToggled;
 40 	}
 41 	
 42 	this.isToggled = function() { return this._isToggled; }
 43 	
 44 	this.updateState = function() {
 45 		this._state = !this._isEnabled ? 'disabled'
 46 					 : this._isPressed ? 'pressed'
 47 					 : this.isInputOver() ? 'hover'
 48 					 : 'normal';
 49 		this.needsRepaint();
 50 	}
 51 	
 52 	this.onInputStart = function() {
 53 		this._isPressed = true;
 54 		this.updateState();
 55 	}
 56 	
 57 	this.onInputSelect = function() {
 58 		this._isPressed = false;
 59 		this.updateState();
 60 
 61 		if (this._isEnabled) {
 62 			// TODO: ui widgets need a better way to delegate events
 63 			this.publish('Select');
 64 		}
 65 	}
 66 	
 67 	this.onInputOver = function() { this.updateState(); }
 68 	this.onInputOut = function() { this.updateState(); }
 69 	
 70 	this.setLabel = function(label) { this.label = label; }
 71 
 72 	this.render = function(ctx) {
 73 		supr(this, 'render', [ctx]);
 74 		ctx.font = this.style.font;
 75 		var w = this.style.width | 0;
 76 		var h = this.style.height | 0;
 77 
 78 		var textOffsetX = 0, textOffsetY = 0;
 79 		if (this._isPressed) {
 80 			textOffsetX = textOffsetY = 1;
 81 		}
 82 		
 83 		ctx.fillStyle = this.style.backgroundColor;
 84 		CanvasUtils.fillRoundedRect(ctx, 0, 0, w, h, 10);
 85 		
 86 		ctx.fillStyle = "rgb(255,0,0)";
 87 		CanvasUtils.fillRoundedRect(ctx, 1, 1, w, h / 2 - 1, 10, 10, 0, 0);	
 88 		
 89 		// draw the border
 90 		ctx.strokeStyle = 'rgba(0,0,0, 0.7)';
 91 		CanvasUtils.roundRectPath(ctx, 0.5, 0.5, w - 1.0, h - 1.0, 10);
 92 		ctx.stroke();
 93 
 94 		ctx.textAlign = 'center';
 95 		ctx.textBaseline = 'middle';
 96 
 97 		// draw the inset text shadow
 98 		ctx.fillStyle = 'rgba(0,0,0,0.6)';
 99 		ctx.fillText(this.label, w / 2 - 1.0 + textOffsetX, h / 2 - 1.0 + textOffsetY, w - 20);
100 
101 		// draw the text
102 		ctx.fillStyle = this.style.color;
103 		ctx.fillText(this.label, w / 2 + textOffsetX, h / 2 + textOffsetY, w - 20);
104 
105 		if (this._state == 'disabled') {
106 			ctx.fillStyle = 'rgba(80,80,80,0.4)';
107 			CanvasUtils.fillRoundedRect(ctx, 0, 0, w, h, 10, 10, 0, 0);	
108 		}
109 		else if (this._state == 'hover') {
110 			ctx.fillStyle = 'rgba(255,255,255,0.05)';
111 			CanvasUtils.fillRoundedRect(ctx, 0, 0, w, h, 10, 10, 0, 0);	
112 		} else if (this._state == 'pressed') {
113 			ctx.fillStyle = 'rgba(0,0,0,0.1)';
114 			CanvasUtils.fillRoundedRect(ctx, 0, 0, w, h, 10, 10, 0, 0);	
115 		} 
116 	}
117 });
