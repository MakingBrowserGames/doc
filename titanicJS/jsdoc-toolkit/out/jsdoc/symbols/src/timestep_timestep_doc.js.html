1 "use import";
  2 
  3 import lib.PubSub;
  4 import lib.Enum as Enum;
  5 import std.js as JS;
  6 from util.browser import $;
  7 
  8 import .device;
  9 import .simulateDevice;
 10 
 11 var SCALING = Enum('FIXED', 'RESIZE', 'MANUAL');
 12 var defaultScalingMode = device.isMobileBrowser || device.simulating ? SCALING.RESIZE : SCALING.FIXED;
 13 
 14 /**
 15  * @extends lib.PubSub
 16  */
 17 var Document = Class(lib.PubSub, function() {
 18 	this.init = function() {
 19 		if (!$) { return; }
 20 		
 21 		var doc = GLOBAL.document,
 22 			body = doc && doc.body;
 23 		
 24 		this._el = $({
 25 			parent: body,
 26 			style: {
 27  				position: 'absolute',
 28 				overflow: 'hidden',
 29 				width: '100%',
 30 				height: '100%'
 31 			}
 32 		});
 33 		
 34 		device.screen.subscribe('Resize', this, 'onResize');
 35 						
 36 		if (exports.postCreateHook) { exports.postCreateHook(this); }
 37 		this.setScalingMode(defaultScalingMode);
 38 	}
 39 	
 40 	this.unsubscribeResize = function() { device.screen.unsubscribe('Resize', this, 'onResize'); }
 41 	
 42 	this.setApp = function(app) {
 43 		if (app == this._app) { return; }
 44 		
 45 		this._app = app;
 46 		this._canvas = this._app.getCanvas();
 47 
 48 		this.appendChild(this._canvas);
 49 
 50 		var ctx = this._canvas.getContext('2d');
 51 		if (ctx.setParentNode) { ctx.setParentNode(this._el); }
 52 	}
 53 	
 54 	this.getElement = function() { return this._el; }
 55 	
 56 	this.setScalingMode = function(scalingMode, opts) {
 57 		this._scalingMode = scalingMode;
 58 		
 59 		var el = this._el,
 60 			s = el.style;
 61 		
 62 		switch (scalingMode) {
 63 			case SCALING.FIXED:
 64 				opts = merge(opts, {
 65 						width: device.width,
 66 						height: device.height
 67 					});
 68 				s.width = opts.width + 'px';
 69 				s.height = opts.height + 'px';
 70 				break;
 71 			case SCALING.RESIZE:
 72 				opts = merge(opts, {
 73 						resizeCanvas: true
 74 					});
 75 				// fall through:
 76 			case SCALING.MANUAL:
 77 				s.margin = '0px';
 78 				s.width = '100%';
 79 				s.height = '100%';
 80 				break;
 81 		}
 82 		
 83 		this._scalingOpts = opts;
 84 		this.onResize();
 85 		setTimeout(bind(this, 'onResize'), 1000);
 86 	}
 87 	
 88 	this.onResize = function() {
 89 		var el = this._el;
 90 		var s = this._el.style;
 91 		
 92 		el.className = device.screen.orientation;
 93 		logger.log('resize', device.width, device.height);
 94 		
 95 		var width = device.width;
 96 		var height = device.height;
 97 		var mode = this._scalingMode;
 98 		var opts = this._scalingOpts;
 99 		
100 		if (mode == SCALING.FIXED) {
101 			width = opts.width;
102 			height = opts.height;
103 		}
104 		
105 		// enforce maxWidth/maxHeight
106 		// if maxWidth/maxHeight is met, switch a RESIZE scaling mode to FIXED (center the document on the screen)
107 		if (opts.maxWidth && width > opts.maxWidth) {
108 			width = opts.maxWidth;
109 			if (mode == SCALING.RESIZE) { mode = SCALING.FIXED; }
110 		}
111 		
112 		if (opts.maxHeight && height > opts.maxHeight) {
113 			height = opts.maxHeight;
114 			if (mode == SCALING.RESIZE) { mode = SCALING.FIXED; }
115 		}
116 		
117 		switch (mode) {
118 			case SCALING.MANUAL:
119 				break; // do nothing
120 			case SCALING.FIXED:
121 				// try to center the container
122 				el.style.top = Math.round(Math.max(0, (window.innerHeight - height) / 2)) + 'px';
123 				el.style.left = Math.round(Math.max(0, (window.innerWidth - width) / 2)) + 'px';
124 				
125 				s.width = width + 'px';
126 				s.height = height + 'px';
127 				break;
128 			case SCALING.RESIZE:
129 				// if we have a canvas element, scale it
130 				if (opts.resizeCanvas && this._canvas
131 						&& (this._canvas.width != width || this._canvas.height != height)) {
132 					this._canvas.width = width;
133 					this._canvas.height = height;
134 				}
135 				
136 				s.width = width + 'px';
137 				s.height = height + 'px';
138 				break;
139 		}
140 		
141 		// make sure to force a render immediately (should we use needsRepaint instead?)
142 		this._setDim(width, height);
143 		if (this._app) { this._app.render(); }
144 	}
145 	
146 	this._setDim = function(width, height) {
147 		if (this.width != width || this.height != height) {
148 			this.width = width;
149 			this.height = height;
150 			this.publish('Resize', width, height);
151 		}
152 	}
153 	
154 	this.setColors = function(bgColor, appColor) {
155 		if (this._el) {
156 			this._el.style.background = appColor;
157 			document.documentElement.style.background = document.body.style.background = bgColor;
158 		}
159 	}
160 	
161 	this.appendChild = function(el) {
162 		this._el.appendChild(el);
163 	}
164 	
165 	this.getOffset = function() {
166 		return {
167 			x: this._el.offsetLeft,
168 			y: this._el.offsetTop
169 		};
170 	}
171 });
172 
173 exports = new Document();
174 exports.SCALING = SCALING;
175 
176 exports.setDocStyle = function() {
177 	var doc = GLOBAL.document,
178 		body = doc && doc.body;
179 	
180 	if (body) {
181 		var docStyle = {
182 			height: '100%',
183 			margin: '0px',
184 			padding: '0px'
185 		};
186 	
187 		$.style(document.documentElement, docStyle);
188 		$.style(document.body, docStyle);
189 	}
190 }
191 
192 exports.defaultParent = null;
193 exports.postCreateHook = null;
