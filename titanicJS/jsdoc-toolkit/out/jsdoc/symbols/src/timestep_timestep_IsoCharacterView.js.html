 20 var roundDown = function(num){
 21 	//rounds down positive and negative numbers to the next lowest integer
 22 	if (num >= 0){
 23 		return num | 0;
 24 	}else{
 25 		return num - 1 | 0;
 26 	}
 27 }

107 	this._getSquares = function(center, rad) {
108 		/* returns array of tiles overlapped by 
109 		   character's bounding circle */
110 		
111 		var squares = [];
112 		var i = center.i | 0;
113 		var j = center.j | 0;
114 		squares.push({i: center.i | 0, j: center.j | 0, walkable: this._isoView.isWalkable(i,j)});
115 
116 		var minI = roundDown(center.i - this._bound.rad);
117 		var maxI = roundDown(center.i + this._bound.rad);
118 		
119 		var minJ = roundDown(center.j - this._bound.rad);
120 		var maxJ = roundDown(center.j + this._bound.rad);
121 
122 		for (var i = minI; i <= maxI; ++i){
123 			for(var j = minJ; j <= maxJ; ++j){
124 				var walk = this._isoView.isWalkable(i, j);
125 				var tile = {i:i, j:j, walkable: walk};
126 				squares.push(tile);
127 			}
128 		}
129 		return squares;
130 	};

182 	this.tick = function(dt) {
183 
184 		var velocity = this._velocity;
185 
186 		if (!velocity.di && !velocity.dj) { 
187 			supr(this,'tick',arguments);
188 			return;
189 		}
190 		
191 		var center = this._getIsoBound();
192 		var toGo = new Line(center.i, center.j,this._targetPosition.i, this._targetPosition.j);
193 		var magToGo = toGo.getLength();
194 
195 		var di = velocity.di * dt * this._speed;
196 		var dj = velocity.dj * dt * this._speed;
197 		
198 		var dl = di * di + dj * dj;
199 		
200 		var tiles = this._getSquares({i: center.i + di, j: center.j + dj}, center.rad);
201 		
202 		var allowed = true;
203 		for (var k = 0, tile; tile = tiles[k]; ++k){
204 			if (!tile.walkable){
205 				allowed = false;
206 				break;
207 			}
208 		}
209 		
210 		if (allowed){
211 			if (dl > magToGo * magToGo){
212 				velocity.di = velocity.dj = 0;
213 				this.setAnimation(velocity);
214 				this._setIsoBound(this._targetPosition);
215 				this.publish('position:target');
216 			} else {
217 				this._setIsoBound({i: center.i + di, j: center.j + dj});
218 			}
219 		}else{
220 			velocity.di = velocity.dj = 0;
221 			this.setAnimation(velocity);
222 			this.publish('position:stopped');
223 		}
224 
225 		supr(this,'tick',arguments);
226 	
227 	};
