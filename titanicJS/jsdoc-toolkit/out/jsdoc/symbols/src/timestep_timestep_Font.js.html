1 "use import";
  2 import .device;
  3 import std.js as JS;
  4 
  5 var cache = {},
  6 	weights = 'normal|bold|bolder|lighter|[1-9]00',
  7 	styles = 'normal|italic|oblique',
  8 	units = 'px|pt|pc|in|cm|mm|%',
  9 	name = '[\\w\\- ]+',
 10 	fontParser = new RegExp(
 11 		'^ *'
 12 			+ '(?:(' + weights + ') *)?'
 13 			+ '(?:(' + styles + ') *)?'
 14 			+ '([\\d\\.]+)(' + units + ')'
 15 			+ '('+ name + ')',
 16 			'i'
 17 	),
 18 	deviceUnit = device.isMobile && device.isIOS ? 'pt' : 'px',
 19     sizeParser = new RegExp('([\\d\\.]+)(' + units + ')', 'i'),
 20 	TO_PT = {
 21 		'pt': 1,
 22 		'px': 3 / 4,
 23 		'in': 3 / 4 * 96,
 24 		'mm': 3 / 4 * 96 / 25.4,
 25 		'cm': 3 / 4 * 96 / 2.54
 26 	},
 27 	TO_PX = {
 28 		'pt': 4 / 3,
 29 		'px': 1,
 30 		'in': 96,
 31 		'mm': 96 / 25.4,
 32 		'cm': 96 / 2.54
 33 	},
 34 	IOS_FONT_SCALE = 2;
 35 
 36 import .leaks;
 37 leaks.register('Font', leaks.createObjectKeyCounter(cache));
 38 
 39 function parseSize(sizeStr, unit) {
 40 	var match = sizeStr.match(sizeParser);
 41 	if (!match) { throw 'invalid font size'; }
 42 	return {
 43 		value: parseFloat(match[1]),
 44 		unit: match[2]
 45 	};
 46 }
 47 
 48 function toPx(size) {
 49 	return {
 50 		value: size.value * TO_PX[size.unit],
 51 		unit: 'px'
 52 	};
 53 }
 54 
 55 function toPt(size) {
 56 	return {
 57 		value: size.value * TO_PT[size.unit],
 58 		unit: 'pt'
 59 	};
 60 }
 61 
 62 function parseFont(fontStr) {
 63 	var match = fontStr.match(fontParser);
 64 	if (!match) { throw 'invalid font string'; }
 65 	
 66 	var res = {};
 67 	res.weight = match[1] || 'normal';
 68 	res.style = match[2] || 'normal';
 69 	res.size = {
 70 		value: parseFloat(match[3]),
 71 		unit: match[4]
 72 	};
 73 	res.name = match[5]
 74 		.replace(/[\-_]/g, ' ')
 75 		.replace(/[ ]+/g, ' ')
 76 		.replace(/^\s+|\s+$/g, '');
 77 	
 78 	return res;
 79 };

 81 var Font = exports = Class(function(){
 82 	
 83 	var defaults = {
 84 		name: device.defaultFontFamily,
 85 		size: 20,
 86 		unit: 'px',
 87 		style: '',
 88 		weight: ''
 89 	};
 90 
 91 	this.init = function(opts) {
 92 		if (typeof opts === 'string') {
 93 			if (opts in cache) { return cache[opts]; }
 94 			cache[opts] = this;
 95 			this._string = opts;
 96 			opts = parseFont(opts);
 97 		} else {
 98 			opts = JS.merge(opts, defaults);
 99 		}
100 		
101 		if (typeof opts.size == 'string') {
102 			opts.size = parseSize(opts.size);
103 		}
104 		
105 		this._name = opts.name;
106 		this._style = opts.style;
107 		this._size = opts.size;
108 		this._weight = opts.weight;
109 	};
110 	
111 	this.recalculateIosFont = function(fontDict){
112 		this._iosName = null;
113 		this.getIOSName(fontDict);
114 	};
115 	
116 	this.getSize = function() {
117 		return toPx(this._size).value;
118 	}
119 	
120 	this.getIOSSize = function() {
121 		if (!this._iosSize) {
122 			this._iosSize = toPt(this._size).value * IOS_FONT_SCALE;
123 		}
124 		
125 		return this._iosSize;
126 	}
127 	
128 	this.getIOSName = function(fontDict) {
129 		if (!this._iosName) {
130 			var fontName = this._name.toLowerCase();
131 			if (fontName in fontDict) {
132 				var	fontMap = fontDict[fontName];
133 				var cache = {
134 					bold: /bold/i.test(this._weight),
135 					italic: /(italic|oblique)/i.test(this._style),
136 					weight: this.weight
137 				};
138 				cache.family = fontName;
139 				cache.fontString = 
140 					(cache.bold && cache.italic
141 						? fontMap.bolditalic || fontMap.bold || fontMap.italic
142 					: cache.bold
143 						? fontMap.bold || fontMap.bolditalic
144 					: cache.italic
145 						? fontMap.italic || fontMap.bolditalic
146 					: null) || fontMap.normal;
147 				this._iosName = cache.fontString;
148 			} else {
149 				this._iosName = 'helvetica';
150 			}
151 		}
152 		
153 		return this._iosName;
154 	}
155 
156 	this.getName = function() { return this._name; }
157 });
158 
159 exports.parse = function(str) { return new Font(str); }
