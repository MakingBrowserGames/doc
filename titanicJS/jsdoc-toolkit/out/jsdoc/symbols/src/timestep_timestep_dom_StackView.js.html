1 "use import";
  2 import .View;
  3 
  4 /**
  5  * @extends timestep.dom.View
  6  */
  7 exports = Class(View, function(supr) {
  8 
  9 	this.init = function(opts) {
 10 		supr(this, 'init', arguments);
 11 		this.stack = [];
 12 	}
 13 	
 14 	this.getCurrentView = function() {
 15 		if (!this.stack.length) { return null; }
 16 		return this.stack[this.stack.length - 1];
 17 	}
 18 	
 19 	this.push = function(view, dontAnimate) {
 20 		// don't animate the first (base) view of a stackview unless explicitly asked to
 21 		if (!this.stack[0] && dontAnimate !== false) {
 22 			dontAnimate = true;
 23 		}
 24 		
 25 		var current = this.getCurrentView();
 26 		if (current) { this._hide(current, dontAnimate); }
 27 		view.style.width = this.style.width;
 28 		view.style.height = this.style.height;
 29 		this.stack.push(view);
 30 		this._show(view, dontAnimate);
 31 		return view;
 32 	}
 33 	
 34 	this._hide = function(view, dontAnimate, backward) {
 35 		view.publish('ViewWillDisappear');
 36 		if (!dontAnimate) {
 37 			// Prevent touches from triggering buttons/UI on the
 38 			// disappearing view. Unfortunately, canHandleEvents()
 39 			// doesn't affect subviews, so an overlay is added here
 40 			// so that touches just don't go through while it animates out.
 41 			var overlay = new View({parent: view, zIndex: 100000});
 42 			view.then({x: (backward ? 1 : -1) * view.style.width})
 43 				.then(bind(this, 'removeSubview', view))
 44 				.then(bind(view, 'publish', 'ViewDidDisappear'))
 45 				.then(bind(overlay, 'removeFromSuperview'));
 46 		} else {
 47 			this.removeSubview(view);
 48 			view.publish('ViewDidDisappear');
 49 		}
 50 	}
 51 
 52 	this._show = function(view, dontAnimate, backward) {
 53 		view.publish('ViewWillAppear');
 54 		view.style.visible = true;
 55 		if (!dontAnimate) {
 56 			view.style.x = (backward ? -1 : 1) * this.style.width;
 57 			this.addSubview(view);
 58 			view.then({x: 0})
 59 				.then(bind(view, 'publish', 'ViewDidAppear'));
 60 		} else {
 61 			this.addSubview(view);
 62 			view.style.x = 0;
 63 			view.publish('ViewDidAppear');
 64 		}
 65 	}
 66 	
 67 	this.pop = function(dontAnimate) {
 68 		if (!this.stack.length) { return false; }
 69 		var view = this.stack.pop();
 70 		this._hide(view, dontAnimate, true);
 71 		
 72 		if (this.stack.length) {
 73 			this._show(this.stack[this.stack.length - 1], dontAnimate, true);
 74 		}
 75 		
 76 		return view;
 77 	}
 78 	
 79 	this.popAll = function(dontAnimate) {
 80 		while (this.stack[1]) {
 81 			this.pop(dontAnimate);
 82 		}
 83 	}
 84 });
