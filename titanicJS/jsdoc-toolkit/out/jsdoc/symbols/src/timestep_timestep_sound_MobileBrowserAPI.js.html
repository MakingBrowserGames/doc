1 "use import";
  2 
  3 import lib.PubSub;
  4 import ..device;
  5 from util.underscore import _;
  6 
  7 /**
  8  * @extends lib.PubSub
  9  */
 10 var AudioAPI = exports = Class(lib.PubSub, function(supr) {
 11 
 12 	var defaults = {
 13 		oneChannelOnly: device.isMobileBrowser,
 14 		compiledFilename: 'compiled'
 15 	};
 16 
 17 	this.init = function(opts) {
 18 		opts = merge(opts, defaults);
 19 		supr(this, 'init', [opts]);
 20 		this._opts = opts;
 21 		this._map = opts.map;
 22 		this._audios = {};
 23 		this.oneChannelOnly = opts.oneChannelOnly;
 24 
 25 		setInterval(bind(this, '_ontimeupdate'), 200);
 26 		
 27 		if (!this.oneChannelOnly) {
 28 			_.each(opts.background, function(name) {
 29 				opts.map[name] = {'name': name}
 30 			}, this);
 31 		}
 32 
 33 		// Install the event listener right away. Set usecapture=true so that
 34 		// nothing else will affect us from intercepting this event.
 35 
 36 		this._load();
 37 		if (this.oneChannelOnly) {
 38 			this._boundLoadHandler = bind(this, '_playFirst');
 39 			document.body.addEventListener(device.events.start, 
 40 										   this._boundLoadHandler, true);
 41 		}
 42 		window.addEventListener('pagehide', bind(this, 'pause'), false);
 43 	}
 44 
 45 	this._createChannel = function(name, src) {
 46 		var audio = new Audio(src);
 47 		this._audios[name] = audio;
 48 
 49 		audio.addEventListener('error', bind(this, '_onerror'));
 50 		audio.addEventListener('timeupdate', bind(this, '_ontimeupdate'));
 51 
 52 		audio.load();
 53 	}
 54 
 55 	this.setMuted = function(muted) {
 56 		this.muted = muted;
 57 		if (muted) {
 58 			this.setVolume(0);
 59 		}
 60 	}
 61 	
 62 	this.setVolume = function(volume) {
 63 		_.each(this._audios, function(audio, key) {
 64 			audio.volume = volume;
 65 		});
 66 	}
 67 
 68 	this.unload = function() {
 69 		this.pause();
 70 		_.each(this._audios, function(audio, key) {
 71 			audio.src = '';
 72 		}, this);
 73 		// TODO remove event listeners
 74 	}
 75 
 76 	this._load = function() {
 77 	
 78 		this._audios = {};
 79 		var path = this._opts.path.replace(/\/$/, '');
 80 		if (this.oneChannelOnly) {
 81 			this._createChannel('AUDIO', path + '/' + this._opts.compiledFilename + '.m4a');
 82 		} else {
 83 			for (var key in this._map) if (this._map.hasOwnProperty(key)) {
 84 				if (key == 'SILENCE') { continue; }
 85 				this._createChannel(key, path + '/' + key + '.mp3');
 86 			}
 87 		}
 88 
 89 		logger.info('now loading', this._opts.src);
 90 
 91 		if (!this._publishedReady) {
 92 			this.publish('Ready'); // this is as close as we'll get with multiple sounds
 93 			this._publishedReady = true;
 94 		}
 95 	}
 96 
 97 	this._playFirst = function() {
 98 		document.body.removeEventListener(device.events.start, 
 99 										  this._boundLoadHandler, true);
100 		
101 		this._audios['AUDIO'].play();
102 	}
103 
104 	this._ontimeupdate = function(evt) {
105 		_.each(this._audios, function(audio, key) {
106 			if (audio.paused && !audio._pausedOnce) {
107 				audio.pause();
108 				audio._pausedOnce = true;
109 				audio._ready = true;
110 			}
111 
112 			if (this.oneChannelOnly) {
113 				if (!this._nowPlaying || audio.currentTime >= this._nowPlaying.end) {
114 					this.play('SILENCE');
115 				}
116 			}
117 		}, this);
118 
119 	}
120 	
121 	this._onerror = function(event) {
122 		var s = '';
123 		for (var key in event) {
124 			s += event[key] + ' ';
125 		}
126 		logger.info('ERROR', s);
127 		// this.unload();
128 		// this.publish('AudioError', event);
129 		// this._status = 'error';
130 	}
131 
132 	this.canPlay = function(name) {
133 		if (!this._map[name]) { return false; }
134 
135 		var requiredEnd = null;
136 		if (this.oneChannelOnly) {
137 			requiredEnd = this._map[name].end;
138 			name = 'AUDIO';
139 		}
140 		var audio = this._audios[name];
141 		if (!audio) {
142 			return false;
143 		}
144 
145 		// if (!audio._ready) {
146 		// 	return; // try downloading the whole file...
147 		// }
148 		if (audio._ready || !requiredEnd) {
149 			return true;
150 		} else {
151 			try {
152 				var end = audio.seekable.end()
153 				return (requiredEnd <= end);
154 			} catch(e) {
155 				logger.log(e);
156 				return false;
157 			}
158 		}
159 	}
160 
161 	this.play = function(name, volume, loop) {		
162 		if (this.muted) { return; }
163 
164 		if (volume === undefined) {
165 			volume = 1.0;
166 		}
167 		if (!this.canPlay(name)) {
168 			logger.info('Not ready yet');
169 			return;
170 		}
171 
172 		var audio = this._audios[this.oneChannelOnly ? 'AUDIO' : name];
173 		try {
174 			if (!audio.paused) {
175 				audio.pause(); // it glitches if you move currentTime while playing?
176 			}
177 			var startTime = 0;
178 			if (this.oneChannelOnly && this._map[name].start != null) {
179 				startTime = this._map[name].start;
180 			}
181 			if (audio.currentTime != startTime) {
182 				// logger.log('SEEK to play',name, startTime, 'from',audio.currentTime); // 
183 				audio.currentTime = startTime;
184 			}
185 			audio.volume = volume;
186 			audio.play();
187 			this._nowPlaying = this._map[name];
188 		} catch(e) {
189 			
190 		}
191 	}
192 	
193 	this.pause = function() {
194 		_.each(this._audios, function(audio, key) {
195 			audio.pause();
196 		}, this);
197 	}
198 
199 	this.playBackgroundMusic = function(name, volume) {
200 		if (this.muted) { return; }
201 
202 		if (this.oneChannelOnly) {
203 			return false; // cannot play bg music here.
204 		}
205 
206 		this._backgroundSoundPlaying = name;
207 		this.play(name, volume);
208 	}
209 
210 	this.pauseBackgroundMusic = function() {
211 		if (!this._backgroundSoundPlaying) { return; }
212 		this._audios[this._backgroundSoundPlaying].pause();
213 	}
214 
215 });
