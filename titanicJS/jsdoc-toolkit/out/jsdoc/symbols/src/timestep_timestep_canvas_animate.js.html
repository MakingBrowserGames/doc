 10 var anim_uid = 0;
 11 
 12 exports = function(subject, groupId) {
 13 	
 14 	// TODO: we have a circular import, so do the Application import on first use
 15 	if (typeof Application == 'undefined') {
 16 		import ..Application;
 17 		import ..View;
 18 	}
 19 	
 20 	var groupId = groupId || 0,
 21 		group = groups[groupId] || (groups[groupId] = new Group()),
 22 		animID = subject.__anim_id || (subject.__anim_id = '__anim_' + (++anim_uid)),
 23 		anim = group.get(animID);
 24 
 25 	if (!anim) {
 26 		anim = subject instanceof View 
 27 				? new ViewAnimator(subject, group)
 28 				: new Animator(subject, group);
 29 		
 30 		group.add(animID, anim);
 31 	}
 32 	
 33 	return anim;
 34 }

 40 var Group = Class(lib.PubSub, function(supr) {
 41 	this.init = function() {
 42 		this._anims = {};
 43 		this._pending = [];
 44 	}
 45 	
 46 	this.get = function(id) { return this._anims[id]; }
 47 
 48 	this.add = function(id, q) {
 49 		this._anims[id] = q;
 50 		q.id = id;
 51 		return q;
 52 	}
 53 	
 54 	this.isActive = function() {
 55 		for (var id in this._anims) {
 56 			if (this._anims[id].hasFrames()) { return true; }
 57 		}
 58 
 59 		return false;
 60 	}
 61 	
 62 	this.onAnimationFinish = function(anim) {
 63 		delete this._anims[anim.id];
 64 
 65 		if (!this.isActive()) {
 66 			// if called from a Finish event, republish it
 67 			this.publish('Finish');
 68 		}
 69 	}
 70 });

 80 var TRANSITIONS = [
 81 	transitions.easeInOut, // 0: default
 82 	transitions.linear,    // 1
 83 	transitions.easeIn,    // 2
 84 	transitions.easeOut,   // 3
 85 	transitions.easeInOut, // 4
 86 	transitions.bounce     // 5
 87 ];
 88 
 89 exports.linear = 1;
 90 exports.easeIn = 2;
 91 exports.easeOut = 3;
 92 exports.easeInOut = 4;
 93 exports.bounce = 5;
 94 
 95 function getTransition(n) {
 96 	return (typeof n == 'function' ? n : TRANSITIONS[n | 0]);
 97 }

122 var ObjectFrame = Class(Frame, function() {
123 	this.exec = function(tt, t, debug) {
124 		if (!this.base) {
125 			this.base = {};
126 			for (var key in this.target) {
127 				this.base[key] = this.subject[key];
128 			}
129 		}
130 		
131 		for (var key in this.target) {
132 			this.subject[key] = (this.target[key] - this.base[key]) * tt + this.base[key];
133 		}
134 		
135 		if (debug) {
136 			var changed = {};
137 			for (var key in this.target) {
138 				changed[key] = this.subject[key] + ' -> ' + this.target[key];
139 			}
140 			logger.log(this.duration, tt, JSON.stringify(changed));
141 		}
142 	}
143 });

145 // a ViewStyleFrame updates a view's style in exec
146 var ViewStyleFrame = Class(Frame, function() {
147 	
148 	this._resolvedDeltas = false;
149 	
150 	this.init = function(opts) {
151 		this.subject = opts.subject;
152 		this.target = JS.shallowCopy(opts.target);
153 		this.duration = opts.duration === 0 ? 0 : (opts.duration || 500);
154 		this.transition = getTransition(opts && opts.transition);
155 		this.onTick = opts.onTick;
156 	}
157 	
158 	this.resolveDeltas = function(againstStyle) {
159 		var style = this.target;
160 		this._resolvedDeltas = true;
161 		for (var key in style) {
162 			var baseKey = key.substring(1);
163 			if (key.charAt(0) == 'd' && !(key in againstStyle) && (baseKey in againstStyle)) {
164 				style[baseKey] = style[key] + againstStyle[baseKey];
165 				delete style[key];
166 			}
167 		}
168 	}
169 	
170 	this.exec = function(tt, t, debug) {
171 		var oldStyle = this._baseStyle || (this._baseStyle = this.subject.style.copy()),
172 			newStyle = this.target;
173 		
174 		if (!this._resolvedDeltas) { this.resolveDeltas(this._baseStyle); }
175 		
176 		var oldStyle = this._baseStyle;
177 		for (var key in newStyle) {
178 			if (key in oldStyle) {
179 				this.subject.style[key] = (newStyle[key] - oldStyle[key]) * tt + oldStyle[key];
180 			}
181 		}
182 		
183 		this.subject.needsRepaint();
184 		
185 		if (debug) {
186 			var changed = {};
187 			for (var key in newStyle) {
188 				changed[key] = this.subject.style[key] + ' -> ' + newStyle[key];
189 			}
190 			logger.log(Timer.now, this.duration, tt, JSON.stringify(changed));
191 		}
192 	}
193 	
194 	this.onInterrupt = function(newFrame) {
195 		// var preStyle = this.subject.style.copy();
196 
197 		// You might want to resolve any deltas against the post-committed style.
198 		// But I think this should be off by default.  You can commit the style,
199 		// resolve your deltas, then restore the old style yourself if you want.
200 		// I don't think you _always_ want this to be the behavior?
201 		//
202 		// this.commit(true);
203 		// newFrame.resolveDeltas(this.subject.style);
204 		
205 		// If you're animating multiple properties, and you've only
206 		// interrupted some of them, you may want the rest to continue.  
207 		// This only looks good if the timing is the same too -- to do 
208 		// this properly, you'd have to branch animations based on the 
209 		// style property being animated...
210 		//
211 		// for (var i in ViewStyle.keys) {
212 		// 	if (postCommitStyle[i] != preStyle[i] && !(i in newStyle)) {
213 		// 		newStyle[i] = postCommitStyle[i];
214 		// 	}
215 		// }
216 		
217 		// this.subject.style.update(preStyle);
218 	}
219 });

221 var Animator = exports.Animator = Class(lib.PubSub, function() {
222 	this.init = function(subject, group) {
223 		this.subject = subject;
224 		this._group = group;
225 		this.clear();
226 		this._isPaused = false;
227 	}
228 	
229 	this.clear = function() {
230 		this._elapsed = 0;
231 		this._queue = [];
232 		this._unschedule();
233 		return this;
234 	}
235 
236 	// this.getQueue = function() { return this._queue; }
237 	
238 	// Careful: pause will *not* fire the finish event, so anything pending the end of the
239 	// animation will have to wait until the animation is resumed.  
240 	this.pause = function() {
241 		if (!this._isPaused) {
242 			this._isPaused = true;
243 			this._unschedule();
244 		}
245 	}
246 	
247 	this.resume = function() {
248 		if (this._isPaused) {
249 			this._isPaused = false;
250 			this._schedule();
251 		}
252 	}
253 
254 	this._schedule = function() {
255 		if (!this._isScheduled) {
256 			this._isScheduled = true;
257 			Application.get().subscribe('Tick', this, 'onTick');
258 		}
259 	}
260 
261 	this._unschedule = function() {
262 		if (this._isScheduled) {
263 			this._isScheduled = false;
264 			Application.get().unsubscribe('Tick', this, 'onTick');
265 		}
266 	}
267 	
268 	this.isPaused = function() { return this._isPaused; }
269 	this.hasFrames = function() { return !!this._queue[0]; }
270 	
271 	this.wait = function(duration) {
272 		return this.then(new WaitFrame({duration: duration}));
273 	}
274 	
275 	this.buildFrame = function(opts) {
276 		if (typeof opts.target == 'function') {
277 			return new CallbackFrame(opts);
278 		}
279 		
280 		if (typeof opts.target == 'object') {
281 			return new ObjectFrame(opts);
282 		}
283 		
284 		return new WaitFrame(opts);
285 	}
286 	
287 	this.now = function(target, duration, transition, onTick) {
288 		transition = transition || (this._queue[0] ? exports.easeOut : exports.easeInOut);
289 		
290 		var nextFrame = target instanceof Frame
291 				? target
292 				: this.buildFrame({
293 						subject: this.subject,
294 						target: target,
295 						duration: duration,
296 						transition: transition,
297 						onTick: onTick
298 					});
299 		
300 		var frame = this._queue[0];
301 		frame && frame.onInterrupt(nextFrame);
302 		
303 		this.clear();
304 		return this.then(nextFrame);
305 	}
306 	
307 	this.then = function(target, duration, transition, onTick) {
308 		var nextFrame = target instanceof Frame
309 				? target
310 				: this.buildFrame({
311 						subject: this.subject,
312 						target: target,
313 						duration: duration,
314 						transition: transition,
315 						onTick: onTick
316 					});
317 		
318 		this._queue.push(nextFrame);
319 		this._schedule();
320 		return this;
321 	}
322 
323 	this.debug = function() { this._debug = true; return this; }
324 	
325 	this.commit = function() {
326 		this._elapsed = 0;
327 		for (var i = 0, p; p = this._queue[i]; ++i) {
328 			this._elapsed += p.duration;
329 		}
330 		
331 		this.next();
332 		return this;
333 	}
334 	
335 	this.onTick = function(dt) {
336 		this._elapsed += dt;
337 		this.next();
338 	}
339 	
340 	this.next = function() {
341 		var p,
342 			target;
343 		
344 		if (!this._queue[0]) { return; }
345 		
346 		while ((p = this._queue[0])) {
347 			var frameFinished = this._elapsed >= p.duration,
348 				t = frameFinished ? 1 : this._elapsed / p.duration,
349 				tt = p.transition(t);
350 			
351 			if (frameFinished) {
352 				this._elapsed -= p.duration;
353 			}
354 			
355 			try {
356 				p.exec(tt, t, this._debug);
357 				if (p.onTick) { p.onTick.call(p.subject, tt, t); }
358 			} finally {
359 				// if we haven't modified the queue in a callback, remove the frame if it is finished
360 				if (frameFinished && p == this._queue[0]) {
361 					this._queue.shift();
362 				}
363 
364 				// if we got paused during a callback or the 
365 				// frame is not finished yet, don't continue
366 				if (!frameFinished || this._isPaused) { return; }
367 			}
368 		}
369 		
370 		// nothing left in the queue!
371 		this._unschedule();
372 		this._group.onAnimationFinish(this);
373 	}
374 });
