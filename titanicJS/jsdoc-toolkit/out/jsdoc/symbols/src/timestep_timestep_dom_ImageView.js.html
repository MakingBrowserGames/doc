1 "use import";
  2 
  3 import .View;
  4 import timestep.Image as Image;
  5 
  6 function getImageURL(image) {
  7 	if (image.getSource) {
  8 		return image.getSource().src;
  9 	} else {
 10 		return image;
 11 	}
 12 }
 13 
 14 /**
 15  * @extends timestep.dom.View
 16  */
 17 exports = Class(View, function (supr) {
 18 
 19 	this.init = function (opts) {
 20 		opts = merge(opts, {
 21 			image: null
 22 		});
 23 
 24 		supr(this, "init", [opts]);
 25 
 26 		if (opts.image) {
 27 			this.setImage(opts.image); 
 28 		}
 29 		
 30 		if (opts.autoSize) {
 31 			this.doOnLoad(this, function(url, width, height) {
 32 				this.style.width = width;
 33 				this.style.height = height;
 34 			});
 35 		}
 36 	};
 37 
 38 	this.tick = function() {
 39 		if (this._img) {
 40 			var boundsHash = JSON.stringify(this._img.getBounds());
 41 			if (this._cachedBounds != boundsHash) {
 42 				this._cachedBounds = boundsHash;
 43 				this.updateImage();
 44 			}
 45 		}
 46 	}
 47 
 48 	this.getImage = function() { return this._img; }
 49 	this.setImage = function(img) {
 50 		if (!img) { return; }
 51 		if (!img.getBounds) {
 52 			img = new Image({url: img});
 53 		}
 54 		
 55 		this._img = img;
 56 
 57 		img.doOnLoad(this, 'updateImage');
 58 	}
 59 
 60 	this.updateImage = function() {
 61 		if (!this._img || !this._img.isReady()) { return; }
 62 		var img = this._img;
 63 		var s = this._node.style;
 64 		var bounds = img.getBounds();
 65 		var origW = img.getOrigW();
 66 		var origH = img.getOrigH();
 67 		var w = (bounds.width == -1 ? origW : bounds.width);
 68 		var h = (bounds.height == -1 ? origH : bounds.height);
 69 
 70 		//s.overflow = 'hidden';
 71 		s.backgroundImage = 'url("' + getImageURL(img) + '")';
 72 		s.backgroundPositionX = (bounds.x / (origW - bounds.width) * 100) + '%';
 73 		s.backgroundPositionY = (bounds.y / (origH - bounds.height) * 100) + '%';
 74 		s.backgroundSize = (100 * (origW / w) | 0) + '% ' + (100 * (origH / h) | 0) + '%';
 75 	}
 76 
 77 	this.getOrigW = function() { return this._img.getOrigW(); }
 78 	this.getOrigH = function() { return this._img.getOrigH(); }
 79 
 80 	this.doOnLoad = function() {
 81 		if (arguments.length == 1) {
 82 			this._img.doOnLoad(this, arguments[0]);
 83 		} else {
 84 			this._img.doOnLoad.apply(this._img, arguments);
 85 		}
 86 		return this;
 87 	}
 88 
 89 });
