1 "use import";
  2 
  3 import ..Sprite;
  4 import ..animate;
  5 import ..device;
  6 
  7 // TODO: inherit from Button
  8 // infer which animation frames we have when image loads
  9 // use pressed for toggled if toggled dne, etc
 10 
 11 /**
 12  * @extends timestep.Sprite
 13  */
 14 exports = Class(Sprite, function(supr) {
 15 	this.init = function(opts) {
 16 		opts = merge(opts, {
 17 			animations: {
 18 				'default': {frames: [[0, 0]], imageURL: opts.imageURL, width: opts.width, height: opts.height},
 19 				'hover': {frames: [[0, opts.height]], imageURL: opts.imageURL, width: opts.width, height: opts.height},
 20 				'pressed': {frames: [[0, opts.height * 2]], imageURL: opts.imageURL, width: opts.width, height: opts.height}
 21 			},
 22 			color: 'white',
 23 			font: 'bold 16px ' + device.defaultFontFamily
 24 		});
 25 
 26 		if ('disabledImageIndex' in opts) {
 27 			opts.animations.disabled = {frames: [[0, opts.height * opts.disabledImageIndex]], imageURL: opts.imageURL, width: opts.width, height: opts.height};
 28 		}
 29 		
 30 		supr(this, 'init', [opts]);
 31 		
 32 		this._label = opts.label || opts.text;
 33 		
 34 		this._isEnabled = 'enabled' in opts ? opts.enabled : true;
 35 		this._isPressed = false;
 36 		this._isToggled = false;
 37 		
 38 		this.style.color = opts.color;
 39 		this.style.font = opts.font;
 40 		this.setAnimation();
 41 	}
 42 	
 43 	this.setToggled = function(isToggled) {
 44 		if (this._isToggled != isToggled) {
 45 			this._isToggled = isToggled;
 46 			this.setAnimation();
 47 			this.publish('ToggleChanged', isToggled);
 48 		}
 49 	}
 50 	
 51 	this.isToggled = function() { return this._isToggled; }
 52 	
 53 	this.setAnimation = function() {
 54 		var hasDisabled = !!this._animations.disabled;
 55 		
 56 		if (this._isEnabled) {
 57 			if (!hasDisabled) { this.animate().then({opacity: 1}); }
 58 			
 59 			this.startAnimation(this._isPressed || this._isToggled ? 'pressed' : this.isInputOver() ? 'hover' : 'default');
 60 		} else if (hasDisabled) {
 61 			this.startAnimation('disabled');
 62 		} else {
 63 			animate(this).then({opacity: 0.5});
 64 		}
 65 	}
 66 	
 67 	this.setEnabled = function(isEnabled) {
 68 		if (isEnabled != this._isEnabled) {
 69 			this._isEnabled = isEnabled;
 70 			this.setAnimation();
 71 		}
 72 		return this;
 73 	}
 74 	
 75 	this.onInputStart = function() {
 76 		this._isPressed = true;
 77 		this.setAnimation();
 78 	}
 79 	
 80 	this.onInputSelect = function() {
 81 		this._isPressed = false;
 82 		this.setAnimation();
 83 		if (this._isEnabled) {
 84 			// TODO: ui widgets need a better way to delegate events
 85 			this.publish('Select');
 86 		}
 87 	}
 88 	
 89 	this.onInputOver = function() { this.setAnimation(); }
 90 	this.onDragStop = function() { this.setAnimation(); } 
 91 	this.onInputOut = function() {
 92 		if (this._isPressed) {
 93 			this._isPressed = false;
 94 		}
 95 		this.setAnimation();
 96 	}
 97 	
 98 	this.setLabel = function(label) { this._label = label; }
 99 	
100 	this.render = function(ctx) {
101 		supr(this, 'render', arguments);
102 		
103 		if (this._label) {
104 			ctx.fillStyle = this.style.color;
105 			ctx.textAlign = "center";
106 			ctx.textBaseline = "middle";
107 			ctx.font = this.style.font;
108 			var top = this.style.height / 2,// + this.style.font.match(/(\d+)px/)[1] * 0.25,
109 				left = this.style.width/2;
110 			ctx.fillText(this._label, left, top, this.style.width - 25);
111 		}
112 	}
113 });
