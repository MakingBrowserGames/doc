  7 function imageOnLoad(success, evt, failCount) {
  8 	if (success && !this.width) { // some browsers report onload before they can report the image width
  9 		if (failCount > 3) {
 10 			this.onLoad = this.onError = null;
 11 			return this.__cb.fire(false);
 12 		}
 13 		
 14 		return setTimeout(bind(this, imageOnLoad, success, evt, (failCount || 0) + 1), 0);
 15 	}
 16 	
 17 	this.__cb.fire(true);
 18 	this.onLoad = this.onError = null;
 19 }
