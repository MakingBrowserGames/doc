1 jsio('import .device');
  2 var Timer = device.get('Timer');
  3 
  4 var MAX_TICK = 10000; // ticks over 10 seconds will be considered too large to process
  5 exports.now = 0;
  6 exports.frames = 0;
  7 exports.reset = function() { this._last = null; }
  8 exports.tick = function(dt) {
  9 	try {
 10 		if (dt > MAX_TICK) {
 11 			exports.onLargeTick(dt, MAX_TICK);
 12 			dt = 1;
 13 		}
 14 		
 15 		exports.now += dt;
 16 		exports.frames++;
 17 		exports.onTick(dt);
 18 		ok = true;
 19 	} finally {
 20 		if (exports.debug && !ok) {
 21 			app.stopLoop()
 22 		}
 23 	}
 24 }
 25 
 26 /**
 27  * If our computer falls asleep, dt might be an insanely large number. 
 28  * If we're running a simulation of sorts, we don't want the computer
 29  * to freeze while computing 1000s of simulation steps, so just drop
 30  * this tick.  Anyone who is interested can listen for a call to 'onLargeTick'
 31  */
 32 exports.onLargeTick = function(largeDt, threshold) {
 33 	logger.warn('Dropping large tick: ' + largeDt + '; Threshold is set at: ' + threshold);
 34 }
 35 
 36 exports.onTick = function(dt) {}
 37 
 38 exports.debug = false;
 39

 43 exports.start = function(minDt) {
 44 	this.reset();
 45 	this.isRunning = true;
 46 	device.get('Timer').start(exports.tick, minDt);
 47 }
 48 
 49 exports.stop = function() {
 50 	this.reset();
 51 	this.isRunning = false;
 52 	device.get('Timer').stop();
 53 }
 54 
 55 exports.getTickProgress = function() {
 56 	var now = +new Date;
 57 	return (-(Timer.last || now) + now);
 58 }
