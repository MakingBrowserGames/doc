1 "use import";
  2 
  3 import squill.models.List;
  4 import timestep.View;
  5 import timestep.ScrollView;
  6 import timestep.input.InputEvent as InputEvent;
  7 
  8 /**
  9  * @extends timestep.ScrollView
 10  */
 11 exports = Class(timestep.ScrollView, function(supr) {
 12 	this.init = function(opts) {
 13 		
 14 		opts.scrollBounds = {
 15 			minX: 0,
 16 			minY: 0,
 17 			maxX: Number.MAX_VALUE,
 18 			maxY: Number.MAX_VALUE
 19 		};
 20 		
 21 		supr(this, 'init', [opts]);
 22 		
 23 		var listOpts = {
 24 			view: this,
 25 			getCell: opts.getCell,
 26 			sorter: opts.sorter,
 27 			selectable: opts.selectable,
 28 			dataSource: opts.dataSource,
 29 		};
 30 		
 31 		if ('isFixedSize' in opts) {
 32 			listOpts.isFixedSize = opts.isFixedSize;
 33 		}
 34 
 35 		if ('isTiled' in opts) {
 36 			listOpts.isTiled = opts.isTiled;
 37 		}
 38 		
 39 		this.model = new squill.models.List(listOpts);
 40 		this.selection = this.model.selection;
 41 		if (listOpts.isFixedSize) {
 42 			this.subscribe('Scrolled', this, function() {
 43 				this._needsModelRender = true;
 44 				this.needsRepaint();
 45 			});
 46 		}
 47 	}
 48 	
 49 	this.addCell = function(cellView) {
 50 		if (!cellView.style.width) {
 51 			cellView.style.width = this.style.width;
 52 		}
 53 		
 54 		this.addSubview(cellView);
 55 		this.needsRepaint(true);
 56 	}
 57 	
 58 	this.setMaxY = function(maxY) {
 59 		this._scrollBounds.minX = this._scrollBounds.maxX = 0;
 60 		this._scrollBounds.maxY = Math.max(0, maxY - this.style.height);
 61 	}
 62 	
 63 	this.render = function(ctx, opts) {
 64 		supr(this, 'render', arguments);
 65 
 66 		if (this._needsModelRender || this.model._needsSort) {
 67 			this._needsModelRender = false;
 68 			this.model.render(opts.viewport);
 69 		}
 70 		
 71 	}
 72 
 73 });
