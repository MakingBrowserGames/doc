 38 	this.init = function (opts) {
 39 
 40 		opts = merge(opts, {
 41 			// TOOD: default opts
 42 		});
 43 
 44 		supr(this, "init", [opts]);
 45 
 46 		// Things we'll need:
 47 		// Buffer object: Bufffffffer!
 48 		// Current viewport offset: x,y offset of the viewport from 0,0 of the map

121 	this.render = function (ctx) {
122 
123 		// GTFO if there isn't a map attached
124 		if (this._map == null) {
125 			return;
126 		}
127 
128 		var view = this.viewport;
129 
130 		if (this._bufferDirty) {
131 
132 			logger.info("Redrawing entire buffer!");
133 
134 			var start = new Point();
135 			var end = this._map.getTileXY(view.width, view.height);
136 
137 			if (this._renderTiles(start, end)) {
138 				// Successfully rendered tiles; buffer is no longer dirty
139 				this._bufferDirty = false;
140 			}
141 
142 		}
143 
144 		ctx.fillStyle = "rgb(0, 0, 255)";
145 		ctx.fillRect(0, 0, this.style.width, this.style.height);
146 
147 		this._renderBuffer(ctx);
148 
149 	};

151 	this._onResize = function () {
152 
153 		logger.info("Resizing viewport and buffer");
154 
155 		// Calculate bounds
156 		this._bounds = {
157 			minX: 0,
158 			minY: 0,
159 			maxX: (this._map.mapWidth - this.style.width - this._map.tileWidth),
160 			maxY: (this._map.mapHeight - this.style.height - this._map.tileHeight)
161 		};
162 
163 		// Resize the container for subviews
164 		this._contentView.style.width = this._map.mapWidth;
165 		this._contentView.style.height = this._map.mapHeight;
166 
167 		// Update size of the viewport in tiles
168 		this._viewportTileSize = {
169 			x: Math.ceil(this.style.width / this._map.tileWidth),
170 			y: Math.ceil(this.style.height / this._map.tileHeight)
171 		};
172 
173 		// Buffer size in tiles (viewport size + 1 tile on each axis)
174 		this._bufferTileSize = {
175 			x: this._viewportTileSize.x + 1,
176 			y: this._viewportTileSize.y + 1
177 		};
178 
179 		// Buffer size in pixels
180 		this._bufferSize = this._map.getMapXY(
181 			this._bufferTileSize.x,
182 			this._bufferTileSize.y
183 		);
184 
185 		// Some debugging
186 		logger.info("Size of viewport in pixels: " + this.style.width + "x" + this.style.height);
187 		logger.info("Size of viewport in tiles: " + this._viewportTileSize.x + "x" + this._viewportTileSize.y);
188 		logger.info("Size of buffer in tiles: " + this._bufferTileSize.x + "x" + this._bufferTileSize.y);
189 		logger.info("Size of buffer in pixels: " + this._bufferSize.x + "x" + this._bufferSize.y);
190 
191 		// Setup or adjust the buffer
192 		if (this._buffer) {
193 			// Resize existing buffer
194 			logger.info("Resizing existing buffer");
195 			this._buffer.width = this._bufferSize.x;
196 			this._buffer.height = this._bufferSize.y;
197 		} else {
198 			// Create a brand new buffer
199 			logger.info("Creating a new buffer");
200 			this._buffer = new GCCanvas({
201 				width: this._bufferSize.x,
202 				height: this._bufferSize.y
203 			});
204 		}
205 
206 		// Force buffer redraw since the sizes have changed
207 		this.clearBuffer();
208 
209 	};

211 	this._initImages = function () {
212 
213 		// Cleanup exisitng images
214 		this._images.forEach(function (item) {
215 			// Detach event handlers
216 			item._srcImg.onload = null;
217 			item._srcImg.onreload = null;
218 		}, this);
219 
220 		// Reset the images collection
221 		this._images = [];
222 
223 		// Create image objects for each tileset image
224 		var images = this._map.getImages();
225 		images.forEach(function (item) {
226 			var imageURL = util.path.resolveRelativePath(item);
227 			var img = new timestep.Image({
228 				url: imageURL
229 			});
230 			img._srcImg.onload = bind(this, "_onImageLoad");
231 			// NOTE: onreload only fires in NATIVE after the app pauses/resumes
232 			img._srcImg.onreload = bind(this, "_onImageLoad");
233 			this._images.push(img);
234 		}, this);
235 
236 	};

238 	this._onImageLoad = function () {
239 		logger.info("An image (re)loaded. Clearing buffer.");
240 		this.clearBuffer();
241 	};

243 	this._renderTiles = function (start, end, offset) {
244 
245 		var ctx = this._buffer.getContext("2d");
246 
247 		var offset = offset || { x: 0, y: 0};
248 		var drawOffset = this._map.getMapXY(offset.x, offset.y);
249 		//logger.log("Draw offset:", offset.x + ", " + offset.y);
250 
251 		for (var i = 0, j = this._map.layers.length; i < j; ++i) {
252 			var layer = this._map.layers[i];
253 
254 			for (var y = start.y; y <= end.y; ++y) {
255 				for (var x = start.x; x <= end.x; ++x) {
256 
257 					var gid = this._map.getTile(layer, x, y);
258 					if (gid < 1) {
259 						// Nothing to draw at this tile x,y; move along
260 						continue;
261 					}
262 
263 					var sprite = this._map.getSprite(gid);
264 					var image = this._images[sprite.tilesetIndex];
265 					if (!image.isReady()) {
266 						// Image not ready; force dirty state until images have loaded
267 						logger.info("Image not ready! Skipping...");
268 						return false;
269 					}
270 
271 					var draw = this._map.getMapXY(x - start.x, y - start.y);
272 
273 					ctx.drawImage(
274 						image._srcImg,
275 						sprite.x, sprite.y, sprite.width, sprite.height,
276 						draw.x + drawOffset.x, draw.y + drawOffset.y,
277 						this._map.tileWidth, this._map.tileHeight
278 					);
279 
280 				}
281 			}
282 
283 		}
284 
285 		return true;
286 
287 	};

289 	this._renderBuffer = function (ctx) {
290 
291 		var view = this.viewport;
292 		var origin = this._bufferOrigin;
293 		var offset = {
294 			x: ~~this._bufferOffset.x,
295 			y: ~~this._bufferOffset.y
296 		};
297 
298 		var tileWidth = this._map.tileWidth;
299 		var tileHeight = this._map.tileHeight;
300 
301 		// Offset of the buffer origin in pixels
302 		var originOffset = this._map.getMapXY(origin.x, origin.y);
303 
304 		// Coordinates to slice the buffer
305 		var slice = {
306 			x: (originOffset.x + offset.x),
307 			y: (originOffset.y + offset.y)
308 		};
309 
310 		if (origin.x == 0 && origin.y == 0) { // 1 draw
311 			this._copyBuffer(ctx, offset.x, offset.y, view.width, view.height, 0, 0);
312 		} else if (origin.y == 0) { // 2 draws
313 
314 			// right to left
315 			this._copyBuffer(
316 				ctx,
317 				slice.x, offset.y, // source x,y
318 				this._buffer.width - slice.x, view.height, // width/height
319 				0, 0 // dest x,y
320 			);
321 
322 			// left to right
323 			this._copyBuffer(
324 				ctx,
325 				0, offset.y, // source x,y
326 				slice.x - tileWidth, view.height, // width/height
327 				this._buffer.width - slice.x, 0 // dest x, y
328 			);
329 
330 		} else if (origin.x == 0) { // 2 draws
331 
332 			// bottom to top
333 			this._copyBuffer(
334 				ctx,
335 				offset.x, slice.y, // source x,y
336 				view.width, this._buffer.height - slice.y, // width/height
337 				0, 0 // dest x,y
338 			);
339 
340 			// top to bottom
341 			this._copyBuffer(
342 				ctx,
343 				offset.x, 0, // source x,y
344 				view.width, slice.y - tileHeight, // width/height
345 				0, this._buffer.height - slice.y // dest x,y
346 			);
347 
348 		} else { // 4 draws
349 
350 			// bottom-right to top-left
351 			this._copyBuffer(
352 				ctx,
353 				slice.x, slice.y, // source x,y
354 				this._buffer.width - slice.x, // width
355 				this._buffer.height - slice.y, // height
356 				0, 0 // dest x,y
357 			);
358 
359 			// top-left to bottom-right
360 			this._copyBuffer(
361 				ctx,
362 				0, 0, // source x,y
363 				slice.x - tileWidth, // width
364 				slice.y - tileHeight, // height
365 				this._buffer.width - slice.x, // dest x
366 				this._buffer.height - slice.y // dest y
367 			);
368 
369 			// top-right to bottom-left
370 			this._copyBuffer(
371 				ctx,
372 				slice.x, 0, // source x,y
373 				this._buffer.width - slice.x, // width
374 				slice.y - tileHeight, // height
375 				0, // dest x
376 				this._buffer.height - slice.y // dest y
377 			);
378 
379 			// bottom-left to top-right
380 			this._copyBuffer(
381 				ctx,
382 				0, slice.y, // source x,y
383 				slice.x - tileWidth, // width
384 				this._buffer.height - slice.y, // height
385 				this._buffer.width - slice.x, // dest x
386 				0 // dest y
387 			);
388 
389 		}
390 
391 	};

393 	this._copyBuffer = function (ctx, sx, sy, sw, sh, dx, dy) {
394 		if (sw <= 0 || sh <= 0) {
395 			// We're not even drawing 1 px wide or tall...GTFO
396 			return;
397 		}
398 		ctx.drawImage(
399 			this._buffer,
400 			sx, sy, sw, sh,
401 			dx, dy, sw, sh
402 		);
403 	};

405 	this._scroll = function (x, y) {
406 
407 		// _screeTileX/X = this._bufferOrigin
408 		// _offsetX/Y =  this._bufferOffset
409 
410 		var tileWidth = this._map.tileWidth;
411 		var tileHeight = this._map.tileHeight;
412 
413 		var bufferCtx = this._buffer.getContext("2d");
414 
415 		// Scroll the content view
416 		this._contentView.style.x += -x;
417 		this._contentView.style.y += -y;
418 
419 		// Update offsets with scoll amount
420 		this._viewportOffset.x += x;
421 		this._bufferOffset.x += x;
422 
423 		// Enforce bounds
424 		if (this._viewportOffset.x > this._bounds.maxX) {
425 			this._bufferOffset.x -= (this._viewportOffset.x - this._bounds.maxX);
426 			this._viewportOffset.x = this._bounds.maxX;
427 			scrollX *= -1;
428 			this._contentView.style.x = -this._bounds.maxX;
429 		} else if (this._viewportOffset.x < this._bounds.minX) {
430 			this._viewportOffset.x = this._bounds.minX;
431 			this._bufferOffset.x = 0;
432 			scrollX *= -1;
433 		}
434 
435 		var view = this.viewport;
436 
437 		var viewOffset = this._viewportOffset;
438 
439 		/*
440 		while (this._bufferOffset.x < 0) {
441 						this._bufferOffset.x += tileWidth;
442 		}
443 		*/
444 
445 		// Render columns of tiles along the X axis
446 		while (this._bufferOffset.x >= tileWidth) {
447 
448 			var startTile = this._map.getTileXY(view.x, view.y);
449 			var start = {
450 				x: startTile.x + this._bufferTileSize.x,
451 				y: 0
452 			};
453 
454 			var end = {
455 				x: startTile.x + this._bufferTileSize.x,
456 				y: this._viewportTileSize.y
457 			};
458 
459 			var offset = {
460 				x: this._bufferOrigin.x,
461 				y: 0
462 			};
463 
464 			//logger.info("Render strip: ", start.x + ", " + start.y, end.x + ", " + end.y, offset.x + ", " + offset.y);
465 
466 			this._renderTiles(start, end, offset);
467 
468 			this._bufferOffset.x -= tileWidth;
469 			++this._bufferOrigin.x;
470 
471 			if (this._bufferOrigin.x >= this._bufferTileSize.x) {
472 				this._bufferOrigin.x -= this._bufferTileSize.x;
473 			}
474 
475 		}
476 
477 		// TODO: Update with changes from X axis logic
478 		this._viewportOffset.y += y;
479 		this._bufferOffset.y += y;
480 		if (this._bufferOffset.y >= this._map.tileHeight) {
481 			this._bufferOffset.y -= this._map.tileHeight;
482 			++this._bufferOrigin.y;
483 		}
484 
485 		// Trigger repaint
486 		this.needsRepaint();
487 
488 	};
