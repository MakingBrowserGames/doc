1 "use import";
  2 
  3 import math.util;
  4 import math2D.Point as Point;
  5 import math2D.intersect as intersect;
  6 
  7 import timestep.ScrollView;
  8 import .TiledMapView;
  9 
 10 /**
 11  * @extends timestep.ScrollView
 12  */
 13 exports = Class(timestep.ScrollView, function (supr) {
 14 
 15 	this.init = function (opts) {
 16 
 17 		opts = merge(opts, {
 18 			drag: false,
 19 			bounce: false,
 20 			focusView: null
 21 		});
 22 
 23 		supr(this, "init", [opts]);
 24 
 25 		this.focusView = opts.focusView;
 26 
 27 		// Used for scrolling towards a specific offset
 28 		this._targetOffset = null;
 29 		this._scrollSpeed = 500; // units per second
 30 
 31 		this._mapView = new TiledMapView({
 32 			parent: this
 33 		});
 34 
 35 		// Republish the "selectTile" event
 36 		this._mapView.subscribe("selectTile", this, function (evt) {
 37 			this.publish("selectTile", evt);
 38 		});
 39 
 40 		this._mapView.subscribe("startTile", this, function (evt) {
 41 			this.publish("startTile", evt);
 42 		});
 43 
 44 		//this._staticViewID = 0;
 45 		this._staticViewGrid = [];
 46 		this._staticViewGridSize = {
 47 			x: (this.style.width * 2),
 48 			y: (this.style.height * 2)
 49 		};
 50 
 51 		this._offsetDelta = {
 52 			x: 0,
 53 			y: 0
 54 		};
 55 
 56 	};
 57 
 58 	this.setOffset = function (x, y) {
 59 		var offset = this.getOffset();
 60 		supr(this, "setOffset", arguments);
 61 		var newOffset = this.getOffset();
 62 		this._offsetDelta.x += (newOffset.x - offset.x);
 63 		this._offsetDelta.y += (newOffset.y - offset.y);
 64 		var sync = false;
 65 		if (Math.abs(this._offsetDelta.x) > (this.style.width / 2)) {
 66 			this._offsetDelta.x = 0;
 67 			sync = true;
 68 		}
 69 		if (Math.abs(this._offsetDelta.y) > (this.style.height / 2)) {
 70 			this._offsetDelta.y = 0;
 71 			sync = true;
 72 		}
 73 		if (sync) {
 74 			this.syncStaticSubviews();
 75 		}
 76 	};
 77 
 78 	this.getViewport = function () {
 79 		var offset = this.getOffset();
 80 		return {
 81 			x: Math.abs(offset.x),
 82 			y: Math.abs(offset.y),
 83 			width: this.style.width,
 84 			height: this.style.height
 85 		};
 86 	};
 87 
 88 	this.syncStaticSubviews = function () {
 89 
 90 		// logger.info("Syncing static subviews...");
 91 
 92 		var grid = this._staticViewGrid;
 93 		var size = this._staticViewGridSize;
 94 		var viewport = this.getViewport();
 95 		var halfW = (viewport.width / 2);
 96 		var halfH = (viewport.height / 2);
 97 		viewport.x -= halfW;
 98 		viewport.y -= halfH;
 99 		viewport.width *= 2;
100 		viewport.height *= 2;
101 
102 		var start = {
103 			x: ~~(viewport.x / size.x),
104 			y: ~~(viewport.y / size.y)
105 		};
106 
107 		var end = {
108 			x: Math.ceil((viewport.x + viewport.width) / size.x),
109 			y: Math.ceil((viewport.y + viewport.height) / size.y)
110 		};
111 
112 		// logger.info("Start/End: " + start.x + ", " + start.y + " / " + end.x + ", " + end.y);
113 
114 		//var visibleCount = 0;
115 
116 		for (var y = 0, endY = grid.length; y < endY; ++y) {
117 			if (!grid[y]) {
118 				continue;
119 			}
120 			for (var x = 0, endX = grid[y].length; x < endX; ++x) {
121 				if (!grid[y][x]) {
122 					continue;
123 				}
124 				var gridVisible = (
125 					y >= start.y &&
126 					y <= end.y &&
127 					x >= start.x &&
128 					x <= end.y
129 				);
130 				//logger.info("Grid " + x + ", " + y + " is " + (gridVisible) ? "visible" : "not visible");
131 				var views = grid[y][x];
132 				for (var i = 0, j = views.length; i < j; ++i) {
133 					var view = views[i];
134 					var superview = view.getSuperview();
135 					//if (gridVisible) { visibleCount++; };
136 					if (gridVisible && superview == null) {
137 						// Visible and not attached to view stack
138 						//logger.info("Showing a view");
139 						this.addSubview(view);
140 					} else if (!gridVisible && superview != null) {
141 						// Not visible and attached to view stack
142 						//logger.info("Hiding a view");
143 						this.removeSubview(view);
144 					}
145 				}
146 			}
147 		}
148 
149 		//logger.info(visibleCount + " view visible");
150 
151 	};
152 
153 	this._getStaticViewGridPosition = function (view) {
154 		return {
155 			x: Math.floor(view.style.x / this._staticViewGridSize.x),
156 			y: Math.floor(view.style.y / this._staticViewGridSize.y)
157 		};
158 	};
159 
160 	this.addStaticSubview = function (view) {
161 		view.__bounds = {
162 			x: view.style.x,
163 			y: view.style.y,
164 			width: view.style.width,
165 			height: view.style.height
166 		};
167 		var pos = view.__gridPos = this._getStaticViewGridPosition(view);
168 		var grid = this._staticViewGrid;
169 		if (!grid[pos.y]) {
170 			grid[pos.y] = [];
171 		}
172 		if (!grid[pos.y][pos.x]) {
173 			grid[pos.y][pos.x] = [];
174 		}
175 		grid[pos.y][pos.x].push(view);
176 		// Add the subview to the view stack if it's visible
177 		if (intersect.rectAndRect(this.getViewport(), view.__bounds)) {
178 			this.addSubview(view);
179 		}
180 	};
181 
182 	this.removeStaticSubview = function (view) {
183 		var pos = view.__gridPos;
184 		var list = this._staticViewGrid[pos.y][pos.x];
185 		var index = list.indexOf(view);
186 		list.splice(index, 1);
187 		this.removeSubview(view);
188 	};
189 
190 	this.__defineSetter__("focusView", function (value) {
191 		this._focusView = value;
192 	});
193 
194 	this.flushBuffer = function () {
195 		this._mapView.flushBuffer();
196 	};
197 
198 	this.setMap = function (map) {
199 
200 		var mapWidth = (map.width * map.tileWidth);
201 		var mapHeight = (map.height * map.tileHeight);
202 
203 		// The content is the size of the map
204 		this._contentView.style.width = mapWidth;
205 		this._contentView.style.height = mapHeight;
206 
207 		// Set bounds so we can't scroll off the map
208 		this.setScrollBounds({
209 			minX: 0,
210 			minY: 0,
211 			maxX: (mapWidth - this.style.width),
212 			maxY: (mapHeight - this.style.height)
213 		});
214 
215 		this._mapView.setMap(map);
216 		this._map = map;
217 
218 		if (map.orientation == "isometric") {
219 			// Default to the center of the map for iso
220 			this.setOffset(
221 				-((map.mapWidth / 2) - (this.style.width / 2)),
222 				-((map.mapHeight / 2) - (this.style.height / 2))
223 			);
224 		}
225 
226 	};
227 
228 	this._updateFocusView = function (dt) {
229 		var focus = this._focusView;
230 		if (!focus) {
231 			return;
232 		}
233 		this.setOffset(
234 			-Math.floor(focus.style.x + (focus.style.width / 2) - (this.style.width / 2)),
235 			-Math.floor(focus.style.y + (focus.style.height / 2) - (this.style.height / 2))
236 		);
237 	};
238 
239 	this._updateScrollTarget = function (dt) {
240 
241 		var target = this._targetOffset;
242 		if (target == null || dt < 1) {
243 			// No target offset
244 			return;
245 		}
246 
247 		var offset = this.getOffset();
248 		if (offset.x == target.x && offset.y == target.y) {
249 			// Reached the target!
250 			this._targetOffset = null;
251 			this.publish("scrollTargetReached");
252 			return;
253 		}
254 
255 		var diff = new Point(target).subtract(offset);
256 		var distance = ((this._scrollSpeed / 1000) * dt);
257 		var offsetChange = new Point();
258 
259 		// Scroll along the X axis
260 		if (offset.x !== target.x) {
261 			if (Math.abs(diff.x) <= distance) {
262 				offsetChange.x = diff.x;
263 			} else {
264 				offsetChange.x = (distance * math.util.sign(diff.x));
265 			}
266 		}
267 
268 		// Scroll along the Y axis
269 		if (offset.y !== target.y) {
270 			if (Math.abs(diff.y) <= distance) {
271 				offsetChange.y = diff.y;
272 			} else {
273 				offsetChange.y = (distance * math.util.sign(diff.y));
274 			}
275 		}
276 
277 		// Apply offset changes
278 		if (offsetChange.x != 0 || offsetChange.y != 0) {
279 			this.addOffset(offsetChange);
280 		}
281 
282 	};
283 
284 	this.tick = function (dt) {
285 		this._updateScrollTarget(dt);
286 		this._updateFocusView(dt);
287 	};
288 
289 	// Scrolls the map so it centers on a specific tile
290 	this.scrollToTile = function (x, y) {
291 		var offset = this._map.getMapXY(x, y);
292 		offset.x -= ((this.style.width / 2) - (this._map.tileWidth / 2));
293 		offset.y -= ((this.style.height / 2) - (this._map.tileWidth / 2));
294 		offset = new Point(offset).scale(-1);
295 		this._targetOffset = this._applyBounds(offset);
296 	};
297 
298 	this._applyBounds = function (offset) {
299 		var bounds = this.getStyleBounds();
300 		return {
301 			x: math.util.clip(offset.x, bounds.minX, bounds.maxX),
302 			y: math.util.clip(offset.y, bounds.minY, bounds.maxY)
303 		};
304 	};
305 
306 	this.getViewportBounds = function () {
307 		return this._mapView.getViewportBounds();
308 	};
309 
310 });
