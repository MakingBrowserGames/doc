1 "use import";
  2 
  3 import timestep.View;
  4 import timestep.device;
  5 
  6 /**
  7  * @extends timestep.View
  8  */
  9 exports = Class(timestep.View, function(supr) {
 10 	this.init = function(opts) {
 11 		supr(this, 'init', arguments);
 12 		
 13 		opts = Class.defaults(opts, {
 14 			text: '',
 15 			font: 'bold 14px ' + timestep.device.defaultFontFamily,
 16 			color: 'white'
 17 		});
 18 		
 19 		this.style.font = opts.font;
 20 		this.style.color = opts.color;
 21 		this._text = opts.text;
 22 	}
 23 	
 24 	this.setText = function(text) { this._text = text; }
 25 	
 26 	this.render = function(ctx) {
 27 		ctx.fillStyle = this.style.color;
 28 		ctx.font = this.style.font;
 29 		ctx.fillText(this._text, 0, 0);
 30 	}
 31 });
