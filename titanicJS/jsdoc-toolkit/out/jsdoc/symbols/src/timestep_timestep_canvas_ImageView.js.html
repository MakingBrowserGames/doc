1 "use import";
  2 
  3 import ..View;
  4 import ..Image;
  5 
  6 /*
  7 
  8 autoSize:
  9 	- 'none': draw image at original pixel width/height
 10 	- 'fit': (default) draw image to fit in view width/height
 11 	- 'proportional': draw image to fit in view without changing image proportions
 12 	- 'resize': scale view when image loads to the dimensions of the image
 13 
 14 */
 15 
 16 /**
 17  * @extends timestep.View
 18  */
 19 var ImageView = exports = Class(View, function(supr) {
 20 
 21 	this.init = function(opts) {
 22 		supr(this, 'init', arguments);
 23 		opts = merge(opts, {
 24 			image: null,
 25 			autoSize: false,
 26 			scaleMethod: false
 27 		});
 28 		
 29 		this._scaleMethod = opts.scaleMethod;
 30 
 31 		if (opts.image) {
 32 			this.setImage(opts.image, opts);
 33 		}
 34 	}
 35 
 36 	this.getImage = function() { return this._img; }
 37 	this.setImage = function(img, opts) {
 38 		if (typeof img == 'string') {
 39 			this._img = new Image({url: img});
 40 		} else {
 41 			this._img = img;
 42 		}
 43 
 44 		if (this._img) {
 45 			if (opts && opts.autoSize) {
 46 				this._img.doOnLoad(this, 'autoSize', opts.autoSize);
 47 			}
 48 
 49 			this._img.doOnLoad(this, 'needsRepaint');
 50 		}
 51 	}
 52 	
 53 	this.doOnLoad = function() {
 54 		if (arguments.length == 1) {
 55 			this._img.doOnLoad(this, arguments[0]);
 56 		} else {
 57 			this._img.doOnLoad.apply(this._img, arguments);
 58 		}
 59 		return this;
 60 	}
 61 	
 62 	this.autoSize = function(method, url, width, height) {
 63 		switch (method) {
 64 			case 'resize':
 65 			case true:
 66 				this._scaleMethod = false;
 67 				this.style.width = width;
 68 				this.style.height = height;
 69 				break;
 70 			default:
 71 				this._scaleMethod = method || false;
 72 				break;
 73 		}
 74 	}
 75 	
 76 	this.getOrigW = function() { return this._img.getOrigW(); }
 77 	this.getOrigH = function() { return this._img.getOrigH(); }
 78 
 79 	this.render = function(ctx) {
 80 		if (this.style.opacity == 0) { return; }
 81 
 82 		var s = this.style;
 83 		if (this._img) {
 84 			if (!this._scaleMethod) {
 85 				if (this._circle) {
 86 					this._img.render(ctx, -s.width / 2, -s.height / 2, s.width, s.height);
 87 				} else {
 88 					this._img.render(ctx, 0, 0, s.width, s.height);
 89 				}
 90 				return;
 91 			}
 92 			
 93 			if (this._scaleMethod == 'none') {
 94 				if (this._circle) {
 95 					var w = this._img.getOrigW(),
 96 						h = this._img.getOrigH();
 97 					this._img.render(ctx, -w / 2, -h / 2, w, h);
 98 				} else {
 99 					this._img.render(ctx, 0, 0);
100 				}
101 				return;
102 			}
103 		}
104 	}
105 	
106 	this.getTag = function() { return 'ImageView' + this.uid + ':' + (this._img && this._img._map.url.substring(0, 16)); }
107 });
