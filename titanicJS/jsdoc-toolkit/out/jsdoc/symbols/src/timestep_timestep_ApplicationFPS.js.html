 13 function strokeRect(ctx, rect, color) {
 14 	ctx.fillStyle = color;
 15 	ctx.fillRect(rect.x, rect.y, 1, rect.height);
 16 	ctx.fillRect(rect.x, rect.y, rect.width, 1);
 17 	ctx.fillRect(rect.x + rect.width - 1, rect.y, 1, rect.height);
 18 	ctx.fillRect(rect.x, rect.y + rect.height - 1, rect.width, 1);
 19 };
 20 
 21 function fillRect(ctx, rect, color) {
 22 	ctx.fillStyle = color;
 23 	ctx.fillRect(rect.x, rect.y, rect.width, rect.height);
 24 };

 26 var Graph = Class(function() {
 27 	this.init = function(opts) {
 28 		Canvas = device.get('Canvas');
 29 
 30 		this._width = opts.width;
 31 		this._height = opts.height;
 32 
 33 		this._maxValue = opts.maxValue;
 34 		this._backgroundColor = opts.backgroundColor;
 35 		this._colors = opts.colors;
 36 		this._axisColor = opts.axisColor;
 37 
 38 		this._canvas = new Canvas({width: opts.width, height: opts.height});
 39 		this._ctx = this._canvas.getContext('2d');
 40 		this._ctx.fillStyle = opts.backgroundColor;
 41 		this._ctx.fillRect(0, 0, opts.width, opts.height);
 42 
 43 		this._offset = 0;
 44 		this._index = 0;
 45 	};
 46 
 47 	this.addValues = function(values, timeAxis) {
 48 		var ctx = this._ctx,
 49 			width = this._width,
 50 			height = this._height,
 51 			value,
 52 			maxValue = this._maxValue,
 53 			x = (this._offset + this._index) % width,
 54 			y = this._height,
 55 			n,
 56 			i, j;
 57 
 58 		ctx.fillStyle = this._backgroundColor;
 59 		ctx.fillRect(x, 0, 2, height);
 60 
 61 		i = values.length;
 62 		while (i) {
 63 			value = values[--i];
 64 			if (value > maxValue) {
 65 				value = maxValue;
 66 			}
 67 			n = ~~(value / maxValue *  height);
 68 			ctx.fillStyle = this._colors[i];
 69 			ctx.fillRect(x, height - n, 2, n);
 70 		}
 71 
 72 		ctx.fillStyle = this._axisColor;
 73 		if (timeAxis) {
 74 			ctx.fillRect(x, 0, 1, height);
 75 		}
 76 		ctx.fillRect(x, 25, 2, 1);
 77 		ctx.fillRect(x, 50, 2, 1);
 78 		ctx.fillRect(x, 75, 2, 1);
 79 
 80 		if (this._index < width) {
 81 			this._index += 2;
 82 		} else {
 83 			this._offset += 2;
 84 			if (this._offset >= width) {
 85 				this._offset = 0;
 86 			}
 87 		}
 88 	};
 89 
 90 	this.render = function(ctx, x, y) {
 91 		var offset = this._offset,
 92 			width = this._width,
 93 			height = this._height;
 94 
 95 		if (offset === 0) {
 96 			ctx.drawImage(this._canvas, 0, 0, width, height, x, y, width, height);
 97 		} else {
 98 			ctx.drawImage(this._canvas, offset, 0, width - offset, height, x, y, width - offset, height);
 99 			ctx.drawImage(this._canvas, 0, 0, offset, height, width - offset + x, y, offset, height);
100 		}
101 	};
102 });

104 exports = Class(function() {
105 	this.init = function(opts) {
106 		this._application = opts.application;
107 
108 		this._time = +(new Date()) + 1000;
109 		this._frames = 0;
110 		this._fps = 20;
111 		this._dt = 10;
112 
113 		this._minimized = true;
114 
115 		var width = 200,
116 			height = 100;
117 
118 		this._rectTop = new Rect(1, 1, width, 16);
119 		this._rectMin = new Rect(0, 0, 24, 15);
120 		this._rectMax = new Rect(0, 0, width + 2, height + 17);
121 		this._rect = this._rectMin;
122 
123 		this._rectFPS = new Rect(141, 1, 30, 15);
124 		this._rectDT = new Rect(171, 1, 30, 15);
125 
126 		this._borderColor = 'rgb(100,100,150)';
127 		this._backgroundColor = '#17182E';
128 		this._textColor = '#FFFFFF';
129 
130 		this._viewMode = viewModes.DT;
131 
132 		this._graphs = {}
133 		this._graphs[viewModes.FPS] = new Graph({
134 			width: width,
135 			height: height,
136 			maxValue: 60,
137 			backgroundColor: this._backgroundColor,
138 			colors: ['rgba(170,170,252,0.3)', 'rgb(170,170,252)'],
139 			axisColor: '#FFFFFF'
140 		});
141 		this._graphs[viewModes.DT] = new Graph({
142 			width: width,
143 			height: height,
144 			maxValue: 66,
145 			backgroundColor: this._backgroundColor,
146 			colors: ['rgba(255,0,0,0.5)', 'rgb(254,255,170)'],
147 			axisColor: '#FFFFFF'
148 		});
149 	};
150 
151 	this.tick = function(dt) {
152 		var time = +(new Date());
153 		if (time > this._time) {
154 			this._time = time + 1000;
155 			this._fps = this._frames;
156 			this._frames = 1;
157 		} else {
158 			this._frames++;
159 		}
160 
161 		var events = this._application.getEvents();
162 		if (events.length) {
163 			this._handleEvents(events);
164 		}
165 
166 		this._dt = this._dt * 0.8 + dt * 0.2;
167 
168 		if (!this._minimized) {
169 			switch (this._viewMode) {
170 				case viewModes.DT:
171 					this._graphs[viewModes.DT].addValues([this._dt, dt], (this._frames === 1));
172 					break;
173 
174 				case viewModes.FPS:
175 					this._graphs[viewModes.FPS].addValues([this._fps, 1000 / dt], (this._frames === 1));
176 					break;
177 			}
178 		}
179 	};
180 
181 	this._handleEvents = function(events) {
182 		var types = input.eventTypes,
183 			event,
184 			i = events.length;
185 
186 		while (i) {
187 			event = events[--i];
188 			switch (event.type) {
189 				case types.START:
190 					if (intersect.ptAndRect(event.srcPt, this._rect)) {
191 						if (this._minimized) {
192 							this._minimized = false;
193 						} else {
194 							if (intersect.ptAndRect(event.srcPt, this._rectFPS)) {
195 								this._viewMode = viewModes.FPS;
196 							} else if (intersect.ptAndRect(event.srcPt, this._rectDT)) {
197 								this._viewMode = viewModes.DT;
198 							} else {
199 								this._minimized = true;
200 							}
201 						}
202 						this._rect = this._minimized ? this._rectMin : this._rectMax;
203 					}
204 					break;
205 			}
206 		}
207 	};
208 
209 	this.render = function(ctx) {
210 		ctx.save();
211 
212 		ctx.textBaseline = 'top';
213 		ctx.textAlign = 'center';
214 		ctx.font = '12px Verdana';
215 
216 		if (this._minimized) {
217 			fillRect(ctx, this._rect, this._backgroundColor);
218 		} else {
219 			fillRect(ctx, this._rectTop, this._backgroundColor);
220 			this._graphs[this._viewMode].render(ctx, 1, 17);
221 			ctx.fillStyle = this._borderColor;
222 			ctx.fillRect(0, 16, this._rect.width, 1);
223 
224 			var fpsMode = (this._viewMode === viewModes.FPS),
225 				dtMode = (this._viewMode === viewModes.DT);
226 
227 			fillRect(ctx, this._rectFPS, fpsMode ? this._borderColor : this._backgroundColor);
228 			ctx.fillStyle = fpsMode ? this._textColor : this._borderColor;
229 			ctx.fillText('FPS', this._rectFPS.x + this._rectFPS.width / 2, 0);
230 
231 			fillRect(ctx, this._rectDT, dtMode ? this._borderColor : this._backgroundColor);
232 			ctx.fillStyle = dtMode ? this._textColor : this._borderColor;
233 			ctx.fillText('DT', this._rectDT.x + this._rectDT.width / 2, 0);
234 		}
235 
236 		strokeRect(ctx, this._rect, this._borderColor);
237 
238 		ctx.fillStyle = this._textColor;
239 		ctx.fillText(this._fps, 12, 0);
240 		ctx.fillText(this._fps, 12, 0);
241 
242 		ctx.restore();
243 	};
244 
245 	this.getFPS = function() {
246 		return this._fps;
247 	};
248 });
