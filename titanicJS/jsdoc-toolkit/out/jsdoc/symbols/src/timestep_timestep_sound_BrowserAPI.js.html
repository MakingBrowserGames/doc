1 "use import";
  2 
  3 // An API for playing named sounds. Sounds can be given a single file source
  4 // or multiple sources which will be chosen at random at play time. Sounds may
  5 // also be given a default volume and loop boolean.
  6 ////////////////////////
  7 // only one background sound can be played at a time. They are streamed, not
  8 // preloaded, on native.
  9 
 10 import lib.PubSub;
 11 import ..device;
 12 from util.underscore import _;
 13 
 14 /**
 15  * @extends lib.PubSub
 16  */
 17 exports = Class(lib.PubSub, function(supr) {
 18 	var pauseAll = function() {
 19 		// this is used as an "instance method" for the sound objects held
 20 		// in the _sound array of BrowserAPI instances.
 21 			this.sources.forEach(function(source) {
 22 				if(!source.paused){
 23 					source.pause();
 24 					try{
 25 						source.currentTime = 0;
 26 					} catch (e){
 27 					}
 28 				}
 29 			});
 30 	};
 31 
 32 	this.init = function(opts) {
 33 		supr(this, 'init', [opts]);
 34 		this._path = opts.path;
 35 		this._map = opts.map;
 36 		this._sounds = {};
 37 		this._backgroundSoundPlaying = null;
 38 		// determine whether browser supports mp3 or ogg. Default to mp3 if
 39 		// both are supported
 40 		
 41 		var sound = new Audio();
 42 		this._type = sound.canPlayType("audio/mpeg") ? '.mp3'
 43 			: sound.canPlayType("audio/ogg") ? '.ogg' : '';
 44 		if (!this._type) {
 45 			logger.log('warning: could not determine sound support type');
 46 		}
 47 
 48 		// add sounds to the audio API's list of sounds and
 49 		// preload them if appropriate
 50 		for (var key in this._map) {
 51 			var item = this._map[key];
 52 			this.addSound(key, item);
 53 		}
 54 
 55 
 56 	}
 57 
 58 	this.addSound = function(name, opts) {
 59 		// creates a sound. Loads non-background sounds
 60 		var path = this._path.replace(/\/$/, '');
 61 		var sound = this._sounds[name] = {};
 62 		sound.sources = [];
 63 		sound.pause = pauseAll;
 64 
 65 		if (!opts) { opts = {}; }
 66 
 67 		// if a list of file names is given in sources, load them as alternative
 68 		// clips for this sound. Else, assume the only clip for this sound
 69 		// is the file with its same name
 70 		if (!opts.sources) { opts.sources = [name]; }
 71 		opts.sources.forEach(bind(this, function(item) {
 72 			var fullSource = path + '/' + item + this._type;
 73 			logger.log('making sound for file', fullSource);
 74 			
 75 			var audio = new Audio();
 76 			/* 
 77 			 * HACK XXX
 78 			 * when loading many sounds, if preload is set to auto (the default)
 79 			 * browsers (chrome 18, at least) will not play sound and crash
 80 			 * by changing this to 'none', the audio will load and play normally
 81 			 * This isn't a great fix because no sounds are preloaded, but its
 82 			 * better than crashing for now.
 83 			 */
 84 			audio.preload = 'none';
 85 			audio.src = fullSource;
 86 			audio.loop = opts.loop || opts.background;
 87 			audio.volume = opts.volume !== undefined ? opts.volume : 1.0;
 88 			sound.sources.push(audio);
 89 		}));
 90 		sound.background = opts.background;
 91 	}
 92 
 93 
 94 	this.setMuted = function(muted) {
 95 		this._muted = muted;
 96 		if(muted){
 97 			this._pauseAll();
 98 		}
 99 	}
100 
101 	this.setVolume = function(name, volume) {
102 		sound = this._sounds[name];
103 		sound.sources.forEach(function(audio){
104 			audio.volume = volume;
105 		});
106 	}
107 
108 	this._pauseAll = function(){
109 		for (var key in this._sounds) {
110 			var sound = this._sounds[key];
111 			sound.sources.forEach(function(source){
112 				if(!source.paused){source.pause();}
113 			});
114 		}
115 	}
116 
117 	this.play = function(name) {
118 		var sound = this._sounds[name];
119 		if(!sound){
120 			logger.log("warning: no sound of that name");
121 			return;
122 		}
123 		
124 		// first stop all other instances of this sound if they are in
125 		// middle of playback
126 		this.pause(name);
127 		sound.sources.forEach(function(item) {
128 			try{
129 				item.currentTime = 0;
130 			} catch (e) {
131 			}
132 		});
133 
134 		var index = Math.random() * sound.sources.length | 0;
135 		var audio = sound.sources[index];
136 		logger.log('playing sound', audio);
137 		audio.play();
138 	}
139 
140 	this.pause = function(name) {
141 		var sound = this._sounds[name];
142 		sound.sources.forEach(function(source){
143 			if(!source.paused){
144 				source.pause();
145 			}
146 		});
147 	}
148 	
149 	this.playBackgroundMusic = function(name) {
150 		if (this._muted) { return; }
151 		this.pauseBackgroundMusic();
152 		logger.log('playing background music', name);
153 		this._backgroundSoundPlaying = name;
154 		this.play(name);
155 	}
156 
157 	this.pauseBackgroundMusic = function() {
158 		if (!this._backgroundSoundPlaying) { return; }
159 		this._sounds[this._backgroundSoundPlaying].pause();
160 	}
161 
162 });
