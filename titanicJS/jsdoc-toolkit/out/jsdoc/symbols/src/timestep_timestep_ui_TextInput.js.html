1 "use import";
  2 
  3 import timestep.Application as Application;
  4 import timestep.ui.UIView;
  5 from util.browser import $;
  6 import timestep.device;
  7 
  8 var TextBox = timestep.device.get('TextBox');
  9 
 10 /**
 11  * @extends timestep.ui.UIView
 12  */
 13 var TextInputView = exports = Class(timestep.ui.UIView, function(supr) {
 14 	
 15 	this.init = function(opts) {
 16 		opts = merge(opts, {
 17 			color: 'black',
 18 			height: 20
 19 		});
 20 		
 21 		this._props = {};
 22 		
 23 		supr(this, 'init', [opts]);
 24 		
 25 		this.updatePosition();
 26 	};
 27 	
 28 	this.getOpacity = function() {
 29 		var view = this;
 30 		var o = this.style.opacity;
 31 		while (view._superview) {
 32 			view = view._superview;
 33 			o *= view.style.opacity;
 34 		}
 35 		
 36 		return o;
 37 	}
 38 	
 39 	this._clearSuperview = function() {
 40 		supr(this, '_clearSuperview', arguments);
 41 		
 42 		this._textBox.destroy();
 43 		this._textBox = null;
 44 	}
 45 	
 46 	this.updatePosition = function() {
 47 		var app = this.getApp();
 48 		if (!app) { return; }
 49 		
 50 		var s = this.style;
 51 		var a = this._props;
 52 		var b = this.getPosition();
 53 		if (this._textBox && a.x == b.x && a.y == b.y && a.width == b.width && a.height == b.height && a.opacity == s.opacity) {
 54 			return;
 55 		}
 56 		
 57 		var box = this._textBox;
 58 		if (!box) {
 59 			box = this._textBox = new TextBox({color: this._opts.color});
 60 			this._textBox.onChange = bind(this, '_onValueChanged');
 61 			box.setValue(this._opts.value || '');
 62 		}
 63 		
 64 		box.setApp(app);
 65 		box.setPosition(b);
 66 		box.setDimensions(b);
 67 		box.setOpacity(s.opacity);
 68 		box.setVisible(true);
 69 		
 70 		this._props = b;
 71 	};
 72 	
 73 	this._onValueChanged = function() {
 74 		this.publish('ValueChanged', this.getValue());
 75 	}
 76 
 77 	this.setValue = function(value) { return this._textBox.setValue(value); }
 78 	this.getValue = function() { return this._textBox.getValue(); }
 79 	
 80 	this.updateStyle = function(style) {
 81 		this._el.style = JS.merge(style, this._el.style);
 82 	};
 83 	
 84 	this.updateOpts = function(opts) {
 85 		this.opts = JS.merge(opts, this.opts);
 86 	};
 87 	
 88 	this.render = function() {
 89 		this.updatePosition();
 90 	};
 91 	
 92 });
