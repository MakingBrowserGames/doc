1 "use import";
  2 
  3 from util.browser import $;
  4 import std.js as JS;
  5 
  6 var soundManager = null;
  7 
  8 var SoundManager = Class(function(){
  9 	// This is largely a workaround for Chrome bugs (as of 8.0.552 OSX) 
 10 	// Firefox 4 beta 9+ can use the HTML5 audio API as expected without
 11 	// such a complex workaround. Chrome will only play a clip reliably if 
 12 	// a new Audio object is created, and creating too many audio objects 
 13 	// will crash the tab. So we limit the number of existing clips. 
 14 
 15 	this.init = function(opts){
 16 		if (soundManager){ return soundManager };
 17 		soundManager = this;
 18 		this._timeout = opts.timeout || 400; //how long to wait before canceling clip
 19 		this._periodicity = opts.periodicity || 100; //how often to try waiting clip (ms)
 20 		this._maxObjects = opts.maxObjects || 10;  //max active clips
 21 		this._pendingClips = [];
 22 		this._clips = [];
 23 
 24 		this._debug = opts.debug || false;
 25 		if (this._debug == true){
 26 			setInterval(bind(this,function(){logger.log('clips',this._clips)}),10000);
 27 			setInterval(bind(this,function(){logger.log('pendingClips',this._pendingClips)}),10000);
 28 		}
 29 
 30 		this._gPaused = false;
 31 	};
 32 
 33 	this.toggleGlobalPause = function() { 
 34 		for (var i = 0, clip; clip = this._clips[i]; ++i) {
 35 			if (!this._gPaused) { 
 36 				clip.pause();
 37 			} else {
 38 				clip.play();
 39 			}
 40 		}
 41 		this._gPaused = !this._gPaused;
 42 	}
 43 
 44 	this.globalPause = function() {
 45 		for (var i = 0, clip; clip = this._clips[i]; ++i) {
 46 			clip.pause();
 47 		}
 48 		this._gPaused = true;
 49 	}
 50 	
 51 	this.globalPlay = function() { 
 52 		for (var i = 0, clip; clip = this._clips[i]; ++i) {
 53 			clip.play();
 54 		}
 55 		this._gPaused = false;
 56 	}
 57 
 58 	this.play = function(opts){
 59 		if (this._totalObjects() >= this._maxObjects){
 60 			this._pendingClips.push({timestamp: new Date(), sound: opts});
 61 			this._pendingClips.length === 1 ? this._processPendingClips() : function(){};
 62 		} else {
 63 			this._play(opts);
 64 		}
 65 	};
 66 
 67 	this.playBackgroundMusic = function(opts){
 68 		this.play(opts);
 69 	};
 70 
 71 	this.stop = function(url){
 72 		for (var i=0, clip; clip = this._clips[i]; ++i){
 73 			if (clip.urlId === url){
 74 				clip.pause();
 75 				this._clips.splice(this._clips.indexOf(clip),1);
 76 				return;
 77 			}
 78 		}
 79 		console.log('no clip with src',url);
 80 	};
 81 
 82 	this.stopBackgroundMusic = function(url){
 83 		this.stop(url);
 84 	};
 85 
 86 	this.setVolume = function (url, volume) {
 87 		for (var i=0, clip; clip = this._clips[i]; ++i){
 88 			if (clip.urlId === url){
 89 				clip.volume = volume;
 90 				return;
 91 			}
 92 		}
 93 	}
 94 	
 95 	this.preload = function(url){
 96 		// on Chrome, this will cause the file located at 
 97 		// url to be downloaded but not played
 98 		var sound = new Audio();
 99 		sound.autoplay = false;
100 		sound.preload = 'auto';
101 		sound.src = url;
102 	};
103 
104 	this.preloadBackgroundMusic = function(url){
105 		this.preload(url);
106 	};
107 
108 	this._totalObjects = function(){
109 		return this._clips.length;
110 	};
111 	
112 	var x = 0;
113 
114 	this._play = function(opts) {
115 		var clip = new Audio();
116 		clip.autoplay = false;
117 		clip.src = opts.src;
118 		clip.volume = opts.hasOwnProperty("volume") ? opts.volume : 1.0;
119 		clip.loop = opts.loop || false;
120 		clip.id = x++;
121 		clip.urlId = opts.src;
122 		this._clips.push(clip);
123 
124 		//remove reference to clip when it finishes playing
125 		clip.addEventListener('ended', bind(this, function() {
126 			if (!clip.loop) {
127 				var sound = this._clips.splice(this._clips.indexOf(clip), 1); 
128 				//logger.log('clip spliced', clip.id);
129 			}
130 		}), false);
131 
132 		if (opts.events) {
133 			for (var event in opts.events) {
134 				clip.addEventListener(event, opts.events[event], false);
135 			}
136 		}
137 		
138 		clip.play();
139 	};
140 
141 
142 	this._processPendingClips = function(){
143 		
144 		//play clips we have room for
145 		while(this._pendingClips.length && this._totalObjects() < this._maxObjects){
146 			var clip = this._pendingClips.shift();			
147 			this._play(clip.sound);			
148 		}
149 		
150 		//cancel expired clips
151 		for (var i=0, clip; clip = this._pendingClips[i]; ++i){
152 			if (new Date() - clip.timestamp > this._timeout){
153 				//logger.log('canceling',clip);
154 				this._pendingClips.splice(this._pendingClips.indexOf(clip),1);
155 			}
156 		}
157 		
158 		//if there are any clips left, setTimeout on this function again
159 		if (this._pendingClips.length){
160 			setTimeout(bind(this,'_processPendingClips'),this._periodicity);
161 		}
162 	};
163 });
164 
165 
166 exports.get = function(){return new SoundManager({})};
