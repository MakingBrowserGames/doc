1 "use import";
  2 
  3 import ...Font;
  4 import ...RGBA;
  5 
  6 var MAX_BUFFERS = 4;
  7 
  8 var _fontMap = {};
  9 var _numBuffers = 0;
 10 var _buffers = {};
 11 function getBuffer(def, name, color, size) {
 12 	if (!def.isReady) { return; }
 13 	
 14 	// TODO: add size here? we should support size someday... though if things get scaled, it might be best just to draw the originals
 15 	var key = color + '|' + name;
 16 	if (!(key in _buffers)) {
 17 		var canvas = getCanvas();
 18 		_buffers[key] = {canvas: canvas};
 19 		
 20 		var w = def._srcImage.width;
 21 		var h = def._srcImage.height;
 22 		canvas.width = w;
 23 		canvas.height = h;
 24 		
 25 		var ctx = canvas.getContext('2d');
 26 		ctx.save();
 27 		ctx.fillStyle = color + '';
 28 		ctx.fillRect(0, 0, w, h);
 29 		ctx.globalCompositeOperation = 'destination-out';
 30 		ctx.drawImage(def._srcImage, 0, 0);
 31 		ctx.restore();
 32 	}
 33 	
 34 	_buffers[key].lastUsed = +new Date();
 35 	return _buffers[key].canvas;
 36 }
 37 
 38 function getCanvas() {
 39 	if (_numBuffers == MAX_BUFFERS) {
 40 		var oldest = Number.MAX_VALUE;
 41 		for (var key in _buffers) {
 42 			if (_buffers[key].lastUsed < oldest) {
 43 				var canvas = _buffers[key].canvas;
 44 				delete _buffers[key];
 45 				return canvas;
 46 			}
 47 		}
 48 	} else {
 49 		return document.createElement('canvas');
 50 	}
 51 }

 53 exports.registerFont = function(name, def, cb) {
 54 	_fontMap[name] = def;
 55 	var img = def._srcImage = new Image();
 56 	img.onload = function() {
 57 		img.onload = null;
 58 		def.isReady = true;
 59 		cb && cb();
 60 	}
 61 	
 62 	img.src = def.img;
 63 }

 65 function measure(defs, text, scale) {
 66 	var width = 0;
 67 	for (var i = 0, n = text.length; i < n; ++i) {
 68 		var c = text[i];
 69 		if (c == ' ') {
 70 			width += defs.spaceWidth * scale;
 71 			continue;
 72 		} else if (c == '\t') {
 73 			width += 4 * defs.spaceWidth * scale;
 74 			continue;
 75 		}
 76 		
 77 		var def = defs[c];
 78 		if (def) {
 79 			width += (def.a + def.c) * scale;
 80 		}
 81 	}
 82 	
 83 	return {
 84 		width: width,
 85 		height: defs.height * scale
 86 	};
 87 }

 89 exports.wrapMeasureText = function(origMeasureText) {
 90 	return function(text) {
 91 		var font = Font.parse(this.font);
 92 		var fontName = font.getName();
 93 		if (!(fontName in _fontMap)) {
 94 			return origMeasureText.apply(this, arguments);
 95 		}
 96 		
 97 		var size = font.getSize();
 98 		var defs = _fontMap[fontName];
 99 		
100 		var scale = size / defs.height;
101 		return measure(defs, text, scale);
102 	}
103 }

105 exports.wrapFillText = function(origFillText) {
106 	
107 	return function(text, x, y) {
108 		var font = Font.parse(this.font);
109 		var fontName = font.getName();
110 		if (!(fontName in _fontMap)) {
111 			return origFillText.apply(this, arguments);
112 		}
113 		
114 		var color = RGBA.parse(this.fillStyle);
115 		var size = font.getSize();
116 		var defs = _fontMap[fontName];
117 		
118 		var scale = size / defs.height;
119 		var img = getBuffer(defs, fontName, color, size);
120 		if (!img) { return; }
121 		
122 		if (this.textBaseline == 'alphabetic') {
123 			y -= defs.baseline * scale;
124 		} else if (this.textBaseline == 'middle') {
125 			y -= defs['M'].h / 2 * scale;
126 		} else if (this.textBaseline == 'bottom') {
127 			y -= defs['M'].h * scale;
128 		}
129 		
130 		var xPos;
131 		if (this.textAlign == 'center') {
132 			xPos = x - measure(defs, text, scale).width / 2;
133 		} else if (this.textAlign == 'right') {
134 			xPos = x - measure(defs, text, scale).width;
135 		} else {
136 			xPos = x;
137 		}
138 		
139 		for (var i = 0, n = text.length; i < n; ++i) {
140 			var c = text[i];
141 			if (c == ' ') {
142 				xPos += defs.spaceWidth * scale;
143 				continue;
144 			} else if (c == '\t') {
145 				xPos += 4 * defs.spaceWidth * scale;
146 				continue;
147 			}
148 			
149 			var def = defs[c];
150 			if (def) {
151 				xPos += def.a * scale;
152 				this.drawImage(img, def.x1, def.y1, def.w, def.h, xPos, y, def.w * scale, def.h * scale);
153 				xPos += def.c * scale;
154 			}
155 		}
156 	}
157 };
