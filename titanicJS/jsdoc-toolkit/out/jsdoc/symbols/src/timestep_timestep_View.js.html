 26 	// Default options:
 27 	//   circle: boolean, default false - if true, use circular bounds for the view rather than rectangular and center the view at the origin
 28 	//   infinite: boolean, default false - if true, no bounding shape at all (e.g. infinite scroll plane)
 29 	//   clip: boolean, default false - if true, always clip to the region
 30 	//   canHandleEvents: boolean, default true - if false, this view is ignored for event handling
 31 	//   parent: object, if provided, sets the initial superview
 32 	this.init = function(opts) {
 33 		this._opts = opts = opts || {};
 34 		this.uid = ++UID + (opts.id || '');
 35 		this.__input = new input.InputHandler(this, opts);
 36 				if (!this.tag) { this.tag = opts.tag; }
 37 		this._filters = opts.filters || {};
 38 		this._circle = opts.circle;
 39 		this._infinite = opts.infinite;
 40 		this._canHandleEvents = 'canHandleEvents' in opts ? opts.canHandleEvents : true;
 41 
 42 		this._createStyle();
 43 
 44 		if (opts.parent) {
 45 			if (!opts.width && !opts.widthPercent) { 
 46 				opts.width = opts.parent.style.width;
 47 			}
 48 			if (!opts.height && !opts.heightPercent) {
 49 				opts.height = opts.parent.style.height;
 50 			}
 51 			this.style.update(opts);
 52 			opts.parent.addSubview(this);
 53 		} else {
 54 			this.style.update(opts);
 55 		}
 56 	}

136 	this.wrapRender = function(ctx, opts) {
137 		var s = this.style;
138 		if (!s.visible) { return; }
139 		
140 		if (this._firstRender) { this._firstRender = false; this.buildView(); }
141 		if (this._needsSort) { this._needsSort = false; this._subviews.sort(); }
142 		if (this._needsReflow) { this.reflow(); }
143 		
144 		ctx.save();
145 		ctx.translate(s.x + s.anchorX, s.y + s.anchorY);
146 		
147 		if (s.r) { ctx.rotate(s.r); }
148 		
149 		// clip this render to be within its view;
150 		if (s.scale != 1) { ctx.scale(s.scale, s.scale); }
151 		if (s.opacity != 1) { ctx.globalAlpha *= s.opacity; }
152 
153 		ctx.translate(-s.anchorX, -s.anchorY);
154 
155 		if (this.style.clip) { ctx.clipRect(0, 0, s.width, s.height); }
156 		var filters = this.getFilters();
157 		ctx.setFilters(filters);
158 
159 		try {
160 			if (this._opts.backgroundColor) {
161 				ctx.fillStyle = this._opts.backgroundColor;
162 				ctx.fillRect(0, 0, s.width, s.height);
163 			}
164 
165 			this.render && this.render(ctx, opts);
166 			this._renderSubviews(ctx, opts);
167 		} catch(e) {
168 		 	logger.error(this, e.message, e.stack);
169 		} finally {
170 			ctx.clearFilters();
171 			ctx.restore();
172 		}
173 	}

179 		}
180 	}

184 	this.containsLocalPt = function(pt) {
185 		if (this._infinite) { return true; }  // infinite plane
186 		
187 		if (this._circle) { // bounding circle
188 			var radius = this.style.radius;
189 			return pt.x * pt.x + pt.y * pt.y < radius * radius;
190 		} else { // bounding box
191 			var s = this.style,
192 				w = s.width,
193 				h = s.height;
194 
195 			if (w > 0 && h > 0) {
196 				return pt.x <= w && pt.y <= h && pt.x >= 0 && pt.y >= 0;
197 			} else if (w > 0) {
198 				return pt.x <= w && pt.y >= h && pt.x >= 0 && pt.y <= 0;
199 			} else if (h > 0) {
200 				return pt.x >= w && pt.y <= h && pt.x <= 0 && pt.y >= 0;
201 			} else {
202 				return pt.x >= w && pt.y >= h && pt.x <= 0 && pt.y <= 0;
203 			}
204 		}
205 	}

207 	this.getBoundingShape = function() {
208 		var s = this.style;
209 		if (this._infinite) {
210 			return true;
211 		} else if (this._circle) {
212 			return new Circle(s.x, s.y, s.radius / s.scale);
213 		} else {
214 			return new Rect(s.x, s.y, s.width * s.scale, s.height * s.scale);
215 		}

234 	this.onEventPropagate = function(evt, pt, atTarget) {
235 		if (atTarget) {
236 			var id = evt.id;
237 			switch(evt.type) {
238 				case input.eventTypes.CLEAR:
239 					input._evtHistory[input.eventTypes.MOVE] = null;
240 					var lastEvt = input._activeInputOver[id];
241 					if (lastEvt) {
242 						delete input._activeInputOver[id];
243 						var trace = lastEvt.trace;
244 						for (var i = 0, view; view = trace[i]; ++i) {
245 							view.__input.onLeave(id);
246 						}
247 					}
248 					break;
249 				
250 				case input.eventTypes.MOVE:
251 					// translate input:move events into two higher-level events:
252 					//   input:over and input:out
253 					
254 					var target = evt.trace[0];
255 					
256 					// fire input:out events first, start with deepest node and work out
257 					var lastEvt = input._activeInputOver[id];
258 					if (lastEvt && target != lastEvt.trace[0]) {
259 						var trace = lastEvt.trace;
260 						for (var i = 0, view; view = trace[i]; ++i) {
261 							if (!(view.uid in evt.pt)) {
262 								view.__input.onLeave(id, target == view);
263 							}
264 						}
265 					}
266 					
267 					if (!lastEvt || target != lastEvt.trace[0]) {
268 						// fire input:over events second, start with outermost node and go to target
269 						var trace = evt.trace;
270 						for (var i = evt.depth - 1; i >= 0; --i) {
271 							trace[i].__input.onEnter(id, target == view);
272 						}
273 						
274 						// update current mouse trace
275 						input._activeInputOver[id] = evt;
276 					}
277 					break;
278 			}
279 		}
280 	}

297 	this.getPosition = function(/* optional */ relativeTo) {
298 		var abs = new Point(),
299 			view = this,
300 			r = 0,
301 			s = this.style,
302 			w = s.width,
303 			h = s.height,
304 			c = 1;
305 		
306 		while (view && view != relativeTo) {
307 			var scale = view.style.scale;
308 			abs.scale(scale).rotate(view.style.r).add(view.style);
309 			r += view.style.r;
310 			w *= scale;
311 			h *= scale;
312 			c *= scale;
313 			view = view._superview;
314 		}
315 		
316 		return {
317 			x: abs.x,
318 			y: abs.y,
319 			r: r % (2 * Math.PI),
320 			width: w,
321 			height: h,
322 			scale: c
323 		};
324 	}
