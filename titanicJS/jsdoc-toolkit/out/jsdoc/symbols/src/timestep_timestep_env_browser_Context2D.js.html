 20 exports = function(opts) {
 21 	var parentNode = opts && opts.parent,
 22 		el = document.createElement('canvas');
 23 	
 24 	//if (!parentNode) {
 25 	//	el.style.cssText = 'position: absolute; top: 0px; left: 0px; z-index: 1';
 26 	//}
 27 	
 28 	el.width = opts.width;
 29 	el.height = opts.height;
 30 	
 31 	var ctx = el.getContext('2d');
 32 	ctx.font = '11px ' + timestep.device.defaultFontFamily;
 33 	ctx.getElement = function() { return el; }
 34 	
 35 	ctx.reset = function() {}
 36 	
 37 	ctx.setParentNode = function(node) { parentNode = node; }
 38 
 39 	ctx.show = function() {
 40 		if (parentNode) {
 41 			parentNode.appendChild(el);
 42 		}
 43 	};
 44 	
 45 	ctx.hide = function() {
 46 		if (el.parentNode) {
 47 			el.parentNode.removeChild(el);
 48 		}
 49 	};
 50 	
 51 	ctx.clear = function() {
 52 		el.width = el.width;
 53 		//this.clearRect(0, 0, el.width, el.height);
 54 	};
 55 	
 56 	ctx.clipRect = function(x, y, w, h) {
 57 		ctx.beginPath();
 58 		ctx.rect(x, y, w, h);
 59 		ctx.clip();
 60 	}
 61 	
 62 	ctx.swap = function() {};
 63 	ctx.execSwap = function() {};
 64 	
 65 	ctx.circle = function(x, y, radius) {
 66 		this.beginPath();
 67 		this.arc(x, y, radius, 0, 2 * Math.PI, true);
 68 	}
 69 
 70 	var _lastPointSprite = {
 71 		img: null,
 72 		color: null,
 73 		canvas: null
 74 	};
 75 
 76 	ctx.drawPointSprites = function(x1, y1, x2, y2) {
 77 		var sprite = this.pointSprite;
 78 		if (!sprite.complete) { return; }
 79 
 80 		var width = sprite.width;
 81 		var height = sprite.height;
 82 		var canvas = _lastPointSprite.canvas || (_lastPointSprite.canvas = document.createElement('canvas'));
 83 
 84 		if (sprite != _lastPointSprite.img || this.strokeStyle != _lastPointSprite.color) {
 85 			_lastPointSprite.img = sprite;
 86 			_lastPointSprite.color = this.strokeStyle;
 87 			canvas.width = width;
 88 			canvas.height = height;
 89 			var ctx = canvas.getContext('2d');
 90 			ctx.fillStyle = this.strokeStyle;
 91 			ctx.fillRect(0, 0, width, height);
 92 			ctx.globalCompositeOperation = 'destination-in';
 93 			ctx.drawImage(sprite, 0, 0);
 94 		}
 95 
 96 		// Add points to the buffer so there are drawing points every X pixels
 97 		var dx = x2 - x1;
 98 		var dy = y2 - y1;
 99 		var count = Math.ceil(Math.sqrt(dx * dx + dy * dy) / this.pointSpriteStep);
100 		if (count < 1) { count = 1; }
101 
102 		var d = this.lineWidth / 2;
103 		for (var i = 0; i < count; ++i) {
104 			var x = x1 + dx * i / count;
105 			var y = y1 + dy * i / count;
106 			this.drawImage(canvas, 0, 0, width, height, x - d / 2, y - d / 2, d, d);
107 		}
108 	}
109 	
110 	ctx.roundRect = function(x, y, width, height, radius) {
111 		this.beginPath();
112 		this.moveTo(x,y+radius);
113 		this.lineTo(x,y+height-radius);
114 		this.quadraticCurveTo(x,y+height,x+radius,y+height);
115 		this.lineTo(x+width-radius,y+height);
116 		this.quadraticCurveTo(x+width,y+height,x+width,y+height-radius);
117 		this.lineTo(x+width,y+radius);
118 		this.quadraticCurveTo(x+width,y,x+width-radius,y);
119 		this.lineTo(x+radius,y);
120 		this.quadraticCurveTo(x,y,x,y+radius);
121 	}
122 
123 	ctx.loadIdentity = function() {
124 		this.setTransform(1, 0, 0, 1, 0, 0);
125 	}
126 	
127 	ctx.measureText = FontRendererDeprecated.wrapMeasureText(
128 			FontRenderer.wrapMeasureText(ctx.measureText)
129 		);
130 	
131 	ctx.fillText = FontRendererDeprecated.wrapFillText(
132 			FontRenderer.wrapFillText(ctx.fillText)
133 		);
134 	
135 	ctx.strokeText = FontRenderer.wrapStrokeText(ctx.strokeText);
136 	ctx.filters = {}
137 	ctx.setFilters = function(filters) {
138 		for (var name in filters) {
139 			this.filters[name]  = filters[name];
140 		}
141 	}
142 
143 	ctx.clearFilters = function() {
144 		for (var name in this.filters) {
145 			delete this.filters[name];
146 		}
147 	}
148 	
149 	return ctx;
150 };
