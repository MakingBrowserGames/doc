1 "use import";
  2 
  3 import timestep.device as device;
  4 
  5 var randomColorElement = function() {
  6 	var e = Math.floor(Math.random() * 255).toString(16);
  7 	return ((e.length === 1) ? '0' : '') + e;
  8 };
  9 
 10 var randomColor = function() {
 11 	return '#' + randomColorElement() + randomColorElement() + randomColorElement();
 12 };
 13 
 14 exports = FontBuffer = Class(function() {
 15 	this.init = function(opts) {
 16 		// 8 * 24
 17 		// 10 * 32
 18 		// 8 * 64
 19 		var lineSizes = [{size: 24, count: 8}, {size: 32, count: 10}, {size: 64, count: 8}],
 20 			lineSize,
 21 			lines,
 22 			item,
 23 			y = 0,
 24 			i, j;
 25 
 26 		this._canvas = document.createElement('canvas');
 27 		this._canvas.width = 1024;
 28 		this._canvas.height = 1024;
 29 		this._ctx = this._canvas.getContext('2d');
 30 
 31 		this._list = [];
 32 		for (i = 0; i < lineSizes.length; i++) {
 33 			lines = [];
 34 			lineSize = lineSizes[i];
 35 
 36 			for (j = 0; j < lineSize.count; j++) {
 37 				item = {
 38 					previous: null,
 39 					next: null,
 40 					x: 0,
 41 					y: y,
 42 					width: 1024,
 43 					height: 0,
 44 					hash: null,
 45 					frame: 0,
 46 					refresh: true,
 47 					ctx: this._ctx
 48 				};
 49 				lines.push(item);
 50 				y += lineSize.size;
 51 			}
 52 
 53 			this._list.push({
 54 				size: lineSize.size,
 55 				lines: lines
 56 			});
 57 		}
 58 
 59 		this._hashMap = {};
 60 
 61 		this._currentFrame = 0;
 62 		this._frameTimeout = 3;
 63 
 64 		jsio('import timestep.Application as Application').get().subscribe('Tick', this, this._onTick);
 65 	};
 66 
 67 	this._onTick = function(dt) {
 68 		this._currentFrame++;
 69 
 70 		var remove,
 71 			currentFrame = this._currentFrame,
 72 			frameTimeout = this._frameTimeout,
 73 			list = this._list,
 74 			lines,
 75 			item,
 76 			i, j, k, l;
 77 
 78 		for (i = 0, j = list.length; i < j; i++) {
 79 			lines = list[i].lines;
 80 			for (k = 0, l = lines.length; k < l; k++) {
 81 				item = lines[k];
 82 				while (item) {
 83 					if (item.hash === null) {
 84 						if (item.next && (item.next.hash === null)) {
 85 							item.width += item.next.width;
 86 							item.next = item.next.next;
 87 						}
 88 					} else if (currentFrame > item.frame + frameTimeout) {
 89 						this._ctx.fillStyle = randomColor();
 90 						this._ctx.fillRect(item.x, item.y, item.width, item.height);
 91 						// Remove old item...
 92 						delete(this._hashMap[item.hash]);
 93 						item.hash = null;
 94 					}
 95 					item = item.next;
 96 				}
 97 			}
 98 		}
 99 	};
100 
101 	this.alloc = function(opts) {
102 		var requestHeight = opts.height,
103 			requestWidth = opts.width + 3, // Add some extra pixels to allow color bleeding...
104 
105 			strokeStyle = opts.strokeStyle || '',
106 			fillStyle = opts.fillStyle || '',
107 			font = opts.font || '',
108 			hash = strokeStyle + '_' + fillStyle + '_' + font + '_' + opts.text,
109 
110 			list = this._list,
111 			lines,
112 			item = this._hashMap[hash],
113 			i, j, k, l;
114 
115 		if (item) {
116 			item.frame = this._currentFrame;
117 			item.refresh = false;
118 			return item;
119 		}
120 
121 		for (i = 0, j = list.length; i < j; i++) {
122 			if (requestHeight <= list[i].size) {
123 				lines = list[i].lines;
124 				for (k = 0, l = lines.length; k < l; k++) {
125 					item = lines[k];
126 					while (item) {
127 						if ((requestWidth <= item.width) && (item.hash === null)) {
128 							if (requestWidth !== item.width) {
129 								//console.log('request:', requestWidth, 'width:', item.width, 'rest:', item.width - requestWidth);
130 								item.next = {
131 									previous: item,
132 									next: item.next,
133 									x: item.x + requestWidth,
134 									y: item.y,
135 									width: item.width - requestWidth,
136 									hash: null,
137 									frame: 0,
138 									ctx: this._ctx
139 								};
140 							}
141 
142 							item.frame = this._currentFrame;
143 							item.hash = hash;
144 							item.width = requestWidth;
145 							item.height = requestHeight;
146 							item.refresh = true;
147 
148 							this._ctx.clearRect(item.x, item.y, requestWidth, requestHeight);
149 
150 							this._hashMap[hash] = item;
151 							return item;
152 						}
153 						item = item.next;
154 					}
155 				}
156 
157 				break;
158 			}
159 		}
160 
161 		return false;
162 	};
163 
164 	this.getCanvas = function() {
165 		return this._canvas;
166 	};
167 });
