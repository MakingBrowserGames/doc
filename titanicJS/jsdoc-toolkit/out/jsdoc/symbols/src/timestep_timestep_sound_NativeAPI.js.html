1 "use import";
  2 
  3 import lib.PubSub;
  4 
  5 /**
  6  * @extends lib.PubSub
  7  */
  8 exports = Class(lib.PubSub, function(supr) {
  9 	var pauseAllSources = function(){
 10 		// this is used as an "instance method" for the sound objects held
 11 		// in the _sound array of BrowserAPI instances.
 12 
 13 		//TODO set sound.isLoaded to false here.
 14 
 15 		logger.log("PAUSE CALLED", this.sources);
 16 
 17 		this.sources.forEach(function(source) {
 18 			logger.log("PAUSE CALLED", source);
 19 			NATIVE.sound.pauseSound(source);
 20 		});
 21 	};
 22 
 23 	this.init = function(opts) {
 24 		supr(this, 'init', [opts]);
 25 		this._opts = opts;
 26 		this._map = merge({}, opts.map);
 27 		this._path = opts.path;
 28 		this._sounds = {};
 29 		this.muted = false;
 30 		this.noload = !!opts.noload;
 31 
 32 		// add sounds to the audio API's list of sounds and
 33 		// preload them if appropriate
 34 		for (var key in this._map) {
 35 			var item = this._map[key];
 36 			this.addSound(key, item);
 37 		}
 38 	}
 39 
 40 
 41 
 42 	this.addSound = function(name, opts) {
 43 		var sound = this._sounds[name] = {};
 44 		var path = this._path.replace(/\/$/, '');
 45 		sound.sources = [];
 46 		sound.pause = pauseAllSources;
 47 		sound.isLoaded = !opts.background && !this.noload;
 48 
 49 		// if a list of file names is given in sources, load them as alternative
 50 		// clips for this sound. Else, assume the only clip for this sound
 51 		// is the file with its same name
 52 		if (!opts.sources) { opts.sources = [name]; }
 53 		opts.sources.forEach(bind(this, function(source) {
 54 			var fullPath = path + '/' + source + '.mp3';
 55 			sound.sources.push(fullPath);
 56 			// var t = +new Date();
 57 			if(sound.isLoaded) { NATIVE.sound.loadSound(fullPath); }
 58 			// logger.log("loadsound", +new Date() - t);
 59 		}));
 60 		sound.volume = opts.volume;
 61 		sound.loop = opts.loop || opts.background;
 62 		sound.background = opts.background;
 63 
 64 	}
 65 
 66 	this.setMuted = function(muted) {
 67 		this._muted = muted;
 68 		if(muted) {
 69 			this._pauseAll();
 70 		}
 71 	}
 72 
 73 	this.setVolume = function(name, volume) {
 74 		sound = this._sounds[name];
 75 		sound.volume = volume;
 76 		if (sound.isLoaded) {
 77 			sound.sources.forEach(function(item){
 78 				// if more than one instance of a sound is playing, then 
 79 				// this will only change the volume of the most recently
 80 				// started instance 
 81 				NATIVE.sound.setVolume(item, volume);
 82 			});
 83 		}
 84 	}
 85 
 86 	this._pauseAll = function(){
 87 		for (var key in this._sounds) {
 88 			this._sounds[key].pause();
 89 		};
 90 	}
 91 
 92 	this.play = function(name) {
 93 		if (this.muted) { return; }
 94 
 95 		var sound = this._sounds[name];
 96 		if (!sound) {
 97 			return;
 98 		}
 99 		if (sound.background) { 
100 			this.playBackgroundMusic(name);
101 			return;
102 		}
103 		if (typeof sound.volume == 'number') {
104 			var volume = sound.volume;
105 		} else {
106 			var volume = 1.0;
107 		}
108 		var loop = sound.loop || false;
109 		var index = Math.random() * sound.sources.length | 0;
110 		var source = sound.sources[index];
111 		logger.log('playing', source, index, JSON.stringify(sound), JSON.stringify(sound.sources));
112 		NATIVE.sound.playSound(source, volume, loop);
113 		sound.isLoaded = true;
114 	}
115 
116 	this.pause = function(name) {
117 		this._sounds[name].pause();
118 	}
119 
120 	this.playBackgroundMusic = function(name) {
121 		// this.pauseBackgroundMusic();
122 
123 		if (this.muted) { return; }
124 		var sound = this._sounds[name];
125 		if (!sound) { return; }
126 
127 		if (typeof sound.volume == 'number') {
128 			var volume = sound.volume;
129 		} else {
130 			var volume = 1.0;
131 		}
132 		var index = Math.random() * sound.sources.length | 0;
133 		var source = sound.sources[index];
134 		logger.log('playing', source, JSON.stringify(sound));
135 		sound.currentSource = source;
136 
137 		this._currentMusic = sound;
138 		NATIVE.sound.playBackgroundMusic(source, volume, true);
139 		sound.isLoaded = true;
140 	}
141 
142 	this.pauseBackgroundMusic = function() {
143 		var currentMusic = this._currentMusic;
144 		if (!currentMusic) { return; }
145 		if (currentMusic.currentSource) {
146 			NATIVE.sound.pauseSound(currentMusic.currentSource);
147 			currentMusic.currentSource = null;
148 		}
149 
150 		this._currentMusic = null;
151 	}
152 });
