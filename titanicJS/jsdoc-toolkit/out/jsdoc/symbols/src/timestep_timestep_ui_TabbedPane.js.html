  6 /**
  7  * @extends timestep.View
  8  */
  9 exports = Class(timestep.View, function(supr) {
 10 	this.init = function(opts) {
 11 		opts = Class.defaults(opts, {
 12 			tabSpacing: 0,
 13 			tabCtor: exports.DefaultTabCtor,
 14 			tabHeight: 24
 15 		});
 16 		
 17 		supr(this, 'init', [opts]);
 18 		
 19 		this._tabSpacing = opts.tabSpacing;
 20 		this._tabCtor = opts.tabCtor;
 21 		this._tabHeight = opts.tabHeight;
 22 		
 23 		this._tabs = [];
 24 		this._panes = {};
 25 		this._needsLayout = false;
 26 	}
 27 	
 28 	this.buildView = function() {
 29 		
 30 	}
 31 	
 32 	this.newTab = function(def) {
 33 		def = Class.defaults(def, {
 34 			label: '',
 35 			index: this._tabs.length,
 36 			parent: this
 37 		});
 38 		
 39 		var tab = new this._tabCtor(def);
 40 		tab.style.height = this._tabHeight;
 41 		this.addSubview(tab);
 42 		this._tabs.splice(def.index, 0, tab);
 43 		
 44 		var pane = tab.getPane();
 45 		if (pane) { this.positionPane(pane); }
 46 
 47 		tab.subscribe('Select', this, 'onTabSelect', tab);
 48 		this._needsLayout = true;
 49 		
 50 		return tab;
 51 	}
 52 	
 53 	this.select = function() {
 54 		
 55 	}
 56 
 57 	this.positionPane = function(pane) {
 58 		pane.style.y = this._tabHeight;
 59 		pane.style.width = this.style.width;
 60 		pane.style.height = this.style.height - this._tabHeight;
 61 	}
 62 	
 63 	this.getTab = function(i) { return this._tabs[i]; }
 64 	
 65 	this.onTabSelect = function(tab) {
 66 		this.selectTab(tab);
 67 	}
 68 	
 69 	this.selectTab = function(tab) {
 70 		var prevTab = this._currentTab,
 71 			pane = tab.getPane();
 72 		
 73 		this.positionPane(pane);
 74 		
 75 		if (prevTab) {
 76 			prevTab.setToggled(false);
 77 			prevTab.getPane().removeFromSuperview();
 78 		}
 79 		
 80 		tab.setToggled(true);
 81 		this._currentTab = tab;
 82 		this.addSubview(pane);
 83 		this.publish('ShowTab', tab, prevTab);
 84 	}
 85 	
 86 	this.layoutSubviews = function(ctx) {
 87 		if (!this._needsLayout) { return; }
 88 		
 89 		this._needsLayout = false;
 90 		var x = 0;
 91 		for (var i = 0, tab; tab = this._tabs[i]; ++i) {
 92 			var width = tab.measure(ctx);
 93 			tab.style.x = x;
 94 			tab.style.width = width;
 95 			x += width + this._tabSpacing;
 96 		}
 97 	}
 98 	
 99 	this._renderSubviews = function(ctx) {
100 		if (this._needsLayout) { this.layoutSubviews(ctx); }
101 		
102 		supr(this, '_renderSubviews', arguments);
103 	}
104 });

106 exports.DefaultTabCtor = Class(Button, function(supr) {
107 	this.init = function(opts) {
108 		opts = Class.defaults(opts, {
109 			font: 'bold 12px ' + timestep.device.defaultFontFamily,
110 			padding: '15',
111 			pane: null,
112 		});
113 		
114 		supr(this, 'init', [opts]);
115 		
116 		this._pane = opts.pane;
117 		this._padding = new Padding(opts.padding);
118 		this._measuredWidth = {
119 			value: null,
120 			width: 0
121 		};
122 	}
123 	
124 	this.getPane = function() {
125 		return this._pane;
126 	}
127 	
128 	this.onInputStart = function() {
129 		this.publish('Select');
130 	}
131 	
132 	this.getWidth = function() {
133 		return this._padding.left + this._padding.right + (this._measuredWidth.width || 0);
134 	}
135 	
136 	this.measure = function(ctx) {
137 		if (this._measuredWidth.value != this.label) {
138 			this._measuredWidth.value = this.label;
139 			if (this.style.font) { ctx.font = this.style.font; }
140 			this._measuredWidth.width = ctx.measureText(this.label).width;
141 		}
142 		
143 		return this.getWidth();
144 	}
145 	
146 	this.render = function(ctx) {
147 		if (this.label) {
148 			if (this.style.font) { ctx.font = this.style.font; }
149 			ctx.fillStyle = this.style.color;
150 			ctx.fillText(this.label, this._padding.left, 20);
151 		}
152 	}
153 });
