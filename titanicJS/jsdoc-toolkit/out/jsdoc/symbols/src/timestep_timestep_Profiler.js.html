1 "use import";
  2 
  3 import .Timer;
  4 
  5 exports = Class(function() {
  6 	this.init = function(opts) {
  7 		opts = opts || {};
  8 
  9 		this._start = {};
 10 		this._durations = {};
 11 		this._frames = {};
 12 		this._isEnabled = opts.enabled;
 13 		this._reportInterval = opts.reportInterval || 1000;
 14 		this._lastReport = null;
 15 		this._popupsBlocked = opts.disablePopup;
 16 	}
 17 	
 18 	this.setEnabled = function(isEnabled) {
 19 		this._isEnabled = isEnabled;
 20 	}
 21 	
 22 	this.setReportInterval = function(interval) {
 23 		this._reportInterval = interval;
 24 	}
 25 
 26 	this.start = function(key) {
 27 		if (this._isEnabled) {
 28 			this._start[key] = [+new Date(), Timer.frames];
 29 		}
 30 	}
 31 	
 32 	this.stop = function(key) {
 33 		if (this._isEnabled) {
 34 			var list = this._durations[key] || (this._durations[key] = []);
 35 			list.push(+new Date() - this._start[key][0]);
 36 			list = this._frames[key] || (this._frames[key] = []);
 37 			list.push(Timer.frames - this._start[key][1]);
 38 		}
 39 	}
 40 	
 41 	this.report = function() {
 42 		if (!this._isEnabled) return;
 43 		if (this._lastReport) {
 44 			var elapsed = +new Date() - this._lastReport;
 45 			if (elapsed < this._reportInterval) {
 46 				return;
 47 			}
 48 		}
 49 		this._lastReport = +new Date();
 50 
 51 		var usePopup = typeof window != 'undefined' && typeof window.open == 'function' && !this._popupsBlocked;
 52 		var doc;
 53 
 54 		if (usePopup) {
 55 			var win = this._win || (this._win = window.open('about:blank'));
 56 			if (win === undefined) {
 57 				usePopup = false;
 58 				this._popupsBlocked = true;
 59 				logger.log("Popups blocked - sending profiling info to console");
 60 			} else {
 61 				doc = win.document;
 62 			}
 63 		}
 64 				
 65 		var hasHTML = usePopup,
 66 			header = hasHTML ? '<table>' : '',
 67 			footer = hasHTML ? '</table>' : '',
 68 			colJoin = hasHTML ? ['<td>', '</td>'] : ['', '\t'],
 69 			rowJoin = hasHTML ? ['<tr>', '</tr>'] : ['', '\n'],
 70 			result = [header];
 71 		
 72 		result.push(rowJoin[0] + colJoin[0]
 73 			+ ['key', 'count', 'sum', 'avg', 'min', 'max']
 74 					.join(colJoin[1] + colJoin[0])
 75 			+ colJoin[1] + rowJoin[1]);
 76 		
 77 		for (var key in this._durations) {
 78 			var list = this._durations[key];
 79 			result.push(rowJoin[0] + colJoin[0]
 80 				+ [key + '-time', list.length, sum(list), avg(list), min(list), max(list)]
 81 						.join(colJoin[1] + colJoin[0])
 82 				+ colJoin[1] + rowJoin[1]);
 83 		}
 84 		
 85 		for (var key in this._frames) {
 86 			var list = this._frames[key];
 87 			result.push(rowJoin[0] + colJoin[0]
 88 				+ [key + '-frames', list.length, sum(list), avg(list), min(list), max(list)]
 89 						.join(colJoin[1] + colJoin[0])
 90 				+ colJoin[1] + rowJoin[1]);
 91 		}
 92 
 93 		result.push(footer);
 94 		
 95 		result = result.join('');
 96 
 97 		if (doc) {
 98 			doc.open();
 99 			doc.write(result);
100 			doc.close();
101 		} else {
102 			logger.log(result);
103 		}
104 	}
105 	
106 	function sum(arr) {
107 		var len = arr.length,
108 			sum = 0;
109 		for (var i = 0; i < len; ++i) {
110 			sum += arr[i];
111 		}
112 		return sum;
113 	}
114 	
115 	function avg(arr) {
116 		var len = arr.length,
117 			sum = 0;
118 		for (var i = 0; i < len; ++i) {
119 			sum += arr[i];
120 		}
121 		return sum / len;
122 	}
123 
124 	function min(arr) {
125 		var len = arr.length,
126 			min = Number.MAX_VALUE;
127 		for (var i = 0; i < len; ++i) {
128 			if (arr[i] < min) { min = arr[i]; }
129 		}
130 		return min;
131 	}
132 
133 	function max(arr) {
134 		var len = arr.length,
135 			max = Number.MIN_VALUE;
136 		for (var i = 0; i < len; ++i) {
137 			if (arr[i] > max) { max = arr[i]; }
138 		}
139 		return max;
140 	}
141 });
