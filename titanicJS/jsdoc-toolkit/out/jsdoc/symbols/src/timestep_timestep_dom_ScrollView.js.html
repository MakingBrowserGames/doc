1 "use import";
  2 
  3 import .View;
  4 import math2D.Point;
  5 import timestep.animate as animate;
  6 
  7 /**
  8  * @extends timestep.dom.View
  9  */
 10 exports = Class(View, function(supr) {
 11 	
 12 	this.init = function(opts) {
 13 		opts = Class.defaults(opts, {
 14 				fullWidth: 0,
 15 				fullHeight: 0,
 16 				offsetX: 0,
 17 				offsetY: 0,
 18 				clip: true
 19 			});
 20 		
 21 		supr(this, 'init', [opts]);
 22 		
 23 		this._acceleration = 5;
 24 		
 25 		this._anim = {
 26 			offset: 0,
 27 			lastDelta: 0,
 28 			offset: 0
 29 		};
 30 	}
 31 	
 32 	var BOUNCE_MAX_DIST = 50;
 33 	var PI_2 = Math.PI / 2;
 34 	
 35 	this.getStyleBounds = function() {
 36 		var bounds = bounds = this._scrollBounds;
 37 		var minY = -bounds.maxY;
 38 		var maxY = -bounds.minY < minY ? minY : -bounds.minY;
 39 		var minX = -bounds.maxX;
 40 		var maxX = -bounds.minX < minX ? minX : -bounds.minX;
 41 		return {
 42 			minY: minY,
 43 			maxY: maxY,
 44 			minX: minX,
 45 			maxX: maxX
 46 		};
 47 	}
 48 	
 49 	this.getOffset = function() {
 50 		return {
 51 			x: this._computed.scrollLeft,
 52 			y: this._computed.scrollTop
 53 		};
 54 	}
 55 
 56 	this.setOffset = function(x, y) {
 57 		var oldOffset = this.getOffset();
 58 		var delta = {
 59 			x: x - oldOffset.x,
 60 			y: y - oldOffset.y
 61 		};
 62 		this.style.scrollLeft = x
 63 		this.style.scrollTop = y;
 64 		
 65 		if (delta.x || delta.y) {
 66 			this.needsReflow();
 67 			this.publish('Scrolled', delta);
 68 		}
 69 	}
 70 
 71 	this.addOffset = function(x, y) {
 72 		this.setOffset(
 73 				x != undefined && x != null && (this._computed.scrollLeft + x),
 74 				y != undefined && y != null && (this._computed.scrollTop + y)
 75 			);
 76 	}
 77 	
 78 	this.stopScrolling = function() {
 79 		animate(this._anim).now({}, 1);
 80 	}
 81 	
 82 	this.onInputStart = function(evt, pt) {
 83 		this.startDrag({radius: 3});
 84 	}
 85 	
 86 	this.onDragStart = function() {
 87 		this.startBounce();
 88 		this._anim.offset = this.getOffset();
 89 	}
 90 
 91 	this.onDrag = function(dragEvt, moveEvt, delta) {
 92 		this._anim.lastDelta = delta;
 93 		this._anim.offset.add(delta);
 94 		
 95 		//logger.log('delta:', JSON.stringify(delta), 'abs:', JSON.stringify(this._anim.offset));
 96 		this.setOffset(this._anim.offset);
 97 		moveEvt.cancel();
 98 	}
 99 
100 	this.onDragStop = function(dragEvt, selectEvt) {
101 		var delta = new math2D.Point(this._anim.lastDelta).scale(this._acceleration);
102 		var offset = this._anim.offset;
103 		var distance = delta.getMagnitude();
104 		
105 		// if we overshot the bounds, don't waste time animating the acceleration.
106 		var bounds = this.getStyleBounds();
107 		var target = new math2D.Point(offset).add(delta);
108 		if (target.y < bounds.minY - BOUNCE_MAX_DIST) {
109 			if (offset.y < bounds.minY) {
110 				distance = 0;
111 			} else {
112 				delta.y -= (bounds.minY - BOUNCE_MAX_DIST - target.y);
113 				distance = delta.getMagnitude();
114 			}
115 		}
116 		
117 		if (target.y > bounds.maxY + BOUNCE_MAX_DIST) {
118 			if (offset.y > bounds.maxY) {
119 				distance = 0;
120 			} else {
121 				delta.y -= (target.y - (bounds.maxY - BOUNCE_MAX_DIST));
122 				distance = delta.getMagnitude();
123 			}
124 		}
125 		
126 		if (distance) {
127 			animate(this._anim).now(bind(this, function(tt, t) {
128 				this.setOffset(new math2D.Point(offset).add(new math2D.Point(delta).scale(tt)));
129 			}), 100 * Math.log((distance+1) * 100), animate.easeOut).then(bind(this, 'endBounce'));
130 		} else {
131 			this.endBounce();
132 		}
133 	}
134 	
135 	this.startBounce = function() {
136 		this._canBounce = true;
137 	}
138 	
139 	this.endBounce = function() {
140 		var offset = this.getOffset();
141 		var bounds = this.getStyleBounds();
142 		
143 		var delta = {
144 			x: 0,
145 			y: 0
146 		};
147 		
148 		var target;
149 		if (offset.y < bounds.minY) {
150 			target = bounds.minY;
151 		} else if (offset.y > bounds.maxY) {
152 			target = bounds.maxY;
153 		} else {
154 			return;
155 		}
156 		
157 		delta.y = target - offset.y;
158 		animate(this._anim).now(bind(this, function(tt, t) {
159 			this.setOffset(new math2D.Point(offset).add(new math2D.Point(delta).scale(tt)));
160 		}), 500, animate.easeInOut).then(bind(this, function() {
161 			this._canBounce = false;
162 		}));
163 	}
164 	
165 	this.setScrollBounds = function(bounds) { this._scrollBounds = bounds; }
166 	this.getScrollBounds = function() { return this._scrollBounds; }
167 
168 });
