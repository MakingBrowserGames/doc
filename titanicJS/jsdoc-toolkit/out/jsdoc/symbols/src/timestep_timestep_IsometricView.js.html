1 //Handles conversion between world points, isometric points, and screen points
  2 
  3 jsio('import .Image');
  4 jsio('import .ImageView');
  5 jsio('import .View');
  6 jsio('import math2D.Rect');
  7 jsio('import lib.sort');
  8 
  9 /**
 10  * @extends timestep.View
 11  */
 12 exports = Class(View, function(supr) {
 13 	this.init = function(opts) {
 14 		supr(this, 'init', arguments);
 15 		this._grid = opts.grid;
 16 		this.bgColor = opts.bgColor;
 17 	};
 18 
 19 	// map screen x/y coordinate to isometric tile coordinates
 20 	this.getIsoForCartesian = function(x, y) {
 21 
 22 		var i = y - x / 2,
 23 		    j = y + x / 2;
 24 		
 25 		return {
 26 			i: i,
 27 			j: j 
 28 		};
 29 	};
 30 	
 31 	// map isometric tile coordinates to screen XY coordinates
 32 	this.getCartesianForIso = function(i, j) {
 33 		return {
 34 			x: (j - i),
 35 			y: (j + i) / 2
 36 		};
 37 	};
 38 	
 39 	this._renderSubviews = function(ctx) {
 40 		lib.sort(this._subviews, function() { return this._k || 0; }, function() { return this._j; }, function() { return this._i; });
 41 		
 42 		var i = 0, view;
 43 		while (view = this._subviews[i++]) {
 44 			view.wrapRender(ctx);
 45 		}
 46 	}
 47 	
 48 	this.render = function(ctx){
 49 		//outline
 50 		if( !this._mapUrl){
 51 			var top = this.getCartesianForIso(0,0),
 52 			left = this.getCartesianForIso(0, this.style.height),
 53 			bottom = this.getCartesianForIso(this.style.width, this.style.height),
 54 			right = this.getCartesianForIso(this.style.width, 0);
 55 			
 56 			ctx.beginPath();
 57 			ctx.strokeStyle = 'rgb(0,0,0)';
 58 			ctx.fillStyle = this.bgColor || 'rgb(0,100,100)';
 59 			ctx.moveTo(top.x,top.y);
 60 			ctx.lineTo(left.x,left.y);
 61 			ctx.lineTo(bottom.x,bottom.y);
 62 			ctx.lineTo(right.x,right.y);
 63 			ctx.lineTo(top.x,top.y);
 64 			ctx.stroke();
 65 			ctx.fill();
 66 			
 67 			if (this._grid) {
 68 				for (var i = 0; i < this.style.height / this._grid; i++) {
 69 					for (var j = 0; j < this.style.width / this._grid; j++) {
 70 						var top = this.getCartesianForIso(i * this._grid, j * this._grid),
 71 						left = this.getCartesianForIso(i * this._grid, (j + 1) * this._grid),
 72 						bottom = this.getCartesianForIso((i + 1) * this._grid, (j + 1) * this._grid),
 73 						right = this.getCartesianForIso((i + 1) * this._grid, j * this._grid);
 74 
 75 						ctx.beginPath();
 76 						ctx.strokeStyle = 'rgb(0,0,0)';
 77 						ctx.fillStyle = 'rgb(0,100,100)';
 78 						ctx.moveTo(top.x,top.y);
 79 						ctx.lineTo(left.x,left.y);
 80 						ctx.lineTo(bottom.x,bottom.y);
 81 						ctx.lineTo(right.x,right.y);
 82 						ctx.lineTo(top.x,top.y);
 83 						ctx.stroke();
 84 					}
 85 				}
 86 			}
 87 		} 
 88 		supr(this, 'render', arguments);
 89 
 90 	};
 91 
 92 });
