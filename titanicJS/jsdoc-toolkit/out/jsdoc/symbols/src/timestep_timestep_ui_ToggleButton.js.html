1 "use import";
  2 
  3 import std.js as JS;
  4 import .ImageButton
  5 
  6 /**
  7  * @extends timestep.ui.ImageButton
  8  */
  9 exports = Class(ImageButton, function(supr) {
 10 
 11 	this.init = function(opts){
 12 		
 13 		this._isToggled = opts.toggled || true;
 14 
 15 		opts = JS.merge(opts, {
 16 			animations: {
 17 				'default': {frames: [[0, 0]], imageURL: opts.imageURL, width: opts.width, height: opts.height},
 18 				'toggleOn': {frames: [[0, 0]], imageURL: opts.imageURL, width: opts.width, height: opts.height},
 19 				'hoverOn': {frames: [[0, opts.height]], imageURL: opts.imageURL, width: opts.width, height: opts.height},
 20 				'pressed': {frames: [[0, opts.height * 2]], imageURL: opts.imageURL, width: opts.width, height: opts.height},
 21 				'toggleOff': {frames: [[0, opts.height * 3]], imageURL: opts.imageURL, width: opts.width, height: opts.height},
 22 				'hoverOff': {frames: [[0, opts.height * 4]], imageURL: opts.imageURL, width: opts.width, height: opts.height}
 23 			},
 24 		});
 25 
 26 		supr(this,'init',[opts]);
 27 
 28 		this._isToggled ? this.startAnimation('toggleOn') : this.startAnimation('toggleOff');
 29 	}
 30 	
 31 	this.onInputSelect = function() {
 32 		this._isPressed = false;
 33 		this._isToggled = !this._isToggled;
 34 		if (this._isEnabled) {
 35 			if (this._isToggled){
 36 				this.startAnimation('toggleOn'); this.publish('Select'); this.publish('ToggleOn');
 37 			}else{
 38 				this.startAnimation('toggleOff'); this.publish('Select'); this.publish('ToggleOff');
 39 			}
 40 		}
 41 	}
 42 
 43 	this.onInputStart = function() {
 44 		this._isPressed = true;
 45 		if (this._isEnabled) {
 46 			this.startAnimation('pressed');
 47 		}
 48 	}
 49 
 50 	this.onInputOut = function(){
 51 		this._isToggled ? this.startAnimation('toggleOn') : this.startAnimation('toggleOff');
 52 	}
 53 
 54 	this.onInputOver = function(){
 55 		if (this._isEnabled){
 56 			this._isToggled ? this.startAnimation('hoverOn') : this.startAnimation('hoverOff');
 57 		}
 58 	}
 59 
 60 });
