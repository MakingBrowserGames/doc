 31 	this.init = function(opts) {
 32 		supr(this, 'init', [opts]);
 33 		opts = JS.merge(opts, defaults);

 35 		this._scriptQueue = [];
 36 		this._followTargets = [];

 38 		this._centerY = opts.centerY;
 39 		this._targetX = opts.centerX;
 40 		this._targetY = opts.centerY;
 41 		this._fullWidth = opts.fullWidth || this.style.width;
 42 		this._fullHeight = opts.fullHeight || this.style.height;
 43 		
 44 		this._targetScale = opts.initialZoom;

 46 		this._minZoom = opts.minZoom;
 47 		this._maxZoom = opts.maxZoom;
 48 		// TODO: throw errors when user passes in non-sensical options		
 49 		this.setMaxPanSpeed(opts.maxPanSpeed);
 50 		this.setMaxZoomSpeed(opts.maxZoomSpeed);
 51 		this.follow(opts.follow);
 52 		this.pushScript(opts.script);
 53 		if (opts.lock) {
 54 			this.lockMode(this.getMode(), opts.lockEnforce);
 55 		} else {
 56 			this._lock = false;
 57 			this._lockEnforce = false;
 58 		}
 59 		this.setMode(opts.mode);
 60 		this._panEaseType = opts.panEaseType;
 61 		
 62 		if (opts.contentView) {
 63 			this.setContentView(opts.contentView);
 64 		}

 87 	this.render = function(ctx, dt) {
 88 		if (this._contentView) {
 89 			this._contentView.style.scale = this._scale;
 90 			this._contentView.style.x = -this._viewport.x * this._scale;
 91 			this._contentView.style.y = -this._viewport.y * this._scale;
 92 			this._contentView.wrapRender(ctx, dt);
 93 		}
 94 	}

 96 	this._doRender = function(ctx) {
 97 		// in Pixels FIXME pixel/point difference
 98 		var width  = (this.style.width / this._scale + 0.5)|0,
 99 		    height = (this.style.height / this._scale + 0.5)|0;
100 	    
101 		this._viewport = new Rect({
102 			width: width,
103 			height: height,
104 			x: (this._centerX  - width/2)|0,
105 			y: (this._centerY  - height/2)|0
106 		});
107 		
108 		if (!this.counter) { this.counter = 1 };
109 		if (!(++this.counter % 30)) {
110 //			logger.info(this._scale);
111 //			logger.info('viewport', JSON.stringify(this._viewport))
112 		}
113 		this.render(ctx, this._viewport, this._scale)
114 	}

118 	this._renderSubviews = function(ctx, dt) {
119 		if (this._contentView) {return;}
120 		for (var i = 0, view; view = this._subviews[i]; ++i) {
121 			ctx.save();
122 			ctx.translate(-this._viewport.x, -this._viewport.y)
123 			view.wrapRender(ctx, dt);
124 			ctx.restore();
125 		}
126 	}

129 	this.tick = function(dt) {
130 		var mode = this.getMode();	
131 		if (mode == 'follow') {
132 			// TODO: zoom to fit in all followed targets, perhaps.
133 			// TODO: adhere to x/yLyles constraints
134 		
135 			if (this._followTargets.length) {
136 				if (!this._viewport) { return; }
137 				var x = 0;
138 				var y = 0;
139 				var maxRect = new Rect(this._viewport);
140 				var rect = {};
141 				for (var i = 0, follow; follow = this._followTargets[i]; ++i) {
142 					var pt = follow.getPoint();
143 					var lyles = follow.getLyles();
144 					if (rect.minX === undefined || pt.x - lyles.x < rect.minX) {
145 						rect.minX = pt.x - lyles.x;
146 					}
147 					if (rect.maxX === undefined || pt.x + lyles.x > rect.maxX) {
148 						rect.maxX = pt.x + lyles.x;
149 					}
150 					if (rect.minY === undefined || pt.y - lyles.y < rect.minY) {
151 						rect.minY = pt.y - lyles.y;
152 					}
153 					if (rect.maxY === undefined || pt.y + lyles.y > rect.maxY) {
154 						rect.maxY = pt.y + lyles.y;
155 					}
156 				}
157 				var y = rect.minY + (rect.maxY - rect.minY) / 2;
158 				var x = rect.minX + (rect.maxX - rect.minX) / 2;
159 				
160 				var targetWidth = rect.maxX - rect.minX;
161 				var targetHeight = rect.maxY - rect.minY;
162 				
163 				var targetScale = Math.min(this.style.width / targetWidth, this.style.height / targetHeight)
164 				this._targetScale = targetScale;
165 				
166 			} else {
167 				x = this._centerX;
168 				y = this._centerY;
169 			}
170 			this._targetX = x;
171 			this._targetY = y;
172 		}				
173 		
174 		if (mode == 'scripted') {
175 			if (!this._activeScript) {
176 				if (!this.nextScript()) {
177 					logger.info('finished scripts!');
178 					this.popMode();			
179 					return;
180 				}
181 			}
182 			if (this._activeScript.done(this)) {
183 				this.nextScript()
184 			}
185 		}
186 		
187 		
188 		// common code for free, scripted, and follow modes.
189 		
190 		var moveVec = new Vec2D({
191 			x:this._targetX - this._centerX,
192 			y:this._targetY - this._centerY
193 		});
194 		var maxMoveDistance = dt * this._maxPanSpeed / 1000; //FIXME take units into account
195 		if (moveVec.getMagnitude() > maxMoveDistance) {
196 			moveVec = new Vec2D({
197 				magnitude:maxMoveDistance,
198 				angle:moveVec.getAngle()
199 			});
200 		}
201 		this._centerX += moveVec.x;
202 		this._centerY += moveVec.y;
203 		
204 		var maxScale = dt * this._maxZoomSpeed / 1000;
205 		var targetScale = Math.min(Math.max(this._targetScale, this._minZoom), this._maxZoom);
206 		if (targetScale > this._scale) {
207 			this._scale = Math.min(targetScale, this._scale + maxScale);
208 		} else if (targetScale < this._scale) {
209 			this._scale = Math.max(targetScale, this._scale - maxScale);
210 		}
211 		
212 	}

214 	this.setMode = function(mode) {
215 		if (mode == this.getMode()) { return true; }
216 		if (this._lock) {
217 			if (this._lockEnforce) { return false; }
218 			throw new Error("mode is locked"); 
219 		}
220 		
221 		this._mode = mode;
222 		return true;
223 	}

229 	this._getPointArgs = function(args) {
230 		var x = args[0];
231 		var y = args[1];
232 		if (x && typeof x.x == 'number' && typeof x.y == 'number') {
233 			return x;
234 		}
235 		if (typeof x == 'object' && x.length) {
236 			x = x[0];
237 			y = x[1];
238 		}
239 		return new Point(x,y);
240 	}

301 		var units = 'pt';
302 		if (typeof speed == 'string') {
303 			var units = speed.slice(speed.length-2);
304 			var speed = parseFloat(speed.slice(0, speed.length-2));
305 		}
306 		this._maxPanSpeed = speed;
307 		this._maxPanUnits = units;

341 	// Follow mode
342 	
343 	this.follow = function() {
344 		var targets = _.flatten(arguments);
345 		logger.info('targets are', targets);
346 		if (!targets.length) { return; }
347 		this.setMode('follow');
348 		var newFollows = [];
349 		for (var i = 0; i < targets.length; i++) {
350 			var target = targets[i];
351 			var follow = new Follow({target:target});
352 			this._followTargets.push(follow);
353 			newFollows.push(follow);
354 		}
355 		return newFollows;
356 	}
357 	
358 	this.removeFollow = function(follow) {
359 		// TODO: too verbose?
360 		this._followTargets = _.without(follow);
361 	}
362 	
363 	
364 	// Scripted Mode
365 	this.pushScript = function(script) {
366 		logger.log(arguments);
367 		logger.log(script);
368 		this._scriptQueue = this._scriptQueue.concat(_.flatten(arguments));	
369 	}
370 	
371 	this.nextScript = function() {
372 		if (this._scriptQueue.length > 0) {
373 			this._activeScript = this._scriptQueue.shift();
374 			this._activeScript.apply(this);
375 			return true;
376 		}
377 		else {
378 			//TODO: callback on done?
379 			//we should go back to free mode now, even if locked.
380 			//FIXME: if we give a callback for the end of a script
381 			// sequence, maybe we shouldn't automatically unlock
382 			// and go to free mode.  The callback fn should implement it.
383 			this._lock = false;
384 			this.setMode('free');
385 			return false;
386 		}
387 	}

395 var Follow = CameraView.Follow = Class('Follow', function(loger, supr) {
396 	this.init = function(opts) {
397 		this._viewCenter = opts.viewCenter || null;
398 		this._cb = opts.cb || null;
399 		this._pt = opts.pt || null;
400 		this._target = opts.target || null;
401 		
402 		this.setLyles(opts.xLyles, opts.yLyles);
403 	}
404 	
405 	
406 	this.getLyles = function() {
407 		return new Point(this._xLyles, this._yLyles);
408 	}
409 	this.getPoint = function() {
410 		if (this._target.style && this._target.style.x !== undefined && this._target.style.y !== undefined) {
411 			if (this._target.style.width !== undefined && this._target.style.height !== undefined) {
412 				return new Point(
413 					this._target.style.x + this._target.style.width/2,
414 					this._target.style.y + this._target.style.height/2
415 				);
416 			}
417 			return new Point(this._target.style);
418 		}
419 		if (this._target.x !== undefined && this._target.y !== undefined) {
420 			if (this._target.width  !== undefined&& this._target.height !== undefined) {
421 				return new Point(
422 					this._target.x + this._target.width/2,
423 					this._target.y + this._target.height/2
424 				);
425 			}
426 			return new Point(this._target);
427 		}
428 		if (typeof this._target == 'function') {
429 			var obj = this._target();
430 			return this.getPoint.call({_target: obj});
431 		}
432 	}
433 	
434 	this.getX = function() {
435 		if (this._target.style && this._target.style.x) {
436 			return this._target.style.x;
437 		}
438 		if (this._target.x) {
439 			return this._target.x;
440 		}
441 		if (typeof this._target == 'function') {
442 			var pt = this._target();
443 			return pt.x;
444 		}
445 	}
446 	this.getY = function() {
447 		if (this._target.style && this._target.style.y) {
448 			return this._target.style.y;
449 		}
450 		if (this._target.y) {
451 			return this._target.y;
452 		}
453 		if (typeof this._target == 'function') {
454 			var pt = this._target();
455 			return pt.y;
456 		}
457 	}
458 	
459 	this.setLyles = function(xLyles, yLyles) {
460 		var units = 'pt';
461 		if (typeof(xLyles) == 'string') {
462 			var units = xLyles.slice(yLyles.length-2);
463 			xLyles = parseFloat(xLyles.slice(0, xLyles.length-2));
464 			yLyles = parseFloat(yLyles.slice(0, yLyles.length-2));
465 		}
466 		this._xLyles = xLyles;
467 		this._yLyles = yLyles;
468 		this._units = units;
469 	}
