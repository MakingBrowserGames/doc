1 "use import";
  2 
  3 import util.setProperty;
  4 
  5 var STYLE_KEYS = {};
  6 
  7 var ViewStyle = Class(function() {
  8 	
  9 	// keys map to properties
 10 	var props = {
 11 		'x': {value: 0},
 12 		'y': {value: 0},
 13 		'anchorX': {value: 0},
 14 		'anchorY': {value: 0},
 15 		'width': {cb: '_onResize'},
 16 		'height': {cb: '_onResize'},
 17 		'widthPercent': {cb: '_onResize'},
 18 		'heightPercent': {cb: '_onResize'},
 19 		'r': {value: 0},
 20 		'opacity': {value: 1},
 21 		'zIndex': {value: 0, cb: '_onZIndex'},
 22 		'radius': {
 23 			get: function() {
 24 				return this._cache && 'radius' in this._cache ? this._cache.radius : this.updateRadius();
 25 			}
 26 		},
 27 		'scale': {value: 1},
 28 		'visible': {value: true},
 29 		'shadowColor': {value: 'black'},
 30 		'clip': {value: false},
 31 		'backgroundColor': {value: undefined}
 32 	};
 33 	
 34 	for (var key in props) {
 35 		STYLE_KEYS[key] = true;
 36 		util.setProperty(this, key, props[key]);
 37 	}
 38 	
 39 	this._cache = null;
 40 	
 41 	this.init = function(view) {
 42 		this._view = view;
 43 	}
 44 	
 45 	this.clearCache = function() { this._cache = {}; }
 46 	
 47 	// exports the current style object
 48 	this.copy = function() {
 49 		var copy = {};
 50 		for (var key in props) {
 51 			copy[key] = this[key];
 52 		}
 53 		return copy;
 54 	}
 55 	
 56 	this.update = function(style) {
 57 		for (var i in style) {
 58 			if (style.hasOwnProperty(i) && props.hasOwnProperty(i)) {
 59 				this[i] = style[i];
 60 			}
 61 		}
 62 	}
 63 	
 64 	this.updateRadius = function() {
 65 		var w = this.width * 0.5,
 66 			h = this.height * 0.5;
 67 		
 68 		if (!this._cache) { this._cache = {}; }
 69 		return (this._cache.radius = Math.sqrt(w * w + h * h));
 70 	}
 71 	
 72 	this._directSetSize = function(width, height) {
 73 		this._width = width;
 74 		this._height = height;
 75 	}
 76 	
 77 	this._onResize = function(prop, value, prevValue) {
 78 		if (value !== undefined) {
 79 			switch (prop) {
 80 				case '_widthPercent':
 81 					this._widthPercent = value;
 82 					this._width = this._view._superview && this._view._superview.style.width * value / 100;
 83 					break;
 84 				case '_heightPercent':
 85 					this._heightPercent = value;
 86 					this._height = this._view._superview && this._view._superview.style.height * value / 100;
 87 					break;
 88 				case '_width':
 89 					this._widthPercent = null;
 90 					break;
 91 				case '_height':
 92 					this._heightPercent = null;
 93 					break;
 94 			}
 95 		}
 96 		
 97 		// local properties are invalidated
 98 		this._cache = {};
 99 		
100 		// child view properties might be invalidated
101 		this._view.needsReflow();
102 	}
103 	
104 	var LEN_Z = 8;
105 	var MAX_Z = 99999999;
106 	var MIN_Z = -99999999;
107 	var PAD = "00000000";
108 	
109 	this._sortIndex = "00000000";
110 	
111 	this._onZIndex = function(_, zIndex) {
112 		zIndex = ~~zIndex;
113 
114 		if (zIndex < MIN_Z) { zIndex = this._zIndex = MIN_Z; }
115 		if (zIndex > MAX_Z) { zIndex = this._zIndex = MAX_Z; }
116 		if (zIndex < 0) {
117 			zIndex *= -1;
118 			this._sortIndex = '-' + PAD.substring(0, LEN_Z - ('' + zIndex).length) + zIndex;
119 		} else {
120 			this._sortIndex = PAD.substring(0, LEN_Z - ('' + zIndex).length) + zIndex;
121 		}
122 
123 		this.setSortKey();
124 		this._view.needsRepaint();
125 
126 		var superview = this._view.getSuperview();
127 		if (superview) { superview.needsSort(); }
128 	}
129 
130 	this.setAddedAt = function(addedAt) {
131 		this._addedAt = addedAt;
132 		this.setSortKey();
133 	}
134 
135 	this.setSortKey = function() {
136 		this._view.__sortKey = this._sortIndex + this._addedAt;
137 	}
138 });
139 
140 exports = ViewStyle;
141 exports.keys = STYLE_KEYS;
