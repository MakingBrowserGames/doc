1 "use import";
  2 /**
  3 
  4 This class represents an input event.
  5 
  6 It assumes a tree structure of View objects.  The top (root) of the
  7 tree is typically the main Application view.  
  8 
  9 Events are assigned a root and a target.  View::dispatchEvent
 10 computes the root and target from the event location (x, y).
 11 
 12 Propogation has two phases: capturing and bubbling.  Users 
 13 can mostly ignore capturing as the primary hooks are in the 
 14 bubbling.  Bubbling is defined as calling event handlers on
 15 each view starting with 'target' and then continuing up 
 16 superview pointers until reaching 'root'.  Views can listen
 17 to the event bubbling by adding methods such as:
 18  "input:start" -> "function onInputStart(evt)"
 19  "input:drag" -> "function onDrag(dragEvt, moveEvt, delta)" *
 20 
 21 Other code can hook into a view's events by calling:
 22   myView.subscribe('input:start', function() { alert('Mouse Down'); });
 23   myView.subscribe('input:end', this, 'onChildViewClick');
 24 
 25 Propogation of events can be cancelled by calling evt.cancel().
 26 
 27 * Note that onDrag is a special event that receives not only
 28   a move event from an 'input:move' event, but also a custom
 29   dragEvt object that contains extra data such as the start
 30   position of the drag.
 31 
 32 **/
 33 
 34 import math2D.Point as Point;
 35 import ..Timer;
 36 
 37 var InputEvent = exports = Class(function() {
 38 	this.cancelled = false; // If true, this event will not propogate
 39 	this.depth = 0; // Number of levels of the tree from root to target (inclusive)
 40 	
 41 	// Note that under normal usage:
 42 	//   this.depth == this.trace.length
 43 	//   this.root = this.trace[this.trace.length - 1]
 44 	//   this.target = this.trace[0]
 45 	
 46 	this.init = function(id, evtType, x, y, root, target) {
 47 		// unique ID for a particular input - the ID should be constant for a given input
 48 		// for example, the mouse should always have the same ID.  Each finger (touch)
 49 		// should have the same ID throughout the touch start/move/end process
 50 		this.id = id; 
 51 		
 52 		// string evtType, e.g. 'input:start'
 53 		this.type = evtType;
 54 		
 55 		// localized point coordinates, indexed by a view's uid (View::uid)
 56 		this.pt = {};
 57 		
 58 		// raw (x, y) coordinates
 59 		this.srcPt = new Point(x, y);
 60 		
 61 		// list of View nodes from target to root
 62 		this.trace = [];
 63 		
 64 		// Top-most view where event is dispatched (e.g. the tree root)
 65 		this.root = root || null;
 66 		
 67 		// time of dispatch
 68 		this.when = Timer.now;
 69 		
 70 		// Bottom-most view where the event occurred
 71 		this.target = target || null;
 72 	}
 73 	
 74 	this.cancel = function() {
 75 		this.cancelled = true;
 76 	}
 77 
 78 	this.clone = function() {
 79 		return new InputEvent(this.id, this.type, this.srcPt.x, this.srcPt.y, this.root, this.target);
 80 	}
 81 });
