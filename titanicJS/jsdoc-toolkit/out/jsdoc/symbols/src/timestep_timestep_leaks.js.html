1 /**
  2 
  3 leaks.js
  4 
  5 We cache many things in timestep, particularly when parsing common strings.
  6 This class allows us to monitor potential memory leaks.  To use, call:
  7 
  8    leaks.register(name, cb);
  9 
 10 where name is a string identifier (need not be unique) and cb
 11 is a monitoring function.  The monitoring function should return
 12 something.  For example, if you want to count the number of 
 13 keys in some cache, your monitoring function would look like:
 14 
 15    var cache = {};
 16    function monitor() {
 17        var count = 0; for (var key in cache) { ++count; }
 18        return count;
 19    }
 20    leaks.register('foo', monitor);
 21 
 22 Since counting the number of keys in the object is a common task,
 23 you can just call:
 24 
 25    leaks.register('foo', leaks.createObjectKeyCounter(cache));
 26 
 27 to create a monitoring function that counts keys.
 28 
 29 */
 30 
 31 /**
 32  * @namespace
 33  */
 34 	
 35 if (!GLOBAL.__timestep_leak_monitor) {
 36 	GLOBAL.__timestep_leak_monitor = {
 37 		count: 0
 38 	};
 39 }
 40 
 41 var cbs = GLOBAL.__timestep_leak_monitor;
 42 
 43 exports.register = function(name, cb) {
 44 	var id = cbs.count++;
 45 	cbs[id + ':' + name] = cb;
 46 }
 47 
 48 exports.monitor = function(interval) {
 49 	setInterval(exports.report, interval || 1000);
 50 }
 51 
 52 exports.report = function() {
 53 	for (var key in cbs) {
 54 		try {
 55 			logger.log(key, cbs[key]());
 56 		} catch(e) {
 57 			logger.log(key, e.message);
 58 		}
 59 	}
 60 }
 61 
 62 exports.createObjectKeyCounter = function(obj) {
 63 	return function() {
 64 		var count = 0;
 65 		for (var id in obj) {
 66 			++count;
 67 		}
 68 		return count;
 69 	}
 70 }
 71 
 72 // monitor how many things we're monitoring
 73 exports.register('leaks', exports.createObjectKeyCounter(cbs));
