 40 	this.onResize = function(w, h) {
 41 		jsio('import timestep.device');
 42 		var Canvas = timestep.device.get('Canvas');
 43 		
 44 		this._buffer0 = new Canvas({width: w, height: h}).getContext('2d');

 50 	this.render = function(ctx) {
 51 		isoProfile.start('_doRender');
 52 		
 53 		if (!this._isReady) { return; }
 54 		
 55 		var last = this._lastViewport,
 56 			viewport = ctx._viewport,
 57 			repaintAll = !last || (viewport.scale != last.scale),
 58 			w = viewport.width + this._bufferLeft + this._bufferRight,
 59 			h = viewport.height + this._bufferTop + this._bufferBottom,
 60 			dx = 0,
 61 			dy = 0,
 62 			x = viewport.x - this._bufferLeft,
 63 			y = viewport.y - this._bufferTop,
 64 			updatedBuffer = false;
 65 		
 66 		if (!this._buffer0) {
 67 			this.onResize(w, h);
 68 		}
 69 		
 70 		if (repaintAll) {
 71 			updatedBuffer = true;
 72 			this._buffer0.save();
 73 			this._buffer0.loadIdentity();
 74 			this.paintRect(this._buffer0, 0, 0, x, y, w, h);
 75 			this._buffer0.restore();
 76 		} else {
 77 			dx = x - last.x;
 78 			dy = y - last.y;
 79 			
 80 			// should we redraw the sides of the buffer?
 81 			updatedBuffer = dx || dy; //dx < -this._bufferLeft || dx > this._bufferRight || dy < -this._bufferTop || dy > this._bufferBottom;
 82 			
 83 			if (updatedBuffer) {
 84 				isoProfile.start('_doRender:updateBuffer');
 85 				
 86 				this._buffer1.save();
 87 				this._buffer1.loadIdentity();
 88 				this._buffer1.clear();
 89 				
 90 				if (dx < 0) {
 91 					this.paintRect(this._buffer1, 0, 0, x, y, -dx, h);
 92 				} else if (dx > 0) {
 93 					this.paintRect(this._buffer1, w - dx, 0, x + w - dx, y, dx, h);
 94 				}
 95 				
 96 				var offsetX = 0,
 97 					offsetW = 0;
 98 				if (dy && dx) {
 99 					if (dx < 0) { // left-side drawn, so shift left
100 						offsetX = -dx;
101 						offsetW = dx;
102 					} else { // right-side drawn, so shift right
103 						offsetW = -dx;
104 					}
105 				}
106 				
107 				if (dy < 0) {
108 					this.paintRect(this._buffer1, offsetX, 0, x + offsetX, y, w + offsetW, -dy);
109 				} else if (dy > 0) {
110 					this.paintRect(this._buffer1, offsetX, h - dy, x + offsetX, y + h - dy, w + offsetW, dy);
111 				}
112 				
113 				//ctx.drawImage(this._buffer1.canvas, x + w + 10, y);
114 				
115 				isoProfile.start('_doRender:drawBuffer');
116 				this._buffer1.drawImage(this._buffer0.canvas, -dx, -dy);
117 				isoProfile.stop('_doRender:drawBuffer');
118 				
119 				this._buffer1.restore();
120 				
121 				dx = dy = 0;
122 				
123 				var swap = this._buffer0;
124 				this._buffer0 = this._buffer1;
125 				this._buffer1 = swap;
126 				
127 				isoProfile.stop('_doRender:updateBuffer');
128 				
129 				//isoProfile.report();
130 			}
131 		}
132 		
133 		//logger.log(x, y, dx, dy);
134 		ctx.drawImage(this._buffer0.canvas, x - dx, y - dy);
135 		
136 		//ctx.fillStyle = 'rgba(255,255,0,0.2)';
137 		//ctx.fillRect(viewport.x, viewport.y, viewport.width, viewport.height);
138 		
139 		if (this._dirtyTiles.length) {
140 			
141 		}
142 		
143 		if (updatedBuffer) {
144 			this._lastViewport = {
145 				x: x,
146 				y: y,
147 				width: w,
148 				height: h
149 			};
150 		}
151 
152 		///////// DEBUG CODE ////////
153 		var numRows = this.getNumRows();
154 		var numCols = this.getNumCols();
155 
156 		for(var i = -2; i < numRows + 2; ++i){
157 			for (var j = -2; j < numCols + 2; j++){
158 				var coor = this.getCartesianForIso(i,j);
159 				var coori = this.getCartesianForIso(i + 1, j);
160 				var coorj = this.getCartesianForIso(i, j + 1);
161 				var coorij = this.getCartesianForIso(i + 1, j + 1);
162 				var coorMid = this.getCartesianForIso(i + 0.5, j + 0.5);
163 				ctx.beginPath();
164 				ctx.strokeStyle = 'rgb(0,0,0)';
165 				ctx.moveTo(coor.x,coor.y);
166 				ctx.lineTo(coori.x,coori.y);
167 				ctx.lineTo(coorij.x,coorij.y);
168 				ctx.lineTo(coorj.x,coorj.y);
169 				ctx.lineTo(coor.x,coor.y);
170 				ctx.stroke();
171 
172 				ctx.fillStyle = 'rgb(0,0,0)';
173 //				ctx.fillText('' + i + ',' + j, coorMid.x - 10, coorMid.y + 5, 50, 50);
174 				ctx.fillText('' + coorMid.x + ',' + coorMid.y, coorMid.x - 10, coorMid.y + 5, 50, 50);
175 			}
176 		}
177 		
178 		
179 
180 		///////// END DEBUG CODE ////////
181 
182 		isoProfile.stop('_doRender');
183 		
184 	}

228 	this.paintRect = function(ctx, offsetX, offsetY, vx, vy, vw, vh) {
229 		
230 //		logger.log('paintRect', vx, vy, vw, vh, 'offset:', offsetX, offsetY);
231 		ctx.save();
232 		ctx.clipRect(offsetX, offsetY, vw, vh);
233 		
234 		isoProfile.start('_doRender:paintRect');
235 		
236 		//ctx.save();
237 		offsetX -= vx;
238 		offsetY -= vy - 15;
239 		// center on the buffer
240 		//ctx.translate(offsetX - x, offsetY - y);
241 		
242 		//ctx.strokeStyle = 'white';
243 		//ctx.strokeRect(vx + offsetX, vy + offsetY, vw, vh);
244 		
245 		// drawing must occur from top to bottom, left to right
246 		//
247 		//        0, 0
248 		//       /    \
249 		//     /        \
250 		//  +y           +x
251 		//
252 		// Outer-loop: Go across the top row from right to left with delta:
253 		//      (i, j) += (-1, +1)
254 		//   Inner-loop: for each square along the top row, iterate down
255 		//   the row (positive y direction) by incrementing i.  Once the
256 		//   tile is out of bounds, stop this inner-loop.
257 		// 
258 		// If the first tile of the inner-loop is out of bounds, we know we've
259 		// reached the end of the top-row. Start iterating down the right side
260 		// of the grid instead. Go back to the last valid tile (top-right-most
261 		// tile) and continue iterating with a new outer-loop iteration delta:
262 		//     (i, j) += (+1, +1)
263 		
264 		var data = this._dataSource,
265 			numLayers = data.getNumLayers(),
266 			painter = this._painter,
267 			step4 = this._tileWidth,
268 			step2 = step4 / 2,
269 			step1 = step2 / 2,
270 			k = -1;
271 		
272 		while (++k < numLayers) {
273 			var imgData = data.getImageData()[k],
274 				minX = vx - (imgData.tileWidth - step2),
275 				maxX = vx + vw + (imgData.tileWidth - step2),
276 				minY = vy - imgData.tileHeight + step2,
277 				maxY = vy + vh,
278 				i = Math.floor((minY - minX / 2) / step2),
279 				j = Math.floor((minY + minX / 2) / step2);
280 				x = (j - i) * step2,
281 				y = (j + i) * step1 - imgData.tileHeight + step2;
282 			
283 			while (x - step4 <= maxX) {
284 				drawColBefore(i, j, x, y);
285 				
286 				// move right
287 				x += step4;
288 				i--;
289 				j++;
290 			}
291 			
292 			while (x - step2 > maxX) {
293 				// move down & left
294 				++i;
295 				y += step1;
296 				x -= step2;
297 			}
298 		
299 			while (y + step2 < maxY) {
300 				drawColBefore(i, j, x, y);
301 			
302 				// move down
303 				y += step2;
304 				i++;
305 				j++;
306 			}
307 		}
308 		
309 		function drawColBefore(i, j, x, y) {
310 			isoProfile.start('_doRender:drawColBefore');
311 			while (x + step2 >= minX && y - step2 <= maxY) {
312 				var img = data.getTileAt(i, j, k);
313 				img && img.render(ctx, x - step2 + offsetX, y + offsetY);
314 				//paintDebugInfo(ctx, img, step4, i, j, k, x + offsetX, y + offsetY);
315 				y += step1;
316 				x -= step2;
317 				i++;
318 			}
319 			isoProfile.stop('_doRender:drawColBefore');
320 		}
321 		
322 		isoProfile.stop('_doRender:paintRect');
323 		ctx.restore();
324 		
325 //		logger.log('end paint rect');
326 	};

328 	this.getTileForCartesian = function(x, y) {
329 		var pos = this.getIsoForCartesian(x, y);
330 		return {
331 			i: Math.floor(pos.i),
332 			j: Math.floor(pos.j)
333 		}
334 	};
335 	
336 	// map screen x/y coordinate to isometric tile coordinates
337 	this.getIsoForCartesian = function(x, y) {
338 		var width = this._tileWidth / 2;
339 		
340 		var i = y - x / 2,
341 		    j = y + x / 2;
342 		
343 		return {
344 			i: i / width,
345 			j: j / width
346 		};
347 	};
348 	
349 	// map isometric tile coordinates to screen XY coordinates
350 	this.getCartesianForIso = function(i, j) {
351 		return {
352 			x: (j - i) * this._tileWidth / 2,
353 			y: (j + i) * this._tileWidth / 4
354 		};
355 	};

357 	this.getIsoDistance = function(pt1, pt2){
358 		return Math.sqrt(Math.pow(pt1.i - pt2.i,2) + Math.pow(pt1.j - pt2.j,2));
359 	};
360 	
361 	this.getSquareIsoDistance = function(pt1, pt2) {
362 		return Math.pow(pt1.i - pt2.i,2) + Math.pow(pt1.j - pt2.j,2);
363 	};
