 68 	var BOUNCE_MAX_DIST = 50;
 69 	var PI_2 = Math.PI / 2;
 70 	
 71 	this.getStyleBounds = function() {
 72 		var bounds = bounds = this._scrollBounds;
 73 		var minY = -bounds.maxY;
 74 		var maxY = -bounds.minY < minY ? minY : -bounds.minY;
 75 		var minX = -bounds.maxX;
 76 		var maxX = -bounds.minX < minX ? minX : -bounds.minX;
 77 		return {
 78 			minY: minY,
 79 			maxY: maxY,
 80 			minX: minX,
 81 			maxX: maxX
 82 		};
 83 	}

 85 	this.getOffset = function() { return new math2D.Point(this._contentView.style); }
 86 	this.setOffset = function(x, y) {
 87 		var style = this._contentView.style,
 88 			delta = {
 89 				x: 0,
 90 				y: 0
 91 			};
 92 		
 93 		var bounds = this.getStyleBounds();
 94 		
 95 		if (typeof x == 'number') {
 96 			if (this._isBouncing) {
 97 				// do nothing
 98 			} else if (this._canBounce) {
 99 				if (x < bounds.minX) { 
100 					var delta = (bounds.minX - x) / BOUNCE_MAX_DIST;
101 					x = bounds.minX - Math.atan(delta) * BOUNCE_MAX_DIST / PI_2;
102 				}
103 				
104 				if (x > bounds.maxX) {
105 					var delta = (x - bounds.maxX) / BOUNCE_MAX_DIST;
106 					x = bounds.maxX + Math.atan(delta) * BOUNCE_MAX_DIST / PI_2;
107 				}
108 			} else {
109 				if (x < bounds.minX) { x = bounds.minX; }
110 				if (x > bounds.maxX) { x = bounds.maxX; }
111 			}
112 			
113 			if (x != style.x) {
114 				delta.x = x - style.x;
115 				style.x = x;
116 			}
117 		}
118 
119 		if (typeof y == 'number') {
120 			if (this._isBouncing) {
121 				// do nothing
122 			} else if (this._canBounce) {
123 				if (y < bounds.minY) { 
124 					var delta = (bounds.minY - y) / BOUNCE_MAX_DIST;
125 					y = bounds.minY - Math.atan(delta) * BOUNCE_MAX_DIST / PI_2;
126 				}
127 				
128 				if (y > bounds.maxY) {
129 					var delta = (y - bounds.maxY) / BOUNCE_MAX_DIST;
130 					y = bounds.maxY + Math.atan(delta) * BOUNCE_MAX_DIST / PI_2;
131 				}
132 			} else {
133 				if (y < bounds.minY) { y = bounds.minY; }
134 				if (y > bounds.maxY) { y = bounds.maxY; }
135 			}
136 			
137 			if (y != style.y) {
138 				delta.y = y - style.y;
139 				style.y = y;
140 			}
141 		}
142 		
143 		//if (delta.x || delta.y) {
144 			this.needsReflow();
145 			this.publish('Scrolled', delta);
146 		//}
147 	}

161 	this.onDragStart = function() {
162 		this._contentView.getInput().blockEvents = true;
163 		this.startBounce();
164 		this._animState.offset = this.getOffset();
165 		this._anim = animate(this._animState).clear();
166 	}

168 	this.onDrag = function(dragEvt, moveEvt, delta) {
169 		var state = this._animState;
170 		state.dt = delta.t;
171 		state.lastDelta = delta;
172 
173 		if (!this._opts.scrollY) { delta.y = 0; }
174 		if (!this._opts.scrollX) { delta.x = 0; }
175 
176 		this.setOffset(state.offset.x += delta.x, state.offset.y += delta.y);
177 		moveEvt.cancel();
178 	}
179 
180 	this.onDragStop = function(dragEvt, selectEvt) {
181 		this._contentView.getInput().blockEvents = false;
182 
183 		if (this._opts.inertia){
184 			var delta = new math2D.Point(this._animState.lastDelta).scale(this._acceleration);
185 			var offset = this._animState.offset;
186 			var distance = delta.getMagnitude();
187 			
188 			// if we overshot the bounds, don't waste time animating the acceleration.
189 			var bounds = this.getStyleBounds();
190 			var target = new math2D.Point(offset).add(delta);
191 			if (target.y < bounds.minY - BOUNCE_MAX_DIST) {
192 				if (offset.y < bounds.minY) {
193 					distance = 0;
194 				} else {
195 					delta.y -= (bounds.minY - BOUNCE_MAX_DIST - target.y);
196 					distance = delta.getMagnitude();
197 				}
198 			}
199 			
200 			if (target.y > bounds.maxY + BOUNCE_MAX_DIST) {
201 				if (offset.y > bounds.maxY) {
202 					distance = 0;
203 				} else {
204 					delta.y -= (target.y - (bounds.maxY - BOUNCE_MAX_DIST));
205 					distance = delta.getMagnitude();
206 				}
207 			}
208 			
209 			if (distance) {
210 				this._anim.now(bind(this, function(tt, t) {
211 					this.setOffset(
212 						offset.x + delta.x * tt,
213 						offset.y + delta.y * tt);
214 				}), 100 * Math.log((distance + 1) * 100), animate.easeOut).then(bind(this, function() {
215 					this.endBounce(offset);
216 				}));
217 			} else {
218 				this.endBounce(offset);
219 			}
220 		}
221 	}
222 	
223 	this.startBounce = function() {
224 		this._isBouncing = false;
225 		if (this._opts.inertia && this._opts.bounce) {
226 			this._canBounce = true;
227 		}
228 	}
229 	
230 	this.endBounce = function() {
231 		var offset = this.getOffset();
232 		var bounds = this.getStyleBounds();
233 		
234 		var dx = 0;
235 		var dy = 0;
236 		
237 		var ty = offset.y, tx = offset.x;
238 		if (offset.y < bounds.minY) {
239 			ty = bounds.minY;
240 		} else if (offset.y > bounds.maxY) {
241 			ty = bounds.maxY;
242 		}
243 		
244 		if (offset.x < bounds.minX) {
245 			tx = bounds.minX;
246 		} else if (offset.x > bounds.maxX) {
247 			tx = bounds.maxX;
248 		}
249 		
250 		dy = ty - offset.y;
251 		dx = tx - offset.x;
252 
253 		if (dy === 0 && dx === 0) { return; }
254 		
255 		this._isBouncing = true;
256 		this._anim.now(bind(this, function(tt, t) {
257 			this.setOffset(
258 				offset.x + dx * tt,
259 				offset.y + dy * tt);
260 		}), 500, animate.easeInOut).then(bind(this, function() {
261 			this._canBounce = false;
262 			this._isBouncing = false;
263 		}));
264 	}

267 	this.getScrollBounds = function() { return this._scrollBounds; }

270 		this.setOffset(

272 				y != undefined && y != null && (this._contentView.style.y + y)

274 	}

328 	this.scrollTo = function(x, y, duration, cb) {
329 		duration = duration || 500;
330 		var bounds = this.getScrollBounds();
331 			
332 		x = x < bounds.minX ? bounds.minX : x;
333 		x = x > bounds.maxX ? bounds.maxX : x;
334 		y = y < bounds.minY ? bounds.minY : y;
335 		y = y > bounds.maxY ? bounds.maxY : y;
336 		
337 		var anim = animate(this._contentView).now({ x: -x, y: -y }, duration, animate.easeOut);
338 		
339 		if (cb) {
340 			anim.then(cb);
341 		}
