1 "use import";
  2 
  3 import .UIView;
  4 
  5 /**
  6  * @extends timestep.ui.UIView
  7  */
  8 exports = Class(UIView, function(supr) {
  9 	var CYCLES_PER_SECOND = 0.5;
 10 	var RADIUS = 10;
 11 	var SPOKES = 20;
 12 	var THICKNESS = 2;
 13 	var TRAIL = 10;
 14 	
 15 	this.tag = 'spinner';
 16 	this._t = 0;
 17 	this.init = function() {
 18 		this._step = 2 * Math.PI / SPOKES;
 19 		
 20 		supr(this, 'init', arguments);
 21 	}
 22 	
 23 	this.tick = function(dt) {
 24 		this._t += dt;
 25 		
 26 		var r = (this._t / 1000 % (1 / CYCLES_PER_SECOND)) * Math.PI;
 27 		
 28 		var oldR = this._r;
 29 		this._r = r - (r % this._step);
 30 		
 31 		if (oldR != r) {
 32 			this.needsRepaint();
 33 		}
 34 	}
 35 	
 36 	this.render = function(ctx) {
 37 		ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
 38 		
 39 		var w = this.style.width, h = this.style.height;
 40 		
 41 		var x;
 42 		var r2 = RADIUS * RADIUS;
 43 		for (var y = 0; y < RADIUS; ++y) {
 44 			j = y + 1;
 45 			x = Math.round(RADIUS - Math.sqrt(2 * j * RADIUS - j * j));
 46 			ctx.fillRect(x, y, w - 2 * x, 1);
 47 		}
 48 		
 49 		var y = h - RADIUS;
 50 		ctx.fillRect(0, RADIUS, w, y - RADIUS);
 51 		
 52 		var j;
 53 		for (var i = 0; i < RADIUS; ++i) {
 54 			j = RADIUS - i;
 55 			x = Math.round(RADIUS - Math.sqrt(2 * j * RADIUS - j * j));
 56 			ctx.fillRect(x, y + i, w - 2 * x, 1);
 57 		}
 58 		
 59 		ctx.fillStyle = 'white';
 60 		ctx.translate(w / 2, h / 2);
 61 		w /= 2;
 62 		ctx.rotate(this._r);
 63 		
 64 		for (var i = 0; i < SPOKES; ++i) {
 65 			ctx.rotate(this._step);
 66 			ctx.globalAlpha = Math.max(0.1, (i - TRAIL) / TRAIL);
 67 			ctx.fillRect(10, -THICKNESS / 2, w - 15, THICKNESS);
 68 		}
 69 	}
 70 });
