1 "use import";
  2 
  3 import lib.PubSub;
  4 import .SoundManager;
  5 from util.underscore import _;
  6 
  7 var soundManager = new SoundManager();
  8 soundManager.url = 'media/swf';
  9 soundManager.flashVersion = 9;
 10 soundManager.useMovieStar = true;
 11 soundManager.debugMode = false;
 12 soundManager.consoleOnly = true;
 13 soundManager.useHighPerformance = true;
 14 soundManager.useFastPolling = true;
 15 
 16 /**
 17  * @extends lib.PubSub
 18  */
 19 var AudioAPI = exports = Class(lib.PubSub, function(supr) {
 20 	this.init = function(opts) {
 21 
 22 		opts = merge(opts, {
 23 			map: {},
 24 			background: []
 25 		});
 26 
 27 		supr(this, 'init', [opts]);
 28 		var path = opts.path;
 29 		this._map = {};
 30 
 31 		_.each(opts.background, function(name) {
 32 			opts.map[name] = {'name': name}
 33 		}, this);
 34 		
 35 		soundManager.onready(bind(this, function() {
 36 			logger.log('SoundManager onReady');
 37 			for (key in opts.map) {
 38 				logger.log('SoundManager key: ', key);
 39 				var url = 'media/audio/'  + key + '.mp3';
 40 				var k = this._map[key] = soundManager.createSound({
 41 					id: key,
 42 					bufferTime: 3,
 43 					url: url
 44 				});
 45 				k.load();
 46 			}
 47 
 48 			this.publish('Ready');
 49 		}));
 50 	}
 51 	
 52 	this.canPlay = function(name) {
 53 		return (name in this._map);
 54 	}
 55 	
 56 	this.setVolume = function(volume) {
 57 		this._soundPlaying && soundManager.setVolume(this._soundPlaying, volume);
 58 		this._backgroundSoundPlaying && soundManager.setVolume(this._backgroundSoundPlaying, volume);
 59 	}
 60 
 61 	this.setMuted = function(muted) {
 62 		this.muted = muted;
 63 		if (muted) {
 64 			this.setVolume(0);
 65 		}
 66 	}
 67 
 68 	this.play = function(name, volume, channel) {
 69 		if (!this.canPlay(name)) { return; }
 70 		if (this.muted) { return; }
 71 		if (volume === undefined) {
 72 			volume = 1.0;
 73 		}
 74 		this._soundPlaying = name;
 75 		this._map[name].setVolume(volume * 100 | 0);
 76 		this._map[name].play();
 77 	}
 78 
 79 	this.pause = function() {
 80 		this._map[this._soundPlaying].pause();
 81 		this._soundPlaying = null;
 82 	}
 83 
 84 	this.playBackgroundMusic = function(name, volume) {
 85 		if (!this.canPlay(name)) { return; }
 86 		if (this.muted) { return; }
 87 		if (volume === undefined) {
 88 			volume = 1.0;
 89 		}
 90 		this._backgroundSoundPlaying = name;
 91 		this._map[name].setVolume(volume * 100 | 0);
 92 		this._map[name].play();
 93 	}
 94 
 95 	this.pauseBackgroundMusic = function() {
 96 		if (!this._backgroundSoundPlaying) { return; }
 97 		this._map[this._backgroundSoundPlaying].pause();
 98 		this._backgroundSoundPlaying = null;
 99 	}
100 });
