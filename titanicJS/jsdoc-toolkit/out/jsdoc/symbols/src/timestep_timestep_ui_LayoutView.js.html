1 "use import";
  2 
  3 import timestep.View;
  4 import std.js as JS;
  5 from util.underscore import _;
  6 
  7 /**
  8  * @extends timestep.View
  9  */
 10 var LayoutView = exports = Class(timestep.View, function(supr) {
 11 	
 12 	var defaults = {
 13 		margin: 0,
 14 		padding: 0,
 15 		columns: 1
 16 	}
 17 
 18 	this.init = function(opts) {
 19 		this.opts = JS.merge(opts, defaults);
 20 		supr(this, 'init', [opts]);
 21 	}
 22 	
 23 	this.buildView = function() {
 24 		this.layoutSubviews();
 25 	}
 26 
 27 	this.addVerticalPaddingItem = function(height) {
 28 		new timestep.View({
 29 			parent: this,
 30 			height: height
 31 		});
 32 	}
 33 
 34 	function maxY(view) {
 35 		return view.style.y + view.style.height;
 36 	}
 37 	function maxX(view) {
 38 		return view.style.x + view.style.width;
 39 	}
 40 
 41 	this.sizeToFit = function() {
 42 		this.layoutSubviews();
 43 		this.style.height = maxY(_.max(this._subviews, maxY));
 44 		this.style.width = maxX(_.max(this._subviews, maxX));
 45 	}
 46 
 47 	this.layoutSubviews = function() {
 48 		var padding = this.opts.padding;
 49 		var margin = this.opts.margin;
 50 		var i, length = this._subviews.length;
 51 		var x = margin, y = margin, row = 0, col = 0;
 52 
 53 		var view;
 54 		for(i = 0; i < length; i++) {
 55 			view = this._subviews[i];
 56 			if ('layoutSubviews' in view) {
 57 				view.layoutSubviews();
 58 			}
 59 			if (this.opts.columns == 1) {
 60 				view.style.x = (this.opts.width - view.style.width) / 2;
 61 			} else {
 62 				view.style.x = x;
 63 			}
 64 			view.style.y = y;
 65 			col++;
 66 			if (col == this.opts.columns) {
 67 				col = 0;
 68 				x = margin;
 69 				y = (y + view.style.height + padding) | 0;
 70 			} else {
 71 				x = (x + view.style.width + padding) | 0;
 72 			}
 73 		}
 74 	}
 75 
 76 });
