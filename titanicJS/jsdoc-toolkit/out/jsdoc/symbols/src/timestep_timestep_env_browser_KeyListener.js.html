1 "use import";
  2 
  3 import lib.PubSub;
  4 import lib.Enum;
  5 
  6 import ...keyConstants;
  7 import ...device;
  8 import ...Timer;
  9 
 10 from util.browser import $;
 11 
 12 var gListenerSingleton = null;
 13 var gCancelKeys = lib.Enum(keyConstants.SPACE, keyConstants.LEFT, keyConstants.RIGHT, keyConstants.UP, keyConstants.DOWN);
 14 
 15 exports = Class(function() {
 16 	this.init = function(el, events) {
 17 		if (gListenerSingleton) { return gListenerSingleton; }
 18 		gListenerSingleton = this;
 19 		
 20 		this._el = el = el || document;
 21 		this._events = [];
 22 		this._shortcuts = [];
 23 		this._isEnabled = true;
 24 		this._keyMap = {};
 25 		
 26 		$.onEvent(el, 'keydown', this, 'onKeyDown');
 27 		$.onEvent(el, 'keypress', this, 'onKeyPress');
 28 		$.onEvent(el, 'keyup', this, 'onKeyUp');
 29 		$.onEvent(window, 'blur', this, 'liftAll');
 30 	}
 31 	
 32 	this.setEnabled = function(isEnabled) { this._isEnabled = isEnabled; }
 33 
 34 	this.captureShortcut = function(shortcut) {
 35 		this._shortcuts.push(shortcut);
 36 	}
 37 	
 38 	this.getPressed = function() { return this._keyMap; }
 39 
 40 	this.onKeyDown = function(e) {
 41 		
 42 		if (!this._isEnabled) { return; }
 43 		
 44 		var evt = {
 45 			code: e.keyCode,
 46 			ctrl: e.ctrlKey,
 47 			shift: e.shiftKey,
 48 			alt: e.altKey,
 49 			meta: e.metaKey, // for mac keyboards
 50 			lifted: false,
 51 			dt: Timer.getTickProgress()
 52 		};
 53 		
 54 		if (evt.ctrl || evt.shift || evt.alt || evt.meta) {
 55 			var captured = false;
 56 			for (var i = 0, s; s = this._shortcuts[i]; ++i) {
 57 				if (s.compare(evt)) {
 58 					s.publish('Down', evt);
 59 					captured = true;
 60 				}
 61 			}
 62 			
 63 			if (captured) { $.stopEvent(e); }
 64 			return;
 65 		} else {
 66 			// MUST cancel event if we're enabled to prevent browser
 67 			// default behaviors (e.g. scrolling)
 68 			$.stopEvent(e);
 69 		}
 70 		
 71 		// We already know that key is down; ignore repeat events.
 72 		if (e.keyCode in this._keyMap) { return; }
 73 		
 74 		this._events.push(evt);
 75 		this._keyMap[e.keyCode] = +new Date();
 76 	}
 77 	
 78 	this.liftAll = function() {
 79 		var progressDt = Timer.getTickProgress();
 80 		for (var code in this._keyMap) {
 81 			this._events.push({
 82 				code: code,
 83 				lifted: true,
 84 				dt: progressDt
 85 			});
 86 		}
 87 		this._keyMap = {};
 88 	}
 89 	
 90 	this.onKeyUp = function(e) {
 91 		var progressDt = Timer.getTickProgress();
 92 		delete this._keyMap[e.keyCode];
 93 		this._events.push({
 94 			code: e.keyCode,
 95 			lifted: true,
 96 			dt: progressDt
 97 		});
 98 		$.stopEvent(e);
 99 	}
100 	
101 	this.onKeyPress = function(e) {
102 		if (!this._isEnabled) { return; }
103 		if (e.keyCode in gCancelKeys) {
104 			$.stopEvent(e);
105 		}
106 	}
107 	
108 	this.peekEvents = function() { return this._events; }
109 	this.popEvents = function() { return this._events.splice(0, this._events.length); }
110 });
111

115 exports.Shortcut = Class(lib.PubSub, function() {
116 	this.init = function(keyCode, ctrl, shift, alt, meta) {
117 		this.ctrl = !!ctrl;
118 		this.shift = !!shift;
119 		this.alt = !!alt;
120 		this.meta = !!meta;
121 		this.code = !!keyCode;
122 	}
123 	
124 	this.compare = function(shortcut) {
125 		return this.ctrl == shortcut.ctrl 
126 			&& this.alt == shortcut.alt
127 			&& this.meta == shortcut.meta
128 			&& this.shift == shortcut.shift
129 			&& this.code == shortcut.code;
130 	}
131 });
