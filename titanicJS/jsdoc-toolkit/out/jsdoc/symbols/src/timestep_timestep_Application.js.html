1 "use import";
  2 
  3 import lib.PubSub as PubSub;
  4 
  5 import .device;
  6 import .input;
  7 
  8 import .Timer;
  9 import .StackView;
 10 import .View;
 11 
 12 import .ApplicationFPS as ApplicationFPS;
 13 import math2D.Point as Point;
 14 
 15 import timestep.device;
 16 
 17 var __instance = null;
 18 
 19 /**
 20  * @extends lib.PubSub
 21  */
 22 var Application = exports = Class(PubSub, function(supr) {
 23 	this.init = function(opts) {
 24 		if (!__instance) { __instance = this; }
 25 
 26 		this._opts = opts = merge(opts, {
 27 			keyListenerEnabled: true,
 28 			width: device.width,
 29 			height: device.height,
 30 			view: null,
 31 			dtFixed: 0,
 32 			dtMinimum: 0,
 33 			clearEachFrame: true,
 34 			alwaysRepaint: false,
 35 			repaintOnEvent: true,
 36 			mergeMoveEvents: false,
 37 			continuousInputCheck: !device.isMobileBrowser && !device.isMobile,
 38 			showFPS: false
 39 		});
 40 
 41 		Timer.onTick = bind(this, '_tick');
 42 
 43 		this._doubleBuffered = true;
 44 
 45 		var Canvas = device.get('Canvas');
 46 		this._canvas = new Canvas({
 47 			width: opts.width,
 48 			height: opts.height,
 49 			offscreen: false
 50 		});
 51 
 52 		this._ctx = this._canvas.getContext('2d');
 53 
 54 		this._ctx.font = '11px ' + timestep.device.defaultFontFamily;
 55 
 56 		this._view = opts.view || new StackView({
 57 			width: opts.width,
 58 			height: opts.height,
 59 			// TODO: parentDOMNode: timestep.device.useDOM ? this._el : null
 60 		});
 61 
 62 		this._view.__root = this;
 63 		
 64 		this._events = [];
 65 		this._inputListener = new input.InputListener({
 66 			canvas: this._ctx
 67 		});
 68 
 69 		if (input.KeyListener) {
 70 			this._keyListener = new input.KeyListener();
 71 		}
 72 
 73 		this._tickBuffer = 0;
 74 		this._onTick = [];
 75 
 76 		this.updateOpts(this._opts);
 77 	};
 78 
 79 	this.updateOpts = function(opts) {
 80 		this._opts = merge(opts, this._opts);
 81 		if (this._keyListener) {
 82 			this._keyListener.setEnabled(this._opts.keyListenerEnabled);
 83 		}
 84 
 85 		if (this._opts.showFPS) {
 86 			if (!this._applicationFPS) {
 87 				this._applicationFPS = new ApplicationFPS({application: this});
 88 			}
 89 			
 90 			this._renderFPS = bind(this._applicationFPS, this._applicationFPS.render);
 91 			this._tickFPS = bind(this._applicationFPS, this._applicationFPS.tick);
 92 		} else {
 93 			this._renderFPS = function() {};
 94 			this._tickFPS = function() {};
 95 		}
 96 	};
 97 
 98 	this.supports = function(key) { return this._opts[key]; }
 99 
100 	this.getInput = function() { return this._inputListener; }
101 	this.getEvents = function() { return this._events; }
102 	this.getKeyListener = function() { return this._keyListener; }
103 	this.getCanvas = function() { return this._canvas; }
104 	
105 	this.getViewCtor = function() { return View; }
106 	this.getView = function() { return this._view; }
107 	this.setView = function(view) { this._view = view; return this; }
108 
109 	this.show = function() {
110 		if (!device.useDOM) {
111 			this._ctx.show();
112 		}
113 		
114 		return this;
115 	};
116 
117 	this.hide = function() { this._ctx.hide(); return this; };
118 	
119 	this.pause = function() {
120 		this.stopLoop();
121 		if (this._keyListener) {
122 			this._keyListener.setEnabled(false);
123 		}
124 	};
125 
126 	this.resume = function() {
127 		this.startLoop();
128 		if (this._keyListener) {
129 			this._keyListener.setEnabled(true);
130 		}
131 	};
132 
133 	this.startLoop = function(dtMin) {
134 		if (this._running) { return; }
135 		this._running = true;
136 		
137 		this.now = 0;
138 		Timer.start(dtMin || this._opts.dtMinimum);
139 		return this;
140 	};
141 
142 	this.stopLoop = function() {
143 		if (!this._running) { return; }
144 		this._running = false;
145 		
146 		Timer.stop();
147 		return this;
148 	};
149 
150 	this.doOnTick = function(cb) {
151 		if (arguments.length > 1) { cb = bind.apply(this, arguments); }
152 		this._onTick.push(cb);
153 	};
154 
155 	this._tick = function(dt) {
156 		this._needsRepaint = false;
157 		if( this._ctx.getElement) {	
158 			var el = this._ctx.getElement(),
159 				s = this._view.style;
160 			if (el && (s.width != el.width || s.height != el.height)) {
161 				s.width = el.width;
162 				s.height = el.height;
163 				
164 				this._view.needsReflow();
165 			}
166 		}
167 
168 		for (var i = 0, cb; cb = this._onTick[i]; ++i) { cb(dt); }
169 
170 		var events = this._inputListener.getEvents();
171 		var n = events.length;
172 
173 		this._events = events;
174 		if (this._opts._mergeMoveEvents) {
175 			var hasMove = false;
176 			for (var i = n - 1; i >= 0; --i) {
177 				if (events[i].type == input.eventTypes.MOVE) {
178 					if (!hasMove) { 
179 						hasMove = true;
180 					} else {
181 						events.splice(i, 1);
182 					}
183 				}
184 			}
185 		}
186 
187 		for (var i = 0, evt; evt = events[i]; ++i) {
188 			evt.srcApp = this;
189 			input.dispatchEvent(this._view, evt);
190 		}
191 
192 		if (!device.useDOM) {
193 			if (i > 0) {
194 				if (this._opts.repaintOnEvent) { this.needsRepaint(); }
195 			} else if (this._opts.continuousInputCheck) {
196 				var prevMove = input._evtHistory['input:move'];
197 				if (prevMove) {
198 					input.dispatchEvent(this._view, new input.InputEvent(prevMove.id, prevMove.type, prevMove.srcPt));
199 				}
200 			}
201 		}
202 
203 		if (this._opts.dtFixed) {
204 			this._tickBuffer += dt;
205 			while (this._tickBuffer >= this._opts.dtFixed) {
206 				this._tickBuffer -= this._opts.dtFixed;
207 				this.now += this._opts.dtFixed;
208 				this.__tick(this._opts.dtFixed);
209 			}
210 		} else {
211 			this.__tick(dt);
212 		}
213 
214 		var doRepaint = this._opts.alwaysRepaint || this._needsRepaint;
215 		if (!doRepaint && this._doubleBuffered && this._doubleBufferedState > 0) {
216 			// even if we don't repaint, we need to paint at least 1 more time
217 			// for devices that use multiple buffers (like the ios, I think...)
218 			--this._doubleBufferedState;
219 			this.render();
220 		} else if (doRepaint) {
221 			this._doubleBufferedState = 3;
222 			this.render();
223 		}
224 
225 		this._renderFPS(this._ctx, dt);
226 	};
227 
228 	this.render = function() {
229 		if (this._opts.clearEachFrame) {
230 			this._ctx.clear();
231 		}
232 
233 		this._view.wrapRender(this._ctx, {});
234 		this._ctx.swap();
235 	};
236 
237 	this.needsRepaint = function() {
238 		this._needsRepaint = true;
239 		return this;
240 	};
241 
242 	this.__tick = function(dt) {
243 		this._tickFPS(dt);
244 		this.publish('Tick', dt);
245 		this._view.wrapTick(dt, this);
246 	};
247 
248 	var log_counter = 0;	
249 });
250 
251 exports.get = function() { return __instance; }
